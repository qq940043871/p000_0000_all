## 一、虚拟主机原理

```
一个服务器，多个域名：

server.xml:
  <Engine>
    <Host name="www.example.com" appBase="webapps/example" />
    <Host name="www.test.com" appBase="webapps/test" />
    <Host name="localhost" appBase="webapps" />
  </Engine>

DNS配置：
  www.example.com  A  192.168.1.100
  www.test.com     A  192.168.1.100
```

## 二、Host实现

```java
public class StandardHost extends ContainerBase implements Host {
    private String name;
    private String appBase;
    private String[] aliases;
    
    @Override
    public String getName() {
        return name;
    }
    
    @Override
    public void setName(String name) {
        this.name = name;
    }
    
    @Override
    public String getAppBase() {
        return appBase;
    }
    
    @Override
    public void setAppBase(String appBase) {
        this.appBase = appBase;
    }
    
    @Override
    public String[] getAliases() {
        return aliases;
    }
    
    @Override
    public void addAlias(String alias) {
        if (aliases == null) {
            aliases = new String[] { alias };
        } else {
            String[] newAliases = new String[aliases.length + 1];
            System.arraycopy(aliases, 0, newAliases, 0, aliases.length);
            newAliases[aliases.length] = alias;
            aliases = newAliases;
        }
    }
    
    public boolean matches(String hostName) {
        if (name.equals(hostName)) {
            return true;
        }
        
        if (aliases != null) {
            for (String alias : aliases) {
                if (alias.equals(hostName)) {
                    return true;
                }
            }
        }
        
        return false;
    }
}
```

## 三、Engine路由

```java
public class StandardEngine extends ContainerBase implements Engine {
    
    private class StandardEngineValve extends ValveBase {
        @Override
        public void invoke(Request request, Response response) throws IOException {
            // 获取Host头
            String hostName = request.getHeader("Host");
            if (hostName != null) {
                int colon = hostName.indexOf(':');
                if (colon > 0) {
                    hostName = hostName.substring(0, colon);
                }
            }
            
            // 查找匹配的Host
            Host host = null;
            for (Container container : findChildren()) {
                Host h = (Host) container;
                if (h.matches(hostName)) {
                    host = h;
                    break;
                }
            }
            
            // 使用默认Host
            if (host == null) {
                host = (Host) findChild(defaultHost);
            }
            
            if (host != null) {
                host.invoke(request, response);
            } else {
                response.sendError(404, "Host not found");
            }
        }
    }
}
```
