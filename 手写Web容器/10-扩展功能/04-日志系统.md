## 一、日志级别

```
TRACE < DEBUG < INFO < WARN < ERROR < FATAL

TRACE：最详细的日志
DEBUG：调试信息
INFO：一般信息
WARN：警告信息
ERROR：错误信息
FATAL：致命错误
```

## 二、日志实现

```java
public class Logger {
    private String name;
    private Level level = Level.INFO;
    private List<Handler> handlers = new ArrayList<>();
    
    public Logger(String name) {
        this.name = name;
    }
    
    public void trace(String msg) {
        log(Level.TRACE, msg);
    }
    
    public void debug(String msg) {
        log(Level.DEBUG, msg);
    }
    
    public void info(String msg) {
        log(Level.INFO, msg);
    }
    
    public void warn(String msg) {
        log(Level.WARN, msg);
    }
    
    public void error(String msg, Throwable e) {
        log(Level.ERROR, msg, e);
    }
    
    private void log(Level level, String msg) {
        log(level, msg, null);
    }
    
    private void log(Level level, String msg, Throwable e) {
        if (level.getValue() < this.level.getValue()) {
            return;
        }
        
        LogRecord record = new LogRecord(level, msg, e);
        
        for (Handler handler : handlers) {
            handler.publish(record);
        }
    }
    
    public void addHandler(Handler handler) {
        handlers.add(handler);
    }
}

public class LogRecord {
    private Level level;
    private String message;
    private Throwable exception;
    private long timestamp;
    private String loggerName;
    
    public LogRecord(Level level, String message, Throwable exception) {
        this.level = level;
        this.message = message;
        this.exception = exception;
        this.timestamp = System.currentTimeMillis();
    }
    
    public String format() {
        SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss");
        String time = sdf.format(new Date(timestamp));
        
        StringBuilder sb = new StringBuilder();
        sb.append(time).append(" [").append(level).append("] ");
        sb.append(message);
        
        if (exception != null) {
            sb.append("\n");
            StringWriter sw = new StringWriter();
            exception.printStackTrace(new PrintWriter(sw));
            sb.append(sw.toString());
        }
        
        return sb.toString();
    }
}

public class ConsoleHandler extends Handler {
    @Override
    public void publish(LogRecord record) {
        System.out.println(record.format());
    }
}

public class FileHandler extends Handler {
    private PrintWriter writer;
    
    public FileHandler(String filename) throws IOException {
        writer = new PrintWriter(new FileWriter(filename, true), true);
    }
    
    @Override
    public void publish(LogRecord record) {
        writer.println(record.format());
    }
}
```

## 三、使用示例

```java
public class MyServlet extends HttpServlet {
    private static final Logger logger = new Logger(MyServlet.class.getName());
    
    @Override
    protected void doGet(HttpServletRequest request, HttpServletResponse response) {
        logger.info("Received GET request: " + request.getRequestURI());
        
        try {
            // 处理请求
            response.getWriter().println("Hello World");
        } catch (Exception e) {
            logger.error("Error processing request", e);
        }
    }
}
```
