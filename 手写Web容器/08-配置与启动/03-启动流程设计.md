# 启动流程设计

> 模块：配置与启动
> 更新时间：2026-03-29

---

## 一、启动流程

```
1. 解析server.xml
2. 创建Server
3. 创建Service
4. 创建Connector
5. 创建Engine/Host/Context
6. 加载web.xml
7. 初始化Servlet
8. 启动Connector
9. 启动完成
```

---

## 二、Bootstrap启动类

```java
public class Bootstrap {
    private Server server;
    
    public void start() throws Exception {
        // 1. 解析server.xml
        ServerXmlParser parser = new ServerXmlParser();
        server = parser.parse("conf/server.xml");
        
        // 2. 初始化各个Service
        for (Service service : server.getServices()) {
            initService(service);
        }
        
        // 3. 启动各个Service
        for (Service service : server.getServices()) {
            service.start();
        }
        
        System.out.println("Server started successfully");
        
        // 4. 等待关闭信号
        waitForShutdown();
    }
    
    private void initService(Service service) throws Exception {
        Engine engine = service.getEngine();
        
        // 初始化所有Host
        for (Container host : engine.findChildren()) {
            initHost((Host) host);
        }
    }
    
    private void initHost(Host host) throws Exception {
        // 初始化所有Context
        for (Container context : host.findChildren()) {
            initContext((Context) context);
        }
    }
    
    private void initContext(Context context) throws Exception {
        // 1. 创建ClassLoader
        WebAppClassLoader loader = new WebAppClassLoader(
            Thread.currentThread().getContextClassLoader(), context
        );
        context.setClassLoader(loader);
        
        // 2. 解析web.xml
        String webXmlPath = context.getDocBase() + "/WEB-INF/web.xml";
        WebXmlParser parser = new WebXmlParser();
        parser.parse(context, webXmlPath);
        
        // 3. 初始化Servlet
        for (Container wrapper : context.findChildren()) {
            ((Wrapper) wrapper).allocate();
        }
        
        System.out.println("Context initialized: " + context.getPath());
    }
    
    private void waitForShutdown() throws IOException {
        BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
        String line;
        while ((line = reader.readLine()) != null) {
            if ("shutdown".equals(line)) {
                break;
            }
        }
        
        shutdown();
    }
    
    public void shutdown() throws Exception {
        System.out.println("Shutting down server...");
        
        for (Service service : server.getServices()) {
            service.stop();
        }
        
        System.out.println("Server stopped");
    }
    
    public static void main(String[] args) throws Exception {
        Bootstrap bootstrap = new Bootstrap();
        bootstrap.start();
    }
}
```
