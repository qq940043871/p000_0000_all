## 一、性能分析器

```cpp
class Profiler {
public:
    struct ProfileBlock {
        std::string name;
        double startTime;
        double duration;
        int threadId;
    };
    
    std::vector<ProfileBlock> blocks;
    
    void beginBlock(const std::string& name) {
        ProfileBlock block;
        block.name = name;
        block.startTime = getTime();
        block.threadId = getCurrentThreadId();
        blocks.push_back(block);
    }
    
    void endBlock() {
        ProfileBlock& block = blocks.back();
        block.duration = getTime() - block.startTime;
    }
    
    void report() {
        // 按时间排序
        std::sort(blocks.begin(), blocks.end(), 
            [](const auto& a, const auto& b) {
                return a.duration > b.duration;
            });
        
        // 打印报告
        for (auto& block : blocks) {
            printf("%s: %.2fms\n", block.name.c_str(), block.duration * 1000);
        }
    }
};

#define PROFILE_SCOPE(name) ProfilerBlock _profile(name, __FUNCTION__)
```

## 二、帧时间统计

```cpp
class FrameCounter {
private:
    int frameCount = 0;
    double totalTime = 0;
    double minFrameTime = 1e10;
    double maxFrameTime = 0;
    
public:
    void recordFrame(double frameTime) {
        frameCount++;
        totalTime += frameTime;
        minFrameTime = std::min(minFrameTime, frameTime);
        maxFrameTime = std::max(maxFrameTime, frameTime);
        
        if (frameCount % 60 == 0) {
            double avgFrameTime = totalTime / frameCount;
            LOG("FPS: %.1f, Avg: %.2fms, Min: %.2fms, Max: %.2fms",
                1.0 / avgFrameTime,
                avgFrameTime * 1000,
                minFrameTime * 1000,
                maxFrameTime * 1000);
            
            frameCount = 0;
            totalTime = 0;
            minFrameTime = 1e10;
            maxFrameTime = 0;
        }
    }
};
```
