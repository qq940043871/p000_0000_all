## 一、实体操作

```cpp
class Entity {
public:
    EntityID id;
    std::string name;
    bool active = true;
    std::vector<Component*> components;
    
    template<typename T>
    T* addComponent() {
        T* comp = new T();
        comp->entity = this;
        components.push_back(comp);
        return comp;
    }
    
    template<typename T>
    T* getComponent() {
        for (auto* comp : components) {
            if (T* t = dynamic_cast<T*>(comp)) {
                return t;
            }
        }
        return nullptr;
    }
    
    template<typename T>
    bool hasComponent() {
        return getComponent<T>() != nullptr;
    }
    
    template<typename T>
    void removeComponent() {
        for (auto it = components.begin(); it != components.end(); ++it) {
            if (dynamic_cast<T*>(*it)) {
                delete *it;
                components.erase(it);
                return;
            }
        }
    }
};
```
