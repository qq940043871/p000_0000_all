## 一、系统基类

```cpp
class ISystem {
public:
    virtual ~ISystem() = default;
    virtual void update(ECSWorld* world, double dt) = 0;
};

// 运动系统
class MovementSystem : public ISystem {
public:
    void update(ECSWorld* world, double dt) override {
        // 查找有Transform和RigidBody的实体
        for (auto entity : world->getEntitiesWith<TransformComponent, RigidBodyComponent>()) {
            auto* transform = world->getComponent<TransformComponent>(entity);
            auto* rigidbody = world->getComponent<RigidBodyComponent>(entity);
            
            // 更新位置
            transform->position += rigidbody->velocity * dt;
            transform->rotation *= Quaternion::fromAngularVelocity(rigidbody->angularVelocity * dt);
        }
    }
};

// 渲染系统
class RenderSystem : public ISystem {
private:
    Renderer* renderer;
    
public:
    RenderSystem(Renderer* renderer) : renderer(renderer) {}
    
    void update(ECSWorld* world, double dt) override {
        for (auto entity : world->getEntitiesWith<TransformComponent, RenderComponent>()) {
            auto* transform = world->getComponent<TransformComponent>(entity);
            auto* render = world->getComponent<RenderComponent>(entity);
            
            renderer->submit(render->mesh, transform->getMatrix(), render->material);
        }
    }
};
```

## 二、系统管理器

```cpp
class SystemManager {
private:
    std::vector<ISystem*> systems;
    std::unordered_map<std::type_index, ISystem*> systemMap;
    
public:
    template<typename T>
    T* addSystem() {
        T* system = new T();
        systems.push_back(system);
        systemMap[typeid(T)] = system;
        return system;
    }
    
    template<typename T>
    T* getSystem() {
        return dynamic_cast<T*>(systemMap[typeid(T)]);
    }
    
    void update(ECSWorld* world, double dt) {
        for (auto* system : systems) {
            system->update(world, dt);
        }
    }
    
    // 按优先级排序
    void setPriority<ISystem*>(ISystem* system, int priority);
};
```
