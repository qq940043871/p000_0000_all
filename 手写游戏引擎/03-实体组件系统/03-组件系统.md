## 一、常用组件

```cpp
// 变换组件
struct TransformComponent {
    Vector3 position = Vector3::zero();
    Quaternion rotation = Quaternion::identity();
    Vector3 scale = Vector3::one();
    
    Matrix4x4 getMatrix() const {
        return Matrix4x4::translate(position) *
               rotation.toMatrix() *
               Matrix4x4::scale(scale);
    }
};

// 渲染组件
struct RenderComponent {
    Mesh* mesh = nullptr;
    Material* material = nullptr;
    int layer = 0;
    bool castShadow = true;
    bool receiveShadow = true;
};

// 物理组件
struct RigidBodyComponent {
    float mass = 1.0f;
    Vector3 velocity = Vector3::zero();
    Vector3 angularVelocity = Vector3::zero();
    float linearDamping = 0.01f;
    float angularDamping = 0.01f;
    bool isKinematic = false;
};

// 碰撞组件
struct ColliderComponent {
    ColliderShape* shape = nullptr;
    PhysicsMaterial* material = nullptr;
    bool isTrigger = false;
    std::vector<std::string> layers;
};

// 脚本组件
struct ScriptComponent {
    IScript* script = nullptr;
};
```

## 二、组件模板

```cpp
// 可序列化的组件
template<typename T>
struct SerializableComponent : public IComponent {
    static ComponentType getStaticType() { 
        return ComponentTypeRegistry::get<T>(); 
    }
    
    virtual void serialize(Serializer& serializer) = 0;
    virtual void deserialize(Serializer& serializer) = 0;
};

// 变换组件定义
struct Transform : public SerializableComponent<Transform> {
    Vector3 localPosition;
    Quaternion localRotation;
    Vector3 localScale;
    
    Vector3 worldPosition;
    Quaternion worldRotation;
    
    void serialize(Serializer& s) override {
        s.write("position", localPosition);
        s.write("rotation", localRotation);
        s.write("scale", localScale);
    }
    
    void deserialize(Serializer& s) override {
        s.read("position", localPosition);
        s.read("rotation", localRotation);
        s.read("scale", localScale);
    }
};
```
