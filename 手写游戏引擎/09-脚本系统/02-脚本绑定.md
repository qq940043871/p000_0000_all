## 一、游戏对象绑定

```cpp
void ScriptBinding::bindGameObject(sol::state& lua) {
    lua.new_usertype<GameObject>("GameObject",
        "getName", &GameObject::getName,
        "setName", &GameObject::setName,
        "getTransform", &GameObject::getTransform,
        "findByName", [](const std::string& name) {
            return SceneManager::getCurrentScene()->findByName(name);
        },
        "addComponent", [](GameObject* obj, const std::string& type) {
            // 根据type添加组件
        },
        "getComponent", [](GameObject* obj, const std::string& type) {
            return obj->getComponent(type);
        },
        "destroy", &GameObject::Destroy
    );
}
```

## 二、完整绑定示例

```cpp
class ScriptManager {
private:
    sol::state lua;
    
public:
    void initialize() {
        // 绑定基础类型
        ScriptBinding::bindMath(lua);
        ScriptBinding::bindTransform(lua);
        ScriptBinding::bindGameObject(lua);
        
        // 绑定输入
        ScriptBinding::bindInput(lua);
        
        // 绑定资源
        ScriptBinding::bindResources(lua);
        
        // 设置全局变量
        lua["deltaTime"] = 0.0f;
    }
    
    void executeScript(const std::string& script) {
        try {
            lua.script(script);
        } catch (const sol::error& e) {
            LOG_ERROR("Lua Error: %s", e.what());
        }
    }
};
```
