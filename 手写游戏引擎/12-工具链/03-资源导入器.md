## 一、导入管线

```cpp
class AssetImporter {
private:
    std::unordered_map<std::string, IImporter*> importers;
    
public:
    void initialize() {
        importers[".fbx"] = new FBXImporter();
        importers[".obj"] = new OBJImporter();
        importers[".png"] = new TextureImporter();
        importers[".wav"] = new AudioImporter();
        importers[".mat"] = new MaterialImporter();
    }
    
    void import(const std::string& sourcePath) {
        std::string ext = getExtension(sourcePath);
        
        auto it = importers.find(ext);
        if (it != importers.end()) {
            Asset* asset = it->second->import(sourcePath);
            
            // 保存到目标位置
            saveAsset(asset);
            
            // 生成元数据
            generateMeta(asset);
        }
    }
};

class FBXImporter : public IImporter {
public:
    Asset* import(const std::string& path) override {
        FBXDocument doc;
        FBXParser::parse(path, &doc);
        
        // 提取网格
        Mesh* mesh = extractMesh(doc);
        
        // 提取材质
        std::vector<Material*> materials = extractMaterials(doc);
        
        // 提取动画
        std::vector<AnimationClip*> animations = extractAnimations(doc);
        
        return new ModelAsset(mesh, materials, animations);
    }
};
```
