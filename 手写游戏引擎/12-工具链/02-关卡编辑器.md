## 一、场景编辑

```cpp
class LevelEditor {
private:
    std::unique_ptr<Gizmo> transformGizmo;
    ToolMode currentTool = ToolMode::SELECT;
    
public:
    void onMouseDown(const Vector2& mousePos) {
        switch (currentTool) {
            case ToolMode::SELECT:
                selectObject(mousePos);
                break;
            case ToolMode::MOVE:
                beginTranslate(mousePos);
                break;
            case ToolMode::ROTATE:
                beginRotate(mousePos);
                break;
            case ToolMode::SCALE:
                beginScale(mousePos);
                break;
            case ToolMode::PLACE:
                placeObject(mousePos);
                break;
        }
    }
    
    void selectObject(const Vector2& mousePos) {
        Ray ray = camera->screenPointToRay(mousePos);
        GameObject* selected = raycast(ray);
        if (selected) {
            editorSelection.select(selected);
        }
    }
    
    void placeObject(const Vector2& mousePos) {
        Ray ray = camera->screenPointToRay(mousePos);
        Vector3 point = rayIntersectGround(ray);
        
        GameObject* newObj = prefabManager->instantiate(prefabToPlace);
        newObj->getTransform()->setPosition(point);
        SceneManager::getCurrentScene()->addGameObject(newObj);
    }
};
```
