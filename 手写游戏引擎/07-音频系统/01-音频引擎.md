## 一、音频系统

```cpp
class AudioEngine : public IModule {
private:
   ALCcontext* context;
    ALCdevice* device;
    std::unordered_map<std::string, AudioBuffer*> buffers;
    std::vector<AudioSource*> activeSources;
    
public:
    void init() override {
        device = alcOpenDevice(nullptr);
        context = alcCreateContext(device, nullptr);
        alcMakeContextCurrent(context);
    }
    
    AudioBuffer* loadSound(const std::string& path) {
        auto it = buffers.find(path);
        if (it != buffers.end()) return it->second;
        
        AudioBuffer* buffer = new AudioBuffer();
        buffer->load(path);
        buffers[path] = buffer;
        return buffer;
    }
    
    AudioSource* play(const std::string& sound, bool loop = false) {
        AudioBuffer* buffer = loadSound(sound);
        AudioSource* source = new AudioSource();
        source->setBuffer(buffer);
        source->setLooping(loop);
        source->play();
        activeSources.push_back(source);
        return source;
    }
};
```

## 二、音频源

```cpp
class AudioSource {
private:
    ALuint sourceID;
    AudioBuffer* buffer;
    
public:
    void play() {
        alSourcePlay(sourceID);
    }
    
    void pause() {
        alSourcePause(sourceID);
    }
    
    void stop() {
        alSourceStop(sourceID);
    }
    
    void setVolume(float volume) {
        alSourcef(sourceID, AL_GAIN, volume);
    }
    
    void setPitch(float pitch) {
        alSourcef(sourceID, AL_PITCH, pitch);
    }
    
    void setLooping(bool loop) {
        alSourcei(sourceID, AL_LOOPING, loop ? AL_TRUE : AL_FALSE);
    }
    
    void setPosition(const Vector3& pos) {
        alSource3f(sourceID, AL_POSITION, pos.x, pos.y, pos.z);
    }
    
    void setVelocity(const Vector3& vel) {
        alSource3f(sourceID, AL_VELOCITY, vel.x, vel.y, vel.z);
    }
};
```
