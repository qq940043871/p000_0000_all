## 一、空间音频

```cpp
class SpatialAudio {
private:
    AudioListener* listener;
    
public:
    void updateListener(const Vector3& position, 
                      const Vector3& velocity,
                      const Vector3& forward,
                      const Vector3& up) {
        alListener3f(AL_POSITION, position.x, position.y, position.z);
        alListener3f(AL_VELOCITY, velocity.x, velocity.y, velocity.z);
        
        float orientation[] = {forward.x, forward.y, forward.z, up.x, up.y, up.z};
        alListenerfv(AL_ORIENTATION, orientation);
    }
};

class SpatialSource : public AudioSource {
private:
    float refDistance = 1.0f;   // 参考距离
    float maxDistance = 100.0f; // 最大距离
    float rolloff = 1.0f;       // 衰减率
    
public:
    void setPosition(const Vector3& pos) override {
        AudioSource::setPosition(pos);
        alSourcef(sourceID, AL_REFERENCE_DISTANCE, refDistance);
        alSourcef(sourceID, AL_MAX_DISTANCE, maxDistance);
        alSourcef(sourceID, AL_ROLLOFF_FACTOR, rolloff);
    }
};
```

## 二、距离衰减

```cpp
float calculateVolume(float distance, float refDist, float maxDist, float rolloff) {
    if (distance < refDist) return 1.0f;
    if (distance > maxDist) return 0.0f;
    
    // 线性衰减
    float volume = 1.0f - (distance - refDist) / (maxDist - refDist) * rolloff;
    return std::max(0.0f, volume);
}

// 对数衰减（更自然）
float calculateVolumeLog(float distance, float refDist, float rolloff) {
    if (distance < refDist) return 1.0f;
    
    float volume = refDist / (refDist + rolloff * (distance - refDist));
    return std::max(0.0f, std::min(1.0f, volume));
}
```
