## 二、音效管理

```cpp
class SoundEffectPool {
private:
    ObjectPool<AudioSource> sourcePool;
    std::vector<AudioSource*> activeSources;
    
public:
    void play(const std::string& path, float volume = 1.0f) {
        AudioSource* source = sourcePool.acquire();
        source->setBuffer(AudioEngine::getInstance()->loadSound(path));
        source->setVolume(volume);
        source->play();
        
        // 检查是否播放完成
        activeSources.push_back(source);
    }
    
    void update() {
        for (auto it = activeSources.begin(); it != activeSources.end(); ) {
            if (isStopped(*it)) {
                sourcePool.release(*it);
                it = activeSources.erase(it);
            } else {
                ++it;
            }
        }
    }
};
```
