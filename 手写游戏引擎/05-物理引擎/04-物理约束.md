## 一、约束基类

```cpp
class IConstraint {
public:
    virtual ~IConstraint() = default;
    virtual void solve(double dt) = 0;
};

// 铰链约束
class HingeConstraint : public IConstraint {
private:
    RigidBody* bodyA;
    RigidBody* bodyB;
    Vector3 anchorA;
    Vector3 anchorB;
    Vector3 axis;  // 旋转轴
    
public:
    void solve(double dt) override {
        // 计算约束满足条件
        // 应用修正力
    }
};

// 距离约束
class DistanceConstraint : public IConstraint {
private:
    RigidBody* bodyA;
    RigidBody* bodyB;
    Vector3 anchorA;
    Vector3 anchorB;
    float targetDistance;
    
public:
    void solve(double dt) override {
        Vector3 delta = bodyB->position + anchorB - bodyA->position - anchorA;
        float distance = delta.length();
        
        if (abs(distance - targetDistance) < 0.001f) return;
        
        Vector3 normal = delta / distance;
        float correction = (distance - targetDistance) * 0.8f;
        
        if (!bodyA->isKinematic) {
            bodyA->position += normal * correction * 0.5f;
        }
        if (!bodyB->isKinematic) {
            bodyB->position -= normal * correction * 0.5f;
        }
    }
};

// 弹簧约束
class SpringConstraint : public IConstraint {
private:
    RigidBody* bodyA;
    RigidBody* bodyB;
    float restLength;
    float stiffness;
    float damping;
    
public:
    void solve(double dt) override {
        Vector3 delta = bodyB->position - bodyA->position;
        float distance = delta.length();
        
        Vector3 direction = delta / distance;
        float stretch = distance - restLength;
        
        Vector3 force = direction * (stiffness * stretch + damping * 
              dot(bodyB->velocity - bodyA->velocity, direction));
        
        bodyA->applyForce(force);
        bodyB->applyForce(-force);
    }
};
```
