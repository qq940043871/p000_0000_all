```cpp
class ColliderShape {
public:
    virtual ~ColliderShape() = default;
    virtual ColliderType getType() const = 0;
    virtual AABB getAABB(const Transform& transform) const = 0;
    virtual bool intersect(const ColliderShape* other, 
                          const Transform& t1,
                          const Transform& t2,
                          Manifold* manifold) const = 0;
};

class SphereCollider : public ColliderShape {
public:
    float radius;
    
    ColliderType getType() const { return ColliderType::SPHERE; }
    
    AABB getAABB(const Transform& transform) const override {
        Vector3 center = transform.position;
        return AABB(center - radius, center + radius);
    }
};
```

```cpp
class CollisionDetection {
public:
    bool checkSphereSphere(const SphereCollider* a, const Transform& ta,
                           const SphereCollider* b, const Transform& tb,
                           Manifold* manifold) {
        Vector3 posA = ta.position;
        Vector3 posB = tb.position;
        
        Vector3 diff = posA - posB;
        float distSq = diff.lengthSquared();
        float radiusSum = a->radius + b->radius;
        
        if (distSq > radiusSum * radiusSum) return false;
        
        float dist = sqrt(distSq);
        
        manifold->normal = diff / dist;
        manifold->penetration = radiusSum - dist;
        manifold->contactPoints[0] = posA - manifold->normal * a->radius;
        
        return true;
    }
```

## 三、Broad-Phase与Narrow-Phase

```cpp
class CollisionSystem {
private:
    // Broad-Phase: AABB检测
    std::vector<std::pair<int, int>> broadPhase() {
        std::vector<std::pair<int, int>> pairs;
        
        for (size_t i = 0; i < colliders.size(); i++) {
            for (size_t j = i + 1; j < colliders.size(); j++) {
                if (colliders[i]->getAABB().intersects(colliders[j]->getAABB())) {
                    pairs.push_back({i, j});
                }
            }
        }
        
        return pairs;
    }
    
    // Narrow-Phase: 精确碰撞检测
    std::vector<Manifold> narrowPhase(const std::vector<std::pair<int, int>>& pairs) {
        std::vector<Manifold> manifolds;
        
        for (auto& [i, j] : pairs) {
            Manifold m;
            if (detect(colliders[i], transforms[i],
                      colliders[j], transforms[j], &m)) {
                manifolds.push_back(m);
            }
        }
        
        return manifolds;
    }
};
```
