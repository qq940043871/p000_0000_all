## 一、刚体

```cpp
class RigidBody {
private:
    float mass;
    float invMass;
    Matrix3x3 inertiaTensor;
    Matrix3x3 invInertiaTensor;
    
    Vector3 position;
    Quaternion rotation;
    Vector3 velocity;
    Vector3 angularVelocity;
    
    Vector3 force;
    Vector3 torque;
    
    bool isKinematic;
    
public:
    void applyForce(const Vector3& f) {
        force += f;
    }
    
    void applyForceAtPoint(const Vector3& f, const Vector3& point) {
        force += f;
        torque += cross(point - position, f);
    }
    
    void applyImpulse(const Vector3& impulse, const Vector3& contactPoint) {
        velocity += impulse * invMass;
        angularVelocity += invInertiaTensor * cross(contactPoint - position, impulse);
    }
    
    void integrate(double dt) {
        if (isKinematic) return;
        
        // 线性运动
        Vector3 acceleration = force * invMass;
        velocity += acceleration * dt;
        position += velocity * dt;
        
        // 旋转运动
        Vector3 angularAcceleration = invInertiaTensor * torque;
        angularVelocity += angularAcceleration * dt;
        rotation = normalize(rotation + rotation * Quaternion::fromAngularVelocity(angularVelocity * dt));
        
        // 阻尼
        velocity *= 1.0f / (1.0f + linearDamping * dt);
        angularVelocity *= 1.0f / (1.0f + angularDamping * dt);
        
        // 清除力
        force = Vector3::zero();
        torque = Vector3::zero();
    }
};
```

## 二、碰撞响应

```cpp
void resolveCollision(RigidBody* a, RigidBody* b, Manifold* manifold) {
    // 分离轴法线
    Vector3 normal = manifold->normal;
    Vector3 contact = manifold->contactPoints[0];
    
    // 相对速度
    Vector3 relVel = b->velocity - a->velocity;
    float velAlongNormal = dot(relVel, normal);
    
    if (velAlongNormal > 0) return;  // 已经在分离
    
    // 恢复系数
    float e = std::min(a->restitution, b->restitution);
    
    // 反作用冲量
    Vector3 j = -(1 + e) * velAlongNormal;
    j /= a->invMass + b->invMass;
    
    Vector3 impulse = j * normal;
    
    // 应用冲量
    a->applyImpulse(-impulse, contact);
    b->applyImpulse(impulse, contact);
    
    // 摩擦力
    Vector3 tangent = relVel - velAlongNormal * normal;
    float jt = -dot(relVel, tangent);
    jt /= a->invMass + b->invMass;
    
    float mu = (a->friction + b->friction) * 0.5f;
    Vector3 frictionImpulse;
    
    if (abs(jt) < j * mu) {
        frictionImpulse = jt * tangent;
    } else {
        frictionImpulse = -j * mu * tangent;
    }
    
    a->applyImpulse(-frictionImpulse, contact);
    b->applyImpulse(frictionImpulse, contact);
}
```
