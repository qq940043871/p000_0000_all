## 一、物理世界

```cpp
class PhysicsWorld {
private:
    std::vector<RigidBody*> bodies;
    std::vector<Collider*> colliders;
    Vector3 gravity = Vector3(0, -9.81f, 0);
    
    // 碰撞检测
    CollisionSystem* collisionSystem;
    
    // 约束求解
    ConstraintSolver* constraintSolver;
    
    // 配置
    PhysicsConfig config;
    
public:
    void step(double dt) {
        // 1. 应用力
        applyForces();
        
        // 2. 碰撞检测
        std::vector<Manifold> manifolds = collisionSystem->detect(bodies);
        
        // 3. 碰撞响应
        resolveCollisions(manifolds);
        
        // 4. 积分
        integrate(dt);
        
        // 5. 约束求解
        constraintSolver->solve(dt);
        
        // 6. 清理
        removeMarkedBodies();
    }
    
    void addBody(RigidBody* body) {
        bodies.push_back(body);
        if (body->getCollider()) {
            colliders.push_back(body->getCollider());
        }
    }
    
    void removeBody(RigidBody* body) {
        bodies.erase(std::remove(bodies.begin(), bodies.end(), body), bodies.end());
        if (body->getCollider()) {
            colliders.erase(std::remove(colliders.begin(), colliders.end(), 
                                       body->getCollider()), colliders.end());
        }
    }
};
```

## 二、时间步长

```cpp
class PhysicsTimestepper {
private:
    const double FIXED_DT = 1.0 / 120.0;  // 固定120Hz
    double accumulator = 0;
    
public:
    void step(PhysicsWorld* world, double dt) {
        accumulator += dt;
        
        int maxSteps = 4;  // 防止多帧不同步
        int steps = 0;
        
        while (accumulator >= FIXED_DT && steps < maxSteps) {
            world->fixedUpdate(FIXED_DT);
            accumulator -= FIXED_DT;
            steps++;
        }
        
        if (accumulator > FIXED_DT) {
            accumulator = 0;  // 防止累积
        }
    }
};
```
