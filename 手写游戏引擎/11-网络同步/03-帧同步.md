## 一、帧同步实现

```cpp
class FrameSync {
private:
    int localFrame = 0;
    int serverFrame = 0;
    std::vector<InputState> pendingInputs;
    
public:
    void sendInput(const InputState& input) {
        input.frame = localFrame;
        input.playerId = playerId;
        send(Channel::INPUT, &input, sizeof(input));
    }
    
    void onReceiveInput(const uint8_t* data, size_t size) {
        InputState input;
        memcpy(&input, data, sizeof(input));
        
        pendingInputs.push_back(input);
        
        // 等待所有玩家输入到达
        if (allInputsReceived(serverFrame)) {
            applyFrame(serverFrame);
            serverFrame++;
        }
    }
    
    void processFrame() {
        // 收集本地输入
        InputState input = gatherInput();
        input.frame = localFrame;
        sendInput(input);
        
        // 应用所有帧输入
        while (!pendingInputs.empty() && 
               pendingInputs[0].frame <= localFrame) {
            applyInput(pendingInputs[0]);
            pendingInputs.erase(pendingInputs.begin());
        }
        
        localFrame++;
    }
};
```
