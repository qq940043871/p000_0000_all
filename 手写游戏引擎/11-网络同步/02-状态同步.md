## 一、状态同步

```cpp
struct NetworkTransform {
    uint32_t objectId;
    Vector3 position;
    Quaternion rotation;
    Vector3 velocity;
    uint16_t sequenceNumber;
};

class StateSynchronizer {
private:
    std::unordered_map<uint32_t, NetworkTransform> lastReceived;
    std::unordered_map<uint32_t, TransformComponent*> localObjects;
    
public:
    void sendUpdate(NetworkTransform& transform) {
        transform.sequenceNumber = nextSequence++;
        send(Channel::STATE, &transform, sizeof(transform));
    }
    
    void onReceive(const uint8_t* data, size_t size) {
        NetworkTransform received;
        memcpy(&received, data, sizeof(NetworkTransform));
        
        auto it = lastReceived.find(received.objectId);
        if (it == lastReceived.end() || 
            received.sequenceNumber > it->second.sequenceNumber) {
            lastReceived[received.objectId] = received;
            applyState(received);
        }
    }
    
    void applyState(const NetworkTransform& state) {
        TransformComponent* transform = localObjects[state.objectId];
        if (transform) {
            // 插值平滑
            transform->position = state.position;
            transform->rotation = state.rotation;
        }
    }
};
```
