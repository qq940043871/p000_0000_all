## 一、模块接口

```cpp
class IModule {
public:
    virtual ~IModule() = default;
    
    virtual void init() = 0;
    virtual void update(double dt) = 0;
    virtual void destroy() = 0;
    virtual const char* getName() const = 0;
};

class ModuleManager {
private:
    std::vector<IModule*> modules;
    std::unordered_map<std::string, IModule*> moduleMap;
    
public:
    template<typename T>
    T* getModule() {
        return dynamic_cast<T*>(moduleMap[typeid(T).name()]);
    }
    
    void registerModule(IModule* module) {
        modules.push_back(module);
        moduleMap[module->getName()] = module;
    }
    
    void initAll() {
        for (auto* module : modules) {
            module->init();
        }
    }
    
    void updateAll(double dt) {
        for (auto* module : modules) {
            module->update(dt);
        }
    }
    
    void destroyAll() {
        for (auto* module : modules) {
            module->destroy();
            delete module;
        }
    }
};
```

## 二、引擎模块

```cpp
class Engine : public IModule {
public:
    const char* getName() const { return "Engine"; }
    
    void init() override {
        LOG("Initializing Engine...");
        
        ModuleManager::getInstance()->registerModule(new InputManager());
        ModuleManager::getInstance()->registerModule(new ResourceManager());
        ModuleManager::getInstance()->registerModule(new PhysicsEngine());
        ModuleManager::getInstance()->registerModule(new SceneManager());
        ModuleManager::getInstance()->registerModule(new Renderer());
        ModuleManager::getInstance()->registerModule(new AudioEngine());
        ModuleManager::getInstance()->registerModule(new ScriptManager());
        
        ModuleManager::getInstance()->initAll();
    }
    
    void update(double dt) override {
        ModuleManager::getInstance()->updateAll(dt);
    }
    
    void destroy() override {
        ModuleManager::getInstance()->destroyAll();
    }
};
```
