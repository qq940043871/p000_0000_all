## 一、对象池

```cpp
template<typename T>
class ObjectPool {
private:
    std::vector<T*> available;
    std::vector<T*> inUse;
    std::function<T*()> creator;
    
public:
    ObjectPool(size_t initialSize, std::function<T*()> creator) {
        this->creator = creator;
        for (size_t i = 0; i < initialSize; ++i) {
            available.push_back(creator());
        }
    }
    
    T* acquire() {
        T* obj;
        if (available.empty()) {
            obj = creator();
        } else {
            obj = available.back();
            available.pop_back();
        }
        inUse.push_back(obj);
        return obj;
    }
    
    void release(T* obj) {
        auto it = std::find(inUse.begin(), inUse.end(), obj);
        if (it != inUse.end()) {
            inUse.erase(it);
            available.push_back(obj);
        }
    }
    
    void clear() {
        available.insert(available.end(), inUse.begin(), inUse.end());
        inUse.clear();
    }
};

// 游戏对象池
ObjectPool<GameObject>* gameObjectPool;
ObjectPool<Particle>* particlePool;
ObjectPool<Projectile>* projectilePool;
```

## 二、智能指针集成

```cpp
class MemoryManager {
private:
    std::unordered_map<void*, MemoryInfo> allocations;
    size_t totalAllocated = 0;
    size_t totalFreed = 0;
    
public:
    void* allocate(size_t size, const char* file, int line) {
        void* ptr = malloc(size);
        
        MemoryInfo info;
        info.size = size;
        info.file = file;
        info.line = line;
        info.time = TimeManager::getTotalTime();
        
        allocations[ptr] = info;
        totalAllocated += size;
        
        return ptr;
    }
    
    void deallocate(void* ptr) {
        auto it = allocations.find(ptr);
        if (it != allocations.end()) {
            totalFreed += it->second.size;
            allocations.erase(it);
        }
        free(ptr);
    }
    
    void report() {
        LOG("Total Allocated: %zu bytes", totalAllocated);
        LOG("Total Freed: %zu bytes", totalFreed);
        LOG("Current: %zu bytes", totalAllocated - totalFreed);
    }
};

#define NEW(type) new (MemoryManager::allocate(sizeof(type), __FILE__, __LINE__)) type
#define DELETE(ptr) if (ptr) { ptr->~decltype(*ptr); MemoryManager::deallocate(ptr); }
```

## 三、内存对齐

```cpp
// 16字节对齐的分配器
class AlignedAllocator {
public:
    static void* allocateAligned(size_t size, size_t alignment) {
        void* ptr;
        #ifdef _WIN32
        ptr = _aligned_malloc(size, alignment);
        #else
        posix_memalign(&ptr, alignment, size);
        #endif
        return ptr;
    }
    
    static void deallocateAligned(void* ptr) {
        #ifdef _WIN32
        _aligned_free(ptr);
        #else
        free(ptr);
        #endif
    }
};

// SIMD友好的向量
struct alignas(16) Vec4 {
    float x, y, z, w;
};
```
