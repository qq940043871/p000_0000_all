## 一、游戏循环原理

```
游戏循环核心：
  while (running) {
    processInput();     // 处理输入
    update(deltaTime);  // 更新逻辑
    render();           // 渲染画面
    present();           // 交换缓冲区
  }

帧率控制：
  1. 固定时间步长（Fixed Timestep）
     - 物理更新用固定dt
     - 插值平滑显示
     
  2. 可变时间步长（Variable Timestep）
     - 使用实际流逝时间
     - 简单但不稳定
  
  3. 半固定步长（Semi-Fixed）
     - 主循环可变
     - 物理逻辑固定
```

## 二、C++实现

```cpp
class GameEngine {
private:
    bool running = true;
    bool minimized = false;
    
    // 帧率控制
    const double FIXED_DT = 1.0 / 60.0;  // 固定60FPS
    double accumulator = 0.0;
    double currentTime = 0.0;
    
    // 模块
    InputManager* inputManager;
    PhysicsEngine* physicsEngine;
    SceneManager* sceneManager;
    Renderer* renderer;
    AudioEngine* audioEngine;
    
public:
    void run() {
        double lastTime = getCurrentTime();
        
        while (running) {
            double newTime = getCurrentTime();
            double frameTime = newTime - lastTime;
            lastTime = newTime;
            
            // 限制最大帧时间，防止螺旋效应
            if (frameTime > 0.25) {
                frameTime = 0.25;
            }
            
            accumulator += frameTime;
            
            // 输入处理
            processInput();
            
            // 固定时间步长的逻辑更新
            while (accumulator >= FIXED_DT) {
                fixedUpdate(FIXED_DT);
                accumulator -= FIXED_DT;
            }
            
            // 可变时间步长的渲染
            double alpha = accumulator / FIXED_DT;
            render(alpha);
            
            // 渲染后处理
            postRender();
        }
    }
    
    void fixedUpdate(double dt) {
        // 物理更新
        physicsEngine->update(dt);
        
        // 脚本更新
        ScriptManager::getInstance().update(dt);
        
        // 动画更新
        AnimationManager::getInstance().update(dt);
        
        // 音频更新
        audioEngine->update(dt);
    }
    
    void processInput() {
        inputManager->pollEvents();
        
        // 处理退出
        if (inputManager->isKeyPressed(KEY_ESCAPE)) {
            running = false;
        }
    }
    
    void render(double alpha) {
        // 插值渲染
        sceneManager->render(alpha);
    }
};
```

## 三、暂停与恢复

```cpp
class GameLoopControl {
private:
    bool paused = false;
    bool stepped = false;
    double pausedTime = 0;
    
public:
    void pause() {
        paused = true;
        pausedTime = getCurrentTime();
    }
    
    void resume() {
        paused = false;
    }
    
    void step() {
        if (paused) {
            stepped = true;
        }
    }
    
    bool shouldUpdate() {
        if (!paused) return true;
        if (stepped) {
            stepped = false;
            return true;
        }
        return false;
    }
};
```
