## 一、时间管理器

```cpp
class TimeManager {
private:
    double deltaTime = 0;
    double unscaledDeltaTime = 0;
    double timeScale = 1.0;
    double totalTime = 0;
    double realtimeSinceStartup = 0;
    int frameCount = 0;
    float fps = 0;
    float fpsUpdateInterval = 0.5f;
    float fpsAccumulator = 0;
    
public:
    void update(double realDelta) {
        unscaledDeltaTime = realDelta;
        deltaTime = realDelta * timeScale;
        totalTime += deltaTime;
        realtimeSinceStartup += realDelta;
        frameCount++;
        
        // FPS计算
        fpsAccumulator += realDelta;
        if (fpsAccumulator >= fpsUpdateInterval) {
            fps = frameCount / fpsAccumulator;
            frameCount = 0;
            fpsAccumulator = 0;
        }
    }
    
    // 时间缩放（用于慢动作/快进效果）
    void setTimeScale(float scale) {
        timeScale = scale;
    }
    
    // 游戏暂停时的时间
    void pauseTime() {
        // 记录暂停时刻
    }
};
```

## 二、定时器系统

```cpp
class Timer {
private:
    std::string name;
    double duration;
    double elapsed = 0;
    bool looping = false;
    std::function<void()> callback;
    bool completed = false;
    
public:
    void update(double dt) {
        if (completed) return;
        
        elapsed += dt;
        if (elapsed >= duration) {
            if (looping) {
                elapsed = 0;
                callback();
            } else {
                completed = true;
                callback();
            }
        }
    }
    
    void reset() {
        elapsed = 0;
        completed = false;
    }
    
    float getProgress() const {
        return elapsed / duration;
    }
};

class TimerManager {
private:
    std::vector<Timer*> timers;
    
public:
    Timer* createTimer(const std::string& name, double duration,
                       std::function<void()> callback, bool looping = false) {
        Timer* timer = new Timer();
        timer->name = name;
        timer->duration = duration;
        timer->callback = callback;
        timer->looping = looping;
        timers.push_back(timer);
        return timer;
    }
    
    void update(double dt) {
        for (auto timer : timers) {
            timer->update(dt);
        }
        
        // 清理已完成的非循环定时器
        timers.erase(
            std::remove_if(timers.begin(), timers.end(),
                [](Timer* t) { return t->isCompleted() && !t->isLooping(); }),
            timers.end()
        );
    }
};
```

## 三、时间戳与帧同步

```cpp
class FrameSynchronizer {
private:
    int targetFPS = 60;
    double targetFrameTime;
    double lastFrameTime = 0;
    
public:
    FrameSynchronizer(int fps) : targetFPS(fps) {
        targetFrameTime = 1.0 / targetFPS;
    }
    
    void waitForFrame() {
        double currentTime = getCurrentTime();
        double elapsed = currentTime - lastFrameTime;
        
        if (elapsed < targetFrameTime) {
            double sleepTime = (targetFrameTime - elapsed) * 1000;
            std::this_thread::sleep_for(std::chrono::milliseconds((int)sleepTime));
        }
        
        lastFrameTime = getCurrentTime();
    }
};
```
