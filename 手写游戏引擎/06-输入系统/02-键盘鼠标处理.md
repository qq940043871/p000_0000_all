## 一、按键映射

```cpp
namespace KeyCode {
    enum {
        // 字母
        A = 'A', B = 'B', C = 'C', // ...
        
        // 功能键
        SPACE = 32,
        ENTER = 13,
        ESCAPE = 27,
        TAB = 9,
        
        // 方向键
        UP = 38, DOWN = 40, LEFT = 37, RIGHT = 39,
        
        // 修饰键
        SHIFT = 16,
        CTRL = 17,
        ALT = 18
    };
}

namespace MouseButton {
    enum {
        LEFT = 0,
        RIGHT = 1,
        MIDDLE = 2
    };
}

class InputMapper {
private:
    std::unordered_map<std::string, int> bindings;
    
public:
    void bind(const std::string& action, int keycode) {
        bindings[action] = keycode;
    }
    
    bool isActionPressed(const std::string& action) {
        auto it = bindings.find(action);
        if (it != bindings.end()) {
            return InputManager::getInstance()->isKeyDown(it->second);
        }
        return false;
    }
};
```

## 二、鼠标控制

```cpp
class MouseController {
public:
    Vector2 getPosition() {
        return InputManager::getInstance()->getMousePosition();
    }
    
    Vector2 getDelta() {
        return InputManager::getInstance()->getMouseDelta();
    }
    
    void setCursorVisible(bool visible) {
        #ifdef WIN32
        ShowCursor(visible);
        #endif
    }
    
    void setCursorLocked(bool locked) {
        #ifdef WIN32
        if (locked) {
            SetCapture(hwnd);
        } else {
            ReleaseCapture();
        }
        #endif
    }
};
```
