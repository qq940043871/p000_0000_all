## 一、手柄接口

```cpp
class Gamepad {
private:
    int id;
    bool connected;
    std::string name;
    
    float axes[16];      // 模拟摇杆
    bool buttons[32];    // 按键
    float triggers[4];    // 扳机键
    
public:
    float getAxis(int index) const { return axes[index]; }
    bool isButtonDown(int index) const { return buttons[index]; }
    float getTrigger(int index) const { return triggers[index]; }
    
    Vector2 getLeftStick() const {
        return Vector2(axes[0], axes[1]);
    }
    
    Vector2 getRightStick() const {
        return Vector2(axes[2], axes[3]);
    }
};

class GamepadManager {
private:
    std::vector<Gamepad*> gamepads;
    
public:
    void update() {
        for (auto* pad : gamepads) {
            pollGamepad(pad);
        }
    }
    
    Gamepad* getGamepad(int index) {
        if (index >= 0 && index < gamepads.size() && 
            gamepads[index]->isConnected()) {
            return gamepads[index];
        }
        return nullptr;
    }
};
```

## 二、Dead Zone处理

```cpp
Vector2 applyDeadZone(const Vector2& input, float deadZone = 0.15f) {
    float magnitude = input.length();
    
    if (magnitude < deadZone) {
        return Vector2::zero();
    }
    
    // 标准化并重新映射
    float normalized = (magnitude - deadZone) / (1.0f - deadZone);
    normalized = std::min(1.0f, normalized);
    
    return input.normalized() * normalized;
}
```
