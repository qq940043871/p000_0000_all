## 一、事件类型

```cpp
enum class InputEventType {
    KEY_DOWN,
    KEY_UP,
    KEY_PRESS,
    MOUSE_DOWN,
    MOUSE_UP,
    MOUSE_MOVE,
    MOUSE_WHEEL,
    GAMEPAD_CONNECTED,
    GAMEPAD_DISCONNECTED,
    GAMEPAD_BUTTON_DOWN,
    GAMEPAD_BUTTON_UP,
    GAMEPAD_AXIS
};

struct InputEvent {
    InputEventType type;
    union {
        struct { int keycode; } key;
        struct { int button; Vector2 position; } mouse;
        struct { int axis; float value; } gamepad;
    };
};
```

## 二、事件系统

```cpp
class InputManager : public IModule {
private:
    std::vector<InputEvent> eventQueue;
    std::unordered_map<int, bool> keyStates;
    std::unordered_map<int, bool> prevKeyStates;
    Vector2 mousePosition;
    Vector2 mouseDelta;
    
public:
    void pollEvents() {
        eventQueue.clear();
        
        // 键盘事件
        for (auto& [key, state] : keyStates) {
            bool prevState = prevKeyStates[key];
            if (state && !prevState) {
                eventQueue.push_back({InputEventType::KEY_DOWN, .key = {key}});
            } else if (!state && prevState) {
                eventQueue.push_back({InputEventType::KEY_UP, .key = {key}});
            }
        }
        
        prevKeyStates = keyStates;
    }
    
    bool isKeyDown(int keycode) {
        return keyStates[keycode];
    }
    
    bool isKeyPressed(int keycode) {
        return keyStates[keycode] && !prevKeyStates[keycode];
    }
};
```
