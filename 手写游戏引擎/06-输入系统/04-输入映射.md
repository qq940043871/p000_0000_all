## 一、动作映射

```cpp
class InputAction {
public:
    std::string name;
    std::vector<InputBinding> bindings;
    
    bool isPressed() const {
        for (auto& binding : bindings) {
            if (binding.isPressed()) return true;
        }
        return false;
    }
    
    bool isJustPressed() const {
        for (auto& binding : bindings) {
            if (binding.isJustPressed()) return true;
        }
        return false;
    }
    
    float getValue() const {
        float value = 0;
        for (auto& binding : bindings) {
            value = std::max(value, binding.getValue());
        }
        return value;
    }
};

class InputMapping {
private:
    std::unordered_map<std::string, InputAction> actions;
    
public:
    void createAction(const std::string& name) {
        actions[name] = InputAction{name};
    }
    
    void bindKey(const std::string& action, int keycode) {
        actions[action].bindings.push_back(KeyBinding(keycode));
    }
    
    void bindMouseButton(const std::string& action, int button) {
        actions[action].bindings.push_back(MouseBinding(button));
    }
    
    void bindGamepadButton(const std::string& action, int gamepadId, int button) {
        actions[action].bindings.push_back(GamepadButtonBinding(gamepadId, button));
    }
    
    void bindGamepadAxis(const std::string& action, int gamepadId, int axis) {
        actions[action].bindings.push_back(GamepadAxisBinding(gamepadId, axis));
    }
};
```
