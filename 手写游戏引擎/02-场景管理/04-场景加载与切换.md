## 一、同步加载

```cpp
class SceneLoader {
public:
    Scene* loadSync(const std::string& path) {
        Scene* scene = new Scene();
        
        // 解析场景文件
        rapidjson::Document doc;
        doc.Parse(readFile(path));
        
        // 加载场景数据
        const rapidjson::Value& objects = doc["objects"];
        for (auto& obj : objects.GetArray()) {
            GameObject* gameObj = createGameObject(obj);
            scene->addGameObject(gameObj);
        }
        
        return scene;
    }
    
private:
    GameObject* createGameObject(const rapidjson::Value& data) {
        GameObject* obj = new GameObject();
        
        // 设置名称
        if (data.HasMember("name")) {
            obj->setName(data["name"].GetString());
        }
        
        // 设置变换
        if (data.HasMember("transform")) {
            const auto& t = data["transform"];
            obj->getTransform()->setPosition(
                t["position"][0].GetFloat(),
                t["position"][1].GetFloat(),
                t["position"][2].GetFloat()
            );
        }
        
        // 创建组件
        if (data.HasMember("components")) {
            for (auto& comp : data["components"].GetArray()) {
                Component* component = createComponent(comp);
                obj->addComponent(component);
            }
        }
        
        return obj;
    }
};
```

## 二、异步加载

```cpp
class AsyncSceneLoader {
private:
    std::vector<LoadTask> loadTasks;
    ThreadPool* threadPool;
    
public:
    void loadAsync(const std::string& path, 
                  std::function<void(Scene*)> callback) {
        LoadTask task;
        task.path = path;
        task.callback = callback;
        
        threadPool->enqueue([this, task]() {
            Scene* scene = loadSync(task.path);
            
            // 回到主线程执行回调
            MainThreadQueue::enqueue([scene, task]() {
                task.callback(scene);
            });
        });
    }
};
```
