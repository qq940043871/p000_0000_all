## 一、场景结构

```cpp
class Scene {
private:
    std::string name;
    std::vector<GameObject*> rootObjects;
    SceneGraph* sceneGraph;
    std::unordered_map<std::string, GameObject*> namedObjects;
    
public:
    void addGameObject(GameObject* obj) {
        rootObjects.push_back(obj);
        sceneGraph->addNode(obj);
        
        if (!obj->getName().empty()) {
            namedObjects[obj->getName()] = obj;
        }
    }
    
    void removeGameObject(GameObject* obj) {
        rootObjects.erase(
            std::remove(rootObjects.begin(), rootObjects.end(), obj),
            rootObjects.end()
        );
        sceneGraph->removeNode(obj);
    }
    
    GameObject* findByName(const std::string& name) {
        return namedObjects[name];
    }
    
    std::vector<GameObject*> findByTag(const std::string& tag) {
        std::vector<GameObject*> results;
        for (auto* obj : rootObjects) {
            if (obj->hasTag(tag)) {
                results.push_back(obj);
            }
        }
        return results;
    }
};

class SceneManager : public IModule {
private:
    Scene* currentScene = nullptr;
    std::unordered_map<std::string, Scene*> scenes;
    
public:
    Scene* getCurrentScene() { return currentScene; }
    
    void loadScene(const std::string& name) {
        auto it = scenes.find(name);
        if (it != scenes.end()) {
            // 卸载旧场景
            if (currentScene) {
                currentScene->onExit();
            }
            
            // 加载新场景
            currentScene = it->second;
            currentScene->onEnter();
        }
    }
};
```
