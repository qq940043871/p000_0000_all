## 一、生命周期钩子

```cpp
class GameObject : public SceneNode {
public:
    enum class State {
        CREATED,
        ACTIVE,
        INACTIVE,
        DESTROYED
    };
    
private:
    State state = State::CREATED;
    bool activeInHierarchy = false;
    
public:
    void Awake() {
        // 组件初始化
        for (auto* comp : components) {
            comp->Awake();
        }
        state = State::ACTIVE;
    }
    
    void OnEnable() {
        for (auto* comp : components) {
            if (comp->isEnabled()) {
                comp->OnEnable();
            }
        }
    }
    
    void Start() {
        for (auto* comp : components) {
            if (comp->isEnabled()) {
                comp->Start();
            }
        }
    }
    
    void Update(double dt) {
        if (state != State::ACTIVE) return;
        
        for (auto* comp : components) {
            if (comp->isEnabled()) {
                comp->Update(dt);
            }
        }
    }
    
    void OnDisable() {
        for (auto* comp : components) {
            comp->OnDisable();
        }
    }
    
    void OnDestroy() {
        for (auto* comp : components) {
            comp->OnDestroy();
        }
        state = State::DESTROYED;
    }
    
    void Destroy() {
        SceneManager::getCurrentScene()->scheduleDestruction(this);
    }
};
```

## 二、延迟销毁

```cpp
class Scene {
private:
    std::vector<GameObject*> destructionQueue;
    
public:
    void scheduleDestruction(GameObject* obj) {
        destructionQueue.push_back(obj);
    }
    
    void processDestructionQueue() {
        for (auto* obj : destructionQueue) {
            obj->OnDestroy();
            // 从场景移除
            removeGameObject(obj);
            delete obj;
        }
        destructionQueue.clear();
    }
};
```
