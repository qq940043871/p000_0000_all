## 一、骨骼蒙皮

```cpp
struct SkinnedMesh {
    std::vector<Bone> bones;
    std::vector<VertexSkinData> skinData;  // 顶点蒙皮权重
};

struct VertexSkinData {
    int boneIndices[4];
    float boneWeights[4];
};

class SkeletalAnimator {
private:
    SkinnedMesh* mesh;
    std::unordered_map<std::string, AnimationClip*> animations;
    AnimationClip* currentAnimation;
    float currentTime = 0;
    float playbackSpeed = 1.0f;
    
public:
    void update(float dt) {
        if (!currentAnimation) return;
        
        currentTime += dt * playbackSpeed;
        if (currentTime > currentAnimation->duration) {
            currentTime = 0;  // 循环
        }
        
        // 计算骨骼变换
        std::vector<Matrix4x4> boneMatrices;
        for (auto& boneAnim : currentAnimation->boneAnimations) {
            Matrix4x4 transform = interpolateKeyframes(boneAnim.second, currentTime);
            boneMatrices.push_back(transform);
        }
    }
};
```
