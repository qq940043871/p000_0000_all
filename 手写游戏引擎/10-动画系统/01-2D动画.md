## 一、精灵动画

```cpp
class SpriteAnimator {
private:
    TextureAtlas* atlas;
    std::vector<std::string> frames;
    int currentFrame = 0;
    float frameTime = 0;
    float frameRate = 10.0f;  // FPS
    bool playing = true;
    bool looping = true;
    
public:
    void update(float dt) {
        if (!playing) return;
        
        frameTime += dt;
        float interval = 1.0f / frameRate;
        
        if (frameTime >= interval) {
            frameTime -= interval;
            currentFrame++;
            
            if (currentFrame >= frames.size()) {
                if (looping) {
                    currentFrame = 0;
                } else {
                    currentFrame = frames.size() - 1;
                    playing = false;
                }
            }
        }
    }
    
    void play() { playing = true; }
    void pause() { playing = false; }
    void stop() { playing = false; currentFrame = 0; }
    
    std::string getCurrentFrame() {
        return frames[currentFrame];
    }
};
```

## 二、骨骼动画

```cpp
struct Bone {
    std::string name;
    int parentIndex;
    Matrix4x4 bindPose;
    Matrix4x4 inverseBindPose;
};

struct Keyframe {
    float time;
    Vector3 position;
    Quaternion rotation;
    Vector3 scale;
};

struct AnimationClip {
    std::string name;
    float duration;
    std::vector<std::pair<int, std::vector<Keyframe>>> boneAnimations;
};
```
