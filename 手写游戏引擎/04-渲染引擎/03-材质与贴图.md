## 一、材质系统

```cpp
class Material {
private:
    Shader* shader;
    std::unordered_map<std::string, Texture*> textures;
    std::unordered_map<std::string, Vector4> vectorProperties;
    std::unordered_map<std::string, Matrix4x4> matrixProperties;
    
public:
    void setTexture(const std::string& name, Texture* texture) {
        textures[name] = texture;
    }
    
    void setFloat(const std::string& name, float value) {
        vectorProperties[name] = Vector4(value, 0, 0, 0);
    }
    
    void setVector(const std::string& name, const Vector4& value) {
        vectorProperties[name] = value;
    }
    
    void setMatrix(const std::string& name, const Matrix4x4& value) {
        matrixProperties[name] = value;
    }
    
    void bind() {
        shader->use();
        
        int textureSlot = 0;
        for (auto& [name, tex] : textures) {
            tex->bind(textureSlot);
            shader->setUniform(name, textureSlot++);
        }
        
        for (auto& [name, vec] : vectorProperties) {
            shader->setUniform(name, vec);
        }
        
        for (auto& [name, mat] : matrixProperties) {
            shader->setUniform(name, mat);
        }
    }
};
```

## 二、纹理系统

```cpp
class Texture {
private:
    uint32_t id;
    int width, height;
    TextureFormat format;
    TextureType type;
    
public:
    void loadFromFile(const std::string& path) {
        int channels;
        stbi_set_flip_vertically_on_load(true);
        uint8_t* data = stbi_load(path.c_str(), &width, &height, &channels, 0);
        
        format = getFormat(channels);
        
        glGenTextures(1, &id);
        glBindTexture(GL_TEXTURE_2D, id);
        
        glTexImage2D(GL_TEXTURE_2D, 0, format.internalFormat,
                     width, height, 0, format.format,
                     GL_UNSIGNED_BYTE, data);
        
        glGenerateMipmap(GL_TEXTURE_2D);
        
        stbi_image_free(data);
    }
    
    void bind(int slot = 0) {
        glActiveTexture(GL_TEXTURE0 + slot);
        glBindTexture(GL_TEXTURE_2D, id);
    }
};

class TextureAtlas {
private:
    Texture* atlas;
    std::unordered_map<std::string, RectUV> regions;
    
public:
    void addRegion(const std::string& name, int x, int y, int w, int h) {
        RectUV uv;
        uv.u0 = (float)x / atlas->getWidth();
        uv.v0 = (float)y / atlas->getHeight();
        uv.u1 = (float)(x + w) / atlas->getWidth();
        uv.v1 = (float)(y + h) / atlas->getHeight();
        regions[name] = uv;
    }
};
```
