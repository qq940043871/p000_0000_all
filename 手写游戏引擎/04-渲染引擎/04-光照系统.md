## 一、光源类型

```cpp
class Light {
public:
    enum class Type {
        DIRECTIONAL,  // 方向光
        POINT,        // 点光源
        SPOT,         // 聚光灯
        AMBIENT       // 环境光
    };
    
    Type type;
    Vector3 color;
    float intensity;
    
    virtual ~Light() = default;
};

class DirectionalLight : public Light {
public:
    Vector3 direction;  // 光照方向
    
    DirectionalLight() {
        type = Type::DIRECTIONAL;
        direction = Vector3(0, -1, 0).normalized();
    }
};

class PointLight : public Light {
public:
    Vector3 position;
    float radius;         // 影响范围
    float falloff;        // 衰减系数
    
    PointLight() {
        type = Type::POINT;
        radius = 10.0f;
        falloff = 2.0f;
    }
};

class SpotLight : public Light {
public:
    Vector3 position;
    Vector3 direction;
    float angle;          // 锥角
    float penumbra;        // 半影
    
    SpotLight() {
        type = Type::SPOT;
        angle = 45.0f;
        penumbra = 5.0f;
    }
};
```

## 二、光照计算

```cpp
// PBR光照模型
struct PBRMaterial {
    Vector3 albedo;
    float metallic;
    float roughness;
    float ao;
    Texture* albedoMap;
    Texture* normalMap;
    Texture* metallicMap;
    Texture* roughnessMap;
};

Vector3 calculatePBR(const PBRMaterial& mat,
                     const Vector3& viewDir,
                     const Vector3& lightDir,
                     const Vector3& lightColor,
                     const Vector3& normal) {
    // 中间向量
    Vector3 halfDir = (viewDir + lightDir).normalized();
    
    // 漫反射
    Vector3 F0 = mix(Vector3(0.04), mat.albedo, mat.metallic);
    float NDF = distributionGGX(normal, halfDir, mat.roughness);
    float G = geometrySmith(normal, viewDir, lightDir, mat.roughness);
    Vector3 F = fresnelSchlick(max(dot(normal, viewDir), 0.0), F0);
    
    Vector3 kD = (Vector3(1) - F) * (1 - mat.metallic);
    Vector3 numerator = NDF * G * F;
    float denominator = 4.0 * max(dot(normal, viewDir), 0.0) * 
                        max(dot(normal, lightDir), 0.0) + 0.0001;
    Vector3 specular = numerator / denominator;
    
    float NdotL = max(dot(normal, lightDir), 0.0);
    return (kD * mat.albedo / PI + specular) * lightColor * NdotL;
}
```
