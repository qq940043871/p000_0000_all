## 一、后处理管线

```cpp
class PostProcessingPipeline {
private:
    std::vector<PostProcessingEffect*> effects;
    FrameBufferObject sceneBuffer;
    FrameBufferObject outputBuffer;
    
public:
    void addEffect(PostProcessingEffect* effect) {
        effects.push_back(effect);
    }
    
    void render(const Texture* input, RenderContext* context) {
        FBO* currentIn = &sceneBuffer;
        currentIn->attachTexture(input);
        
        for (auto* effect : effects) {
            FBO* currentOut = &outputBuffer;
            if (effect == effects.back()) {
                currentOut = nullptr;  // 输出到屏幕
            }
            
            effect->process(currentIn->getTexture(), currentOut);
            std::swap(currentIn, currentOut);
        }
    }
};
```

// Bloom效果
class BloomEffect : public PostProcessingEffect {
private:
    float threshold = 0.8f;
    float intensity = 1.0f;
    int iterations = 5;
    
public:
    void process(const Texture* input, FBO* output) override {
        // 1. 提取亮部
        ExtractBrightPass(input, brightBuffer, threshold);
        
        // 2. 模糊
        GaussianBlur(brightBuffer, blurBuffer, iterations);
        
        // 3. 混合
        Blend(input, blurBuffer, output, intensity);
    }
};
