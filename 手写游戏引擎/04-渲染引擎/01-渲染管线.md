## 一、渲染流程

```
渲染管线：
  1. 应用阶段（CPU）
     - 视锥裁剪
     - 排序（透明/不透明）
     - 批次合并
  
  2. 几何阶段（GPU）
     - 顶点着色器
     - 曲面细分
     - 几何着色器
     - 图元装配裁剪
     - 光栅化
  
  3. 光栅化阶段（GPU）
     - 片段着色器
     - 深度测试
     - 模板测试
     - 混合
  
  4. 输出合并
     - 帧缓冲区
```

## 二、C++实现

```cpp
class RenderPipeline {
private:
    std::vector<IRenderPass*> passes;
    RenderContext* context;
    
public:
    void initialize() {
        passes.push_back(new ShadowPass());
        passes.push_back(new GeometryPass());
        passes.push_back(new LightingPass());
        passes.push_back(new PostProcessingPass());
    }
    
    void render(const Scene* scene, Camera* camera) {
        context->beginFrame();
        
        for (auto* pass : passes) {
            pass->begin(camera);
            pass->execute(scene, context);
            pass->end();
        }
        
        context->endFrame();
    }
};

class RenderContext {
private:
    GPUBuffer* uniformBuffer;
    std::vector<RenderCommand> commands;
    
public:
    void submit(const Mesh* mesh, const Material* mat, 
               const Matrix4x4& transform) {
        commands.push_back({mesh, mat, transform});
    }
    
    void execute() {
        sort(commands.begin(), commands.end(), compare);
        
        for (auto& cmd : commands) {
            draw(cmd);
        }
        commands.clear();
    }
};
```

## 三、命令缓冲

```cpp
class RenderCommand {
public:
    enum class Type {
        DRAW_MESH,
        DRAW_INSTANCED,
        SET_VIEWPORT,
        SET_RENDER_TARGET,
        CLEAR
    };
    
    Type type;
    const Mesh* mesh;
    const Material* material;
    const Matrix4x4* transforms;
    int instanceCount;
};

class CommandBuffer {
private:
    std::vector<uint8_t> buffer;
    size_t currentOffset = 0;
    
public:
    void begin() {
        currentOffset = 0;
    }
    
    void addCommand(const RenderCommand& cmd) {
        size_t size = sizeof(RenderCommand);
        ensureCapacity(size);
        memcpy(buffer.data() + currentOffset, &cmd, size);
        currentOffset += size;
    }
    
    void end() {
        // 提交到GPU
    }
};
```
