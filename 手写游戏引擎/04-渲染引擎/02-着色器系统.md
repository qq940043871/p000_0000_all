## 一、着色器类型

```
着色器类型：
  1. 顶点着色器（Vertex Shader）
     - 变换顶点
     - 传递属性到片段着色器
  
  2. 片段着色器（Fragment Shader）
     - 计算像素颜色
     - 光照计算
  
  3. 几何着色器（Geometry Shader）
     - 生成/销毁图元
  
  4. 计算着色器（Compute Shader）
     - GPGPU通用计算
```

## 二、C++着色器管理

```cpp
class Shader {
private:
    uint32_t programID;
    std::unordered_map<std::string, int> uniformLocations;
    
public:
    void compile(const std::string& vertexSource,
                 const std::string& fragmentSource) {
        uint32_t vs = compileShader(vertexSource, GL_VERTEX_SHADER);
        uint32_t fs = compileShader(fragmentSource, GL_FRAGMENT_SHADER);
        
        programID = glCreateProgram();
        glAttachShader(programID, vs);
        glAttachShader(programID, fs);
        glLinkProgram(programID);
        
        // 缓存uniform位置
        cacheUniforms();
    }
    
    void setUniform(const std::string& name, const Matrix4x4& value) {
        glUniformMatrix4fv(uniformLocations[name], 1, GL_FALSE, value.data());
    }
    
    void setUniform(const std::string& name, int value) {
        glUniform1i(uniformLocations[name], value);
    }
    
    void use() {
        glUseProgram(programID);
    }
};

class ShaderManager {
private:
    std::unordered_map<std::string, Shader*> shaders;
    
public:
    Shader* loadShader(const std::string& name,
                       const std::string& vsPath,
                       const std::string& fsPath) {
        auto it = shaders.find(name);
        if (it != shaders.end()) return it->second;
        
        Shader* shader = new Shader();
        shader->compile(readFile(vsPath), readFile(fsPath));
        shaders[name] = shader;
        
        return shader;
    }
};
```

## 三、HLSL支持（DirectX）

```cpp
class HLSLShader : public IShader {
private:
    ID3DBlob* vertexBlob;
    ID3DBlob* pixelBlob;
    ID3D12PipelineState* pipelineState;
    
public:
    void compile(const std::string& source,
                 ShaderType type,
                 const std::string& entryPoint) {
        D3DCompile(source.c_str(), source.length(),
                   nullptr, nullptr, nullptr,
                   entryPoint.c_str(),
                   getProfile(type).c_str(),
                   flags, 0,
                   &blob, &errorBlob);
    }
};
```
