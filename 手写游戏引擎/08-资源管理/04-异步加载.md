## 一、异步加载器

```cpp
class AsyncResourceLoader {
private:
    ThreadPool* threadPool;
    std::queue<LoadRequest> pendingRequests;
    std::unordered_map<std::string, LoadRequest*> activeRequests;
    std::vector<LoadRequest*> completedRequests;
    std::mutex mutex;
    
public:
    void loadAsync(const std::string& path, 
                   std::function<void(Resource*)> callback) {
        LoadRequest* request = new LoadRequest();
        request->path = path;
        request->callback = callback;
        request->promise = new std::promise<Resource*>();
        
        {
            std::lock_guard<std::mutex> lock(mutex);
            pendingRequests.push(*request);
        }
        
        threadPool->enqueue([this, request]() {
            Resource* resource = loadResource(request->path);
            request->promise->set_value(resource);
            
            {
                std::lock_guard<std::mutex> lock(mutex);
                completedRequests.push_back(request);
            }
        });
    }
    
    void update() {
        std::lock_guard<std::mutex> lock(mutex);
        
        for (auto* req : completedRequests) {
            req->callback(req->resource);
            delete req;
        }
        completedRequests.clear();
    }
};
```
