## 一、资源缓存

```cpp
class ResourceCache {
private:
    std::unordered_map<std::string, Resource*> resources;
    size_t maxCacheSize = 1024 * 1024 * 1024; // 1GB
    size_t currentSize = 0;
    
public:
    Resource* get(const std::string& path) {
        auto it = resources.find(path);
        if (it != resources.end()) {
            it->second->refCount++;
            return it->second;
        }
        return nullptr;
    }
    
    void add(const std::string& path, Resource* resource) {
        resources[path] = resource;
        currentSize += resource->getSize();
        
        // LRU淘汰
        while (currentSize > maxCacheSize) {
            evictLRU();
        }
    }
    
    void release(Resource* resource) {
        resource->refCount--;
        if (resource->refCount <= 0 && shouldUnload(resource)) {
            resource->unload();
            resources.erase(resource->path);
            currentSize -= resource->getSize();
        }
    }
};
```
