## 一、资源类型

```cpp
enum class ResourceType {
    MESH,
    TEXTURE,
    MATERIAL,
    SHADER,
    AUDIO,
    SCENE,
    FONT,
    ANIMATION
};

class IResource {
public:
    virtual ~IResource() = default;
    virtual ResourceType getType() const = 0;
    virtual void load(const std::string& path) = 0;
    virtual void unload() = 0;
    virtual bool isLoaded() const = 0;
};

class Resource {
public:
    std::string path;
    ResourceType type;
    int refCount = 0;
    bool isLoaded = false;
    IResource* data = nullptr;
};
```

## 二、资源加载器

```cpp
template<typename T>
class ResourceLoader {
public:
    virtual ~ResourceLoader() = default;
    virtual bool canLoad(const std::string& extension) const = 0;
    virtual T* load(const std::string& path) = 0;
};

class MeshLoader : public ResourceLoader<Mesh> {
public:
    bool canLoad(const std::string& extension) const override {
        return extension == ".obj" || extension == ".fbx" || 
               extension == ".gltf" || extension == ".glb";
    }
    
    Mesh* load(const std::string& path) override {
        std::string ext = getExtension(path);
        
        if (ext == ".obj") return loadOBJ(path);
        if (ext == ".gltf" || ext == ".glb") return loadGLTF(path);
        
        return nullptr;
    }
    
    Mesh* loadOBJ(const std::string& path) {
        Mesh* mesh = new Mesh();
        // 解析OBJ文件
        return mesh;
    }
};
```
