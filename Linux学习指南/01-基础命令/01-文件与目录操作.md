## 一、文件与目录管理

### 基础命令

```bash
# ====== 目录操作 ======

# 切换目录
cd /home/user        # 切换到绝对路径
cd ./folder         # 切换到相对路径
cd ..               # 返回上级目录
cd ~                # 返回用户主目录
cd -                # 返回上次目录

# 查看目录
ls -la /home        # 列出详细信息
ls -lh              # 人性化显示大小
ls -lt              # 按修改时间排序
ls -S               # 按文件大小排序
ls -a               # 显示隐藏文件

# 创建目录
mkdir folder            # 创建单个目录
mkdir -p a/b/c          # 递归创建
mkdir -m 755 folder     # 指定权限

# 删除目录
rmdir folder           # 删除空目录
rm -rf folder          # 递归删除（非空）

# ====== 文件操作 ======

# 查看文件
cat file.txt           # 一次性显示
cat -n file.txt        # 显示行号
more file.txt          # 分页显示
less file.txt          # 可前后翻页
tail -f file.txt       # 实时追踪
tail -n 100 file.txt   # 显示最后100行
head -n 20 file.txt    # 显示前20行

# 复制
cp file1.txt file2.txt           # 复制文件
cp -r folder1 folder2            # 递归复制
cp -p file.txt /dest/             # 保留属性
cp -a folder /dest/              # 归档复制

# 移动/重命名
mv file1.txt file2.txt           # 重命名
mv file.txt /folder/              # 移动

# 删除
rm file.txt            # 删除文件
rm -f file.txt         # 强制删除
rm -rf /folder/        # 递归强制删除
```

## 二、查找命令

```bash
# ====== find ======
find / -name "*.log"                 # 按名字查找
find / -type f -size +100M           # 查找大于100M的文件
find / -type f -mtime -7             # 查找7天内修改的文件
find / -type f -mmin -60             # 查找60分钟内修改的文件
find / -user root                     # 按用户查找
find / -perm 755                     # 按权限查找
find / -newer file.txt               # 查找比file.txt新的文件

# 查找并操作
find / -name "*.tmp" -exec rm {} \;  # 查找并删除
find / -name "*.log" -delete          # 查找并删除

# ====== which/whereis ======
which python        # 查找命令所在路径
whereis python      # 查找命令、源码、手册位置

# ====== locate ======
updatedb                           # 更新文件数据库
locate filename                    # 快速查找
```

## 三、文件内容查找

```bash
# ====== grep ======
grep "keyword" file.txt            # 查找关键字
grep -r "keyword" /folder/         # 递归查找
grep -i "keyword" file.txt         # 忽略大小写
grep -n "keyword" file.txt         # 显示行号
grep -c "keyword" file.txt         # 统计匹配行数
grep -v "keyword" file.txt         # 反向查找
grep -E "pattern1|pattern2" file   # 扩展正则
grep -A 5 "keyword" file.txt       # 显示匹配行及后5行
grep -B 5 "keyword" file.txt       # 显示匹配行及前5行

# ====== grep配合管道 ======
ps aux | grep java                  # 查找Java进程
netstat -tlnp | grep 8080          # 查找占用8080端口的进程
df -h | grep sda                   # 查找特定磁盘
```

## 四、文件权限

```bash
# ====== chmod ======
chmod 755 file.txt                 # 数字权限
chmod u+x file.txt                 # 用户添加执行
chmod g-w file.txt                 # 组移除写
chmod a+r file.txt                 # 所有用户添加读
chmod +x file.txt                  # 所有用户添加执行
chmod -R 755 folder/               # 递归设置

# 权限数字对照
# 4 = r (读)
# 2 = w (写)
# 1 = x (执行)

# ====== chown ======
chown user:group file.txt          # 改变所有者
chown -R user:group folder/        # 递归改变
chgrp group file.txt               # 改变所属组

# ====== 特殊权限 ======
chmod +s file.txt                  # SETUID/SETGID
chmod +t folder/                   # Sticky Bit
```

## 五、压缩与打包

```bash
# ====== tar ======
tar -cvf archive.tar folder/       # 创建归档
tar -xvf archive.tar               # 解压归档
tar -xvf archive.tar -C /dest/     # 解压到指定目录
tar -tvf archive.tar               # 查看归档内容

# 压缩格式
tar -czvf archive.tar.gz folder/   # gzip压缩
tar -xzvf archive.tar.gz           # gzip解压
tar -cjvf archive.tar.bz2 folder/  # bzip2压缩
tar -xjvf archive.tar.bz2          # bzip2解压
tar -cJvf archive.tar.xz folder/   # xz压缩

# ====== zip ======
zip -r archive.zip folder/         # 创建zip
unzip archive.zip                  # 解压zip
unzip -l archive.zip              # 查看zip内容

# ====== 其他 ======
gzip file.txt                      # gzip压缩
gunzip file.txt.gz                # gzip解压
bzip2 file.txt                    # bzip2压缩
xz file.txt                       # xz压缩
```

## 六、磁盘管理

```bash
# ====== df ======
df -h                             # 人性化显示
df -i                             # 显示inode使用
df -T                             # 显示文件系统类型

# ====== du ======
du -sh folder/                    # 目录总大小
du -sh *                          # 当前目录各文件大小
du -h --max-depth=1 /             # 限制深度
du -ah                           # 显示所有文件

# ====== mount ======
mount /dev/sdb1 /mnt/usb          # 挂载
mount -o remount,rw /             # 重新挂载
mount -t ntfs /dev/sdb1 /mnt/    # 挂载NTFS
umount /mnt/usb                   # 卸载
```
