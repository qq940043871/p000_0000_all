## 一、Function Calling

```
Function Calling流程：
  1. 模型生成函数调用
  2. 本地执行函数
  3. 返回结果给模型
  4. 模型生成最终回复

与MCP的关系：
  - Function Calling是通用机制
  - MCP是标准化的协议
  - MCP可以基于Function Calling实现
```

## 二、函数注册

```python
class FunctionRegistry:
    """函数注册表"""
    
    def __init__(self):
        self.functions: Dict[str, Callable] = {}
        self.schemas: Dict[str, Dict] = {}
    
    def register(self, name: str, func: Callable, schema: Dict = None):
        """注册函数"""
        self.functions[name] = func
        
        if schema is None:
            schema = self._generate_schema(func)
        
        self.schemas[name] = schema
    
    def _generate_schema(self, func: Callable) -> Dict:
        """从函数签名生成schema"""
        sig = inspect.signature(func)
        properties = {}
        required = []
        
        for name, param in sig.parameters.items():
            properties[name] = {
                "type": self._python_type_to_json(param.annotation),
                "description": param.name
            }
            if param.default == inspect.Parameter.empty:
                required.append(name)
        
        return {
            "name": func.__name__,
            "description": func.__doc__ or "",
            "parameters": {
                "type": "object",
                "properties": properties,
                "required": required
            }
        }
    
    def list_functions(self) -> List[Dict]:
        """列出所有函数"""
        return list(self.schemas.values())
    
    async def call(self, name: str, arguments: Dict) -> Any:
        """调用函数"""
        func = self.functions.get(name)
        if not func:
            raise ValueError(f"Function not found: {name}")
        
        # 检查是否是异步函数
        if asyncio.iscoroutinefunction(func):
            return await func(**arguments)
        else:
            return func(**arguments)


# 使用示例
function_registry = FunctionRegistry()

@function_registry.register("get_weather", schema={
    "description": "获取天气信息",
    "parameters": {
        "type": "object",
        "properties": {
            "city": {"type": "string", "description": "城市名"}
        },
        "required": ["city"]
    }
})
async def get_weather(city: str) -> Dict:
    return {"city": city, "weather": "晴天", "temp": 25}
```
