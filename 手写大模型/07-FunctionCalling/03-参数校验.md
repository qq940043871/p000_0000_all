## 一、结果格式化

```python
class ResultFormatter:
    """结果格式化"""
    
    @staticmethod
    def format(result: Any, format_type: str = "auto") -> str:
        """格式化函数结果"""
        if format_type == "json":
            return json.dumps(result, ensure_ascii=False, indent=2)
        elif format_type == "text":
            return ResultFormatter._to_text(result)
        else:  # auto
            if isinstance(result, (dict, list)):
                return json.dumps(result, ensure_ascii=False)
            return str(result)
    
    @staticmethod
    def _to_text(result: Any, indent: int = 0) -> str:
        """递归转换为文本"""
        if isinstance(result, dict):
            lines = []
            for k, v in result.items():
                lines.append(f"{k}: {ResultFormatter._to_text(v, indent+2)}")
            return "\n".join(lines)
        elif isinstance(result, list):
            return "\n".join(f"- {ResultFormatter._to_text(item, indent)}" 
                          for item in result)
        else:
            return str(result)
```

## 二、循环调用

```python
class ToolLoopExecutor:
    """工具循环执行器"""
    
    def __init__(self, executor: FunctionExecutor, 
                 max_iterations: int = 10):
        self.executor = executor
        self.max_iterations = max_iterations
    
    async def execute_loop(self, messages: List[Dict]) -> str:
        """循环执行直到不调用工具"""
        for iteration in range(self.max_iterations):
            # 1. LLM生成
            response = await llm.chat(messages)
            
            # 2. 检查是否有函数调用
            if not response.tool_calls:
                return response.content
            
            # 3. 执行所有工具调用
            tool_results = []
            for tool_call in response.tool_calls:
                result = await self.executor.execute(tool_call)
                tool_results.append({
                    "tool_call_id": tool_call["id"],
                    "result": result
                })
            
            # 4. 添加到消息
            messages.append({
                "role": "assistant",
                "content": response.content,
                "tool_calls": response.tool_calls
            })
            
            for tool_result in tool_results:
                messages.append({
                    "role": "tool",
                    "tool_call_id": tool_result["tool_call_id"],
                    "content": tool_result["result"]
                })
        
        return "Maximum iterations exceeded"
```
