## 一、JSON Schema校验

```python
from pydantic import BaseModel, ValidationError
from typing import get_type_hints

class FunctionValidator:
    """函数参数校验器"""
    
    def __init__(self, schema: Dict):
        self.schema = schema
        self.model = self._create_model(schema)
    
    def _create_model(self, schema: Dict) -> type[BaseModel]:
        """从schema创建Pydantic模型"""
        properties = schema.get("parameters", {}).get("properties", {})
        required = schema.get("parameters", {}).get("required", [])
        
        # 构建字段定义
        field_definitions = {}
        for name, prop in properties.items():
            field_type = self._json_type_to_python(prop.get("type"))
            default = ... if name in required else None
            field_definitions[name] = (field_type, default)
        
        return create_model("DynamicModel", **field_definitions)
    
    def validate(self, arguments: Dict) -> BaseModel:
        """校验参数"""
        return self.model(**arguments)
    
    @staticmethod
    def _json_type_to_python(json_type: str) -> type:
        """JSON类型转Python类型"""
        type_map = {
            "string": str,
            "integer": int,
            "number": float,
            "boolean": bool,
            "array": list,
            "object": dict
        }
        return type_map.get(json_type, str)
```

## 二、完整验证流程

```python
class FunctionExecutor:
    """函数执行器（含验证）"""
    
    def __init__(self, registry: FunctionRegistry):
        self.registry = registry
        self.validator = PydanticValidator()
    
    async def execute(self, function_call: Dict) -> Dict:
        """执行函数调用"""
        name = function_call["name"]
        arguments = function_call.get("arguments", {})
        
        # 1. 获取schema
        schema = self.registry.schemas.get(name)
        if not schema:
            return {"error": f"Unknown function: {name}"}
        
        # 2. 验证参数
        try:
            validator = FunctionValidator(schema)
            validated_args = validator.validate(arguments)
        except ValidationError as e:
            return {"error": f"Validation failed: {e}"}
        
        # 3. 执行
        try:
            result = await self.registry.call(name, validated_args.dict())
            return {"result": result}
        except Exception as e:
            return {"error": str(e)}
```
