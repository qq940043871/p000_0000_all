## 一、流式生成器

```python
class StreamingGenerator:
    """流式输出生成器"""
    
    def __init__(self, model: LLMModel, tokenizer: Tokenizer):
        self.model = model
        self.tokenizer = tokenizer
    
    async def generate_stream(self, prompt: str, **kwargs) -> AsyncGenerator[str, None]:
        """异步流式生成"""
        input_ids = self.tokenizer.encode(prompt, return_tensors="pt")
        
        past_key_values = None
        generated = []
        
        for _ in range(kwargs.get("max_tokens", 100)):
            if past_key_values is None:
                outputs = self.model(input_ids, use_cache=True)
            else:
                outputs = self.model(
                    input_ids[:, -1:],
                    past_key_values=past_key_values,
                    use_cache=True
                )
            
            logits = outputs.logits[0, -1, :]
            past_key_values = outputs.past_key_values
            
            # 采样
            probs = F.softmax(logits / kwargs.get("temperature", 1.0), dim=-1)
            next_token = torch.multinomial(probs, 1).item()
            
            if next_token == self.tokenizer.eos_token_id:
                break
            
            generated.append(next_token)
            token_text = self.tokenizer.decode([next_token])
            yield token_text
            
            input_ids = torch.tensor([[next_token]])
        
        # 返回完整文本
        yield "[DONE]"


# SSE格式输出
async def stream_sse(response: AsyncGenerator[str, None]):
    """Server-Sent Events格式"""
    for token in response:
        yield f"data: {json.dumps({'token': token})}\n\n"
```

## 二、WebSocket流式

```python
class WebSocketHandler:
    """WebSocket流式处理"""
    
    async def handle(self, websocket, path):
        await websocket.accept()
        
        try:
            # 接收请求
            request = await websocket.receive_json()
            prompt = request["prompt"]
            
            # 创建流式生成器
            generator = StreamingGenerator(model, tokenizer)
            
            # 流式发送
            async for token in generator.generate_stream(prompt):
                if token == "[DONE]":
                    await websocket.send_json({"done": True})
                    break
                await websocket.send_json({"token": token})
                
        except Exception as e:
            await websocket.send_json({"error": str(e)})
        finally:
            await websocket.close()
```
