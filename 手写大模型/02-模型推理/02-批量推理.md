## 一、批量处理

```python
class BatchedInference:
    """批量推理引擎"""
    
    def __init__(self, model: LLMModel, max_batch: int = 8):
        self.model = model
        self.max_batch = max_batch
        self.queue = asyncio.Queue()
        self.cache = {}
    
    async def add_request(self, request: InferenceRequest) -> str:
        """添加请求到队列"""
        request_id = generate_id()
        await self.queue.put((request_id, request))
        return request_id
    
    async def process_batch(self):
        """批量处理请求"""
        requests = []
        request_ids = []
        
        # 收集请求
        while len(requests) < self.max_batch and not self.queue.empty():
            request_id, request = await self.queue.get()
            requests.append(request)
            request_ids.append(request_id)
        
        if not requests:
            return
        
        # 填充到相同长度
        inputs = self._pad_sequence([r.input_ids for r in requests])
        
        # 批量推理
        outputs = await self._batch_forward(inputs)
        
        # 返回结果
        for request_id, output in zip(request_ids, outputs):
            self.cache[request_id] = output
```

## 二、动态批量

```python
class DynamicBatching:
    """动态批量调度"""
    
    def __init__(self, engine: BatchedInference):
        self.engine = engine
        self.pending = []
        self.max_wait_time = 0.1  # 100ms
    
    async def schedule(self, request: Request) -> Response:
        request_id = await self.engine.add_request(request)
        
        # 等待结果或超时
        start = time.time()
        while request_id not in self.engine.cache:
            if time.time() - start > self.max_wait_time:
                # 强制执行当前批次
                await self.engine.process_batch()
            await asyncio.sleep(0.01)
        
        return self.engine.cache.pop(request_id)
```
