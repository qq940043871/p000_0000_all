## 一、推理接口

```python
class InferenceEngine:
    """推理引擎核心"""
    
    def __init__(self, model: LLMModel, tokenizer: Tokenizer, config: InferenceConfig):
        self.model = model
        self.tokenizer = tokenizer
        self.config = config
        self.kv_cache = {}
    
    def generate(self, prompt: str, **kwargs) -> str:
        """同步生成"""
        inputs = self.tokenizer.encode(prompt, return_tensors="pt")
        outputs = self.model.generate(**inputs, **kwargs)
        return self.tokenizer.decode(outputs[0], skip_special_tokens=True)
    
    async def generate_async(self, prompt: str, **kwargs) -> str:
        """异步生成"""
        inputs = self.tokenizer.encode(prompt, return_tensors="pt")
        outputs = await asyncio.to_thread(
            self.model.generate, **inputs, **kwargs
        )
        return self.tokenizer.decode(outputs[0], skip_special_tokens=True)
    
    def generate_stream(self, prompt: str, **kwargs):
        """流式生成"""
        inputs = self.tokenizer.encode(prompt, return_tensors="pt")
        for output in self.model.generate_stream(**inputs, **kwargs):
            token = self.tokenizer.decode([output], skip_special_tokens=True)
            yield token
```

## 三、采样策略

```python
class Sampler:
    """采样策略"""
    
    @staticmethod
    def greedy(logits: Tensor) -> int:
        """贪婪采样"""
        return logits.argmax(-1).item()
    
    @staticmethod
    def random(logits: Tensor, temperature: float = 1.0) -> int:
        """随机采样"""
        probs = F.softmax(logits / temperature, dim=-1)
        return torch.multinomial(probs, 1).item()
    
    @staticmethod
    def top_k(logits: Tensor, k: int, temperature: float = 1.0) -> int:
        """Top-K采样"""
        values, indices = torch.topk(logits, k)
        probs = F.softmax(values / temperature, dim=-1)
        return indices[torch.multinomial(probs, 1)].item()
    
    @staticmethod
    def top_p(logits: Tensor, p: float, temperature: float = 1.0) -> int:
        """Top-P（核）采样"""
        probs = F.softmax(logits / temperature, dim=-1)
        sorted_probs, sorted_indices = torch.sort(probs, descending=True)
        cumsum = torch.cumsum(sorted_probs, dim=-1)
        
        # 找到满足概率阈值的最小集合
        mask = cumsum <= p
        mask[..., -1] = True  # 至少选一个
        
        valid_indices = sorted_indices[mask]
        valid_probs = sorted_probs[mask]
        
        return valid_indices[torch.multinomial(valid_probs, 1)].item()
```
