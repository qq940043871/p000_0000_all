## 一、模型格式

```
常见模型格式：
  1. PyTorch (.pt, .pth)
     - 原生格式
     - 需要PyTorch环境
  
  2. Safetensors (.safetensors)
     - 安全序列化
     - 支持懒加载
     - 内存映射
  
  3. GGML/GGUF (.gguf)
     - C++推理
     - 量化支持
     - 跨平台
  
  4. ONNX (.onnx)
     - 跨框架转换
     - 优化部署
```

## 二、模型加载器

```python
class ModelLoader:
    """模型加载器"""
    
    @staticmethod
    def load(config: ModelConfig) -> LLMModel:
        model_path = config.model_path
        model_type = detect_model_type(model_path)
        
        if model_type == "safetensors":
            return SafetensorsLoader.load(config)
        elif model_type == "gguf":
            return GGUFLoader.load(config)
        elif model_type == "onnx":
            return ONNXLoader.load(config)
        else:
            raise ValueError(f"Unsupported model type: {model_type}")


class SafetensorsLoader:
    """Safetensors格式加载"""
    
    @staticmethod
    def load(config: ModelConfig) -> LLMModel:
        from safetensors import safe_open
        
        model = {}
        shard_files = glob.glob(f"{config.model_path}/*.safetensors")
        
        for shard_file in shard_files:
            with safe_open(shard_file, framework="pt", device="cpu") as f:
                for key in f.keys():
                    model[key] = f.get_tensor(key)
        
        # 重组模型结构
        model = ModelLoader.rebuild_model(model, config)
        
        return TransformerModel(model, config)
```

## 三、配置管理

```python
@dataclass
class ModelConfig:
    model_path: str
    tokenizer_path: str = None
    max_length: int = 2048
    device: str = "cuda"
    dtype: str = "float16"
    
    # 量化配置
    quantization: str = None  # "4bit", "8bit"
    compute_dtype: str = "float16"
    
    # 推理配置
    temperature: float = 0.7
    top_p: float = 0.9
    top_k: int = 50
    repetition_penalty: float = 1.0
    
    @property
    def device_map(self) -> Dict[str, int]:
        """自动计算设备映射"""
        if self.device == "cuda":
            num_gpus = torch.cuda.device_count()
            return {f"cuda:{i % num_gpus}": "auto" for i in range(32)}
        return {"cpu": "auto"}
```
