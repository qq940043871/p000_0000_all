## 一、会话管理

```python
class ConversationManager:
    """会话管理器"""
    
    def __init__(self, storage: SessionStorage):
        self.storage = storage
        self.sessions: Dict[str, Conversation] = {}
        self.max_history = 100
    
    def get_or_create(self, session_id: str) -> Conversation:
        if session_id not in self.sessions:
            self.sessions[session_id] = self.storage.load(session_id)
        return self.sessions[session_id]
    
    def add_message(self, session_id: str, role: str, content: str):
        session = self.get_or_create(session_id)
        session.add_message(role, content)
        self.storage.save(session)
    
    def get_history(self, session_id: str, limit: int = None) -> List[Dict]:
        session = self.get_or_create(session_id)
        messages = session.messages
        if limit:
            messages = messages[-limit:]
        return messages


class Conversation:
    """对话上下文"""
    
    def __init__(self, session_id: str):
        self.session_id = session_id
        self.messages: List[Message] = []
        self.metadata: Dict = {}
        self.created_at = datetime.now()
        self.updated_at = datetime.now()
    
    def add_message(self, role: str, content: str, name: str = None):
        self.messages.append(Message(
            role=role,
            content=content,
            name=name,
            timestamp=datetime.now()
        ))
        self.updated_at = datetime.now()
    
    def to_openai_format(self) -> List[Dict]:
        return [
            {"role": m.role, "content": m.content, "name": m.name}
            for m in self.messages
        ]
```

## 二、消息格式

```python
@dataclass
class Message:
    role: str  # "user", "assistant", "system", "function"
    content: str
    name: str = None
    function_call: Dict = None
    timestamp: datetime = None
    
    def to_dict(self) -> Dict:
        result = {"role": self.role, "content": self.content}
        if self.name:
            result["name"] = self.name
        if self.function_call:
            result["function_call"] = self.function_call
        return result


@dataclass
class ChatRequest:
    messages: List[Dict]
    session_id: str = None
    model: str = "gpt-3.5-turbo"
    temperature: float = 0.7
    max_tokens: int = 1000
    functions: List[Dict] = None
    function_call: str = None  # "auto", "none"
```
