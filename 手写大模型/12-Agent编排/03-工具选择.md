## 一、工具路由器

```python
class ToolRouter:
    """工具路由器"""
    
    def __init__(self, tools: List[Tool], embedder: Embedder):
        self.tools = tools
        self.embedder = embedder
    
    async def route(self, query: str) -> List[Tool]:
        """路由到合适工具"""
        # 1. 语义匹配
        query_emb = await self.embedder.embed(query)
        
        scores = []
        for tool in self.tools:
            tool_emb = await self.embedder.embed(tool.description)
            score = cosine_similarity(query_emb, tool_emb)
            scores.append((tool, score))
        
        # 2. 选择得分最高的工具
        scores.sort(key=lambda x: x[1], reverse=True)
        
        # 3. 阈值过滤
        return [tool for tool, score in scores if score > 0.7]
```
