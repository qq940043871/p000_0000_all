## 一、混合检索

```python
class HybridRetriever:
    """混合检索器"""
    
    def __init__(self, vector_store: VectorStore,
                 bm25: BM25, reranker: Reranker):
        self.vector_store = vector_store
        self.bm25 = bm25
        self.reranker = reranker
    
    async def retrieve(self, query: str, top_k: int = 10) -> List[Document]:
        # 1. 向量检索
        vector_results = await self.vector_store.search(query, top_k * 2)
        
        # 2. BM25检索
        bm25_results = self.bm25.search(query, top_k * 2)
        
        # 3. 融合结果
        fused = self._reciprocal_rank_fusion(
            [vector_results, bm25_results]
        )
        
        # 4. 重排序
        reranked = await self.reranker.rerank(query, fused, top_k)
        
        return reranked
    
    def _reciprocal_rank_fusion(self, 
                               result_lists: List[List[Document]],
                               k: int = 60) -> List[Document]:
        """RRF融合"""
        doc_scores = defaultdict(float)
        doc_sources = defaultdict(list)
        
        for results in result_lists:
            for rank, doc in enumerate(results):
                score = 1.0 / (k + rank + 1)
                doc_scores[doc.id] += score
                doc_sources[doc.id].append(doc)
        
        sorted_docs = sorted(
            doc_scores.items(), key=lambda x: x[1], reverse=True
        )
        
        return [doc_sources[doc_id][0] for doc_id, _ in sorted_docs]
```
