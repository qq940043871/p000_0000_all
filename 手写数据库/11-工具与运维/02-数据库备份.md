## 一、备份工具

```java
public class BackupTool {
    private Catalog catalog;
    private DiskManager diskManager;
    
    public void backup(String targetPath) throws Exception {
        File target = new File(targetPath);
        target.mkdirs();
        
        // 备份元数据
        backupMetadata(target);
        
        // 备份所有表
        for (String tableName : catalog.listTables()) {
            backupTable(tableName, target);
        }
        
        // 备份日志
        backupLogs(target);
        
        System.out.println("Backup completed: " + targetPath);
    }
    
    private void backupTable(String tableName, File target) throws Exception {
        TableSchema schema = catalog.getTableSchema(tableName);
        String backupFile = target.getPath() + "/" + tableName + ".sql";
        
        try (PrintWriter writer = new PrintWriter(backupFile)) {
            // 生成CREATE TABLE
            writer.println("-- Table: " + tableName);
            writer.println(getCreateTableSQL(schema));
            writer.println();
            
            // 导出数据
            for (Row row : tableScanner.scan(tableName)) {
                writer.println(getInsertSQL(tableName, schema, row));
            }
        }
    }
    
    private String getInsertSQL(String tableName, TableSchema schema, Row row) {
        StringBuilder sb = new StringBuilder();
        sb.append("INSERT INTO ").append(tableName).append(" (");
        
        List<String> columns = new ArrayList<>();
        List<String> values = new ArrayList<>();
        
        for (int i = 0; i < schema.getColumnCount(); i++) {
            columns.add(schema.getColumn(i).getName());
            Object val = row.getValue(i);
            values.add(val != null ? "'" + escapeSQL(val.toString()) + "'" : "NULL");
        }
        
        sb.append(String.join(", ", columns));
        sb.append(") VALUES (");
        sb.append(String.join(", ", values));
        sb.append(");");
        
        return sb.toString();
    }
}
```

## 二、恢复工具

```java
public class RestoreTool {
    private SqlParser parser;
    private DDLExecutor ddlExecutor;
    private DMLExecutor dmlExecutor;
    
    public void restore(String backupPath) throws Exception {
        File backupDir = new File(backupPath);
        
        // 按顺序执行SQL文件
        File[] files = backupDir.listFiles((dir, name) -> name.endsWith(".sql"));
        Arrays.sort(files);
        
        for (File file : files) {
            System.out.println("Restoring: " + file.getName());
            restoreFile(file);
        }
    }
    
    private void restoreFile(File file) throws Exception {
        try (BufferedReader reader = new BufferedReader(new FileReader(file))) {
            StringBuilder sql = new StringBuilder();
            String line;
            
            while ((line = reader.readLine()) != null) {
                if (line.endsWith(";")) {
                    sql.append(line, 0, line.length() - 1);
                    executeSQL(sql.toString());
                    sql.setLength(0);
                } else {
                    sql.append(line).append(" ");
                }
            }
        }
    }
}
```
