```java
public class PerformanceMonitor {
    private Map<String, QueryStats> queryStats = new ConcurrentHashMap<>();
    private AtomicLong totalQueries = new AtomicLong(0);
    private AtomicLong totalTime = new AtomicLong(0);
    
    public void recordQuery(String sql, long elapsedMs) {
        totalQueries.incrementAndGet();
        totalTime.addAndGet(elapsedMs);
        
        String key = normalizeSQL(sql);
        queryStats.computeIfAbsent(key, k -> new QueryStats())
            .record(elapsedMs);
    }
    
    public void printStatistics() {
        System.out.println("\n=== Query Statistics ===");
        System.out.println("Total Queries: " + totalQueries.get());
        System.out.println("Average Time: " + 
            (totalQueries.get() > 0 ? totalTime.get() / totalQueries.get() : 0) + "ms");
        System.out.println();
        
        System.out.println("Top 10 Slow Queries:");
        queryStats.values().stream()
            .sorted((a, b) -> Long.compare(b.getMaxTime(), a.getMaxTime()))
            .limit(10)
            .forEach(stats -> {
                System.out.printf("  %s: avg=%dms, max=%dms, count=%d%n",
                    stats.getSql(), stats.getAvgTime(), stats.getMaxTime(), stats.getCount());
            });
    }
}

public class QueryStats {
    private String sql;
    private long count;
    private long totalTime;
    private long maxTime;
    private long minTime = Long.MAX_VALUE;
    
    public void record(long elapsedMs) {
        count++;
        totalTime += elapsedMs;
        maxTime = Math.max(maxTime, elapsedMs);
        minTime = Math.min(minTime, elapsedMs);
    }
    
    public long getAvgTime() {
        return count > 0 ? totalTime / count : 0;
    }
}
```

## 二、缓冲池监控

```java
public class BufferPoolMonitor {
    private BufferPool bufferPool;
    
    public BufferPoolStats getStats() {
        BufferPoolStats stats = new BufferPoolStats();
        
        stats.setPoolSize(bufferPool.getPoolSize());
        stats.setUsedPages(bufferPool.getUsedPages());
        stats.setDirtyPages(bufferPool.getDirtyPageCount());
        stats.setHitCount(bufferPool.getHitCount());
        stats.setMissCount(bufferPool.getMissCount());
        
        double hitRate = (double) stats.getHitCount() / 
            (stats.getHitCount() + stats.getMissCount()) * 100;
        stats.setHitRate(hitRate);
        
        return stats;
    }
    
    public void printReport() {
        BufferPoolStats stats = getStats();
        
        System.out.println("\n=== Buffer Pool Statistics ===");
        System.out.printf("Pool Size: %d pages%n", stats.getPoolSize());
        System.out.printf("Used Pages: %d (%.1f%%)%n", 
            stats.getUsedPages(), 
            stats.getUsedPages() * 100.0 / stats.getPoolSize());
        System.out.printf("Dirty Pages: %d%n", stats.getDirtyPages());
        System.out.printf("Hit Rate: %.2f%%%n", stats.getHitRate());
        System.out.printf("Hits: %d, Misses: %d%n", 
            stats.getHitCount(), stats.getMissCount());
    }
}
```

## 三、会话监控

```java
public class SessionMonitor {
    private Map<Long, SessionInfo> sessions = new ConcurrentHashMap<>();
    
    public void registerSession(long txId, String sql) {
        SessionInfo info = new SessionInfo();
        info.setTransactionId(txId);
        info.setStartTime(System.currentTimeMillis());
        info.setCurrentSQL(sql);
        sessions.put(txId, info);
    }
    
    public void unregisterSession(long txId) {
        SessionInfo info = sessions.remove(txId);
        if (info != null) {
            info.setEndTime(System.currentTimeMillis());
        }
    }
    
    public void printActiveSessions() {
        System.out.println("\n=== Active Sessions ===");
        for (SessionInfo info : sessions.values()) {
            long elapsed = System.currentTimeMillis() - info.getStartTime();
            System.out.printf("[%d] %s - %dms - %s%n",
                info.getTransactionId(),
                info.getState(),
                elapsed,
                info.getCurrentSQL());
        }
    }
}
```
