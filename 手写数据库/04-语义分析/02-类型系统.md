## 一、类型检查

```java
public class TypeChecker {
    
    public boolean isCompatible(DataType left, DataType right) {
        if (left == right) return true;
        
        // 数值类型兼容
        if (isNumeric(left) && isNumeric(right)) return true;
        
        // 字符串类型兼容
        if (isString(left) && isString(right)) return true;
        
        return false;
    }
    
    public DataType getCommonType(DataType a, DataType b) {
        if (a == b) return a;
        
        // INT + DOUBLE → DOUBLE
        if (a == DataType.INT && b == DataType.DOUBLE) return DataType.DOUBLE;
        if (a == DataType.DOUBLE && b == DataType.INT) return DataType.DOUBLE;
        
        // VARCHAR + CHAR → VARCHAR
        if (isString(a) && isString(b)) return DataType.VARCHAR;
        
        return DataType.VARCHAR; // 默认
    }
    
    private boolean isNumeric(DataType type) {
        return type == DataType.INT || type == DataType.BIGINT ||
               type == DataType.FLOAT || type == DataType.DOUBLE ||
               type == DataType.DECIMAL;
    }
    
    private boolean isString(DataType type) {
        return type == DataType.CHAR || type == DataType.VARCHAR ||
               type == DataType.TEXT;
    }
}
```

## 二、类型转换

```java
public class TypeConverter {
    
    public Object convert(Object value, DataType targetType) {
        if (value == null) return null;
        
        switch (targetType) {
            case INT:
                return toInt(value);
            case BIGINT:
                return toLong(value);
            case DOUBLE:
                return toDouble(value);
            case VARCHAR:
                return toString(value);
            case BOOLEAN:
                return toBoolean(value);
            default:
                return value;
        }
    }
    
    private int toInt(Object value) {
        if (value instanceof Number) {
            return ((Number) value).intValue();
        }
        if (value instanceof String) {
            return Integer.parseInt((String) value);
        }
        if (value instanceof Boolean) {
            return (Boolean) value ? 1 : 0;
        }
        throw new TypeConversionException("Cannot convert to INT: " + value);
    }
    
    private double toDouble(Object value) {
        if (value instanceof Number) {
            return ((Number) value).doubleValue();
        }
        if (value instanceof String) {
            return Double.parseDouble((String) value);
        }
        throw new TypeConversionException("Cannot convert to DOUBLE: " + value);
    }
    
    private String toString(Object value) {
        return String.valueOf(value);
    }
    
    private boolean toBoolean(Object value) {
        if (value instanceof Boolean) return (Boolean) value;
        if (value instanceof Number) return ((Number) value).intValue() != 0;
        if (value instanceof String) {
            String s = ((String) value).toLowerCase();
            return "true".equals(s) || "1".equals(s);
        }
        throw new TypeConversionException("Cannot convert to BOOLEAN: " + value);
    }
}
```

## 三、字面量类型推断

```java
public class LiteralTypeInferrer {
    
    public DataType inferType(Token token) {
        switch (token.type) {
            case INTEGER:
                String numStr = token.value;
                long num = Long.parseLong(numStr);
                if (num > Integer.MAX_VALUE || num < Integer.MIN_VALUE) {
                    return DataType.BIGINT;
                }
                return DataType.INT;
                
            case FLOAT:
                return DataType.DOUBLE;
                
            case STRING:
                // 根据长度判断
                if (token.value.length() <= 255) {
                    return DataType.VARCHAR;
                }
                return DataType.TEXT;
                
            case BOOLEAN:
                return DataType.BOOLEAN;
                
            case NULL:
                return DataType.NULL;
                
            default:
                throw new IllegalArgumentException("Unknown literal type: " + token.type);
        }
    }
}
```
