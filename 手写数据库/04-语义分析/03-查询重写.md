## 一、重写规则

public class QueryRewriter {
    
    // 常量折叠
    public Expression constantFolding(Expression expr) {
        if (expr instanceof BinaryExpression) {
            BinaryExpression bin = (BinaryExpression) expr;
            Expression left = constantFolding(bin.getLeft());
            Expression right = constantFolding(bin.getRight());
            
            if (left instanceof LiteralExpression && right instanceof LiteralExpression) {
                return evaluateConstant(bin.getOp(), 
                    ((LiteralExpression) left).getValue(),
                    ((LiteralExpression) right).getValue());
            }
            
            return new BinaryExpression(bin.getOp(), left, right);
        }
        return expr;
    }
    
    // 谓词下推
    public SelectStatement pushDownPredicates(SelectStatement stmt) {
        if (stmt.getWhere() == null) return stmt;
        
        Expression where = stmt.getWhere();
        TableReference table = stmt.getFrom();
        
        // 将涉及表列的谓词下推到FROM子句
        List<Expression> pushDown = new ArrayList<>();
        List<Expression> remaining = new ArrayList<>();
        
        splitPredicate(where, table, pushDown, remaining);
        
        if (!pushDown.isEmpty()) {
            TableReference newTable = addFilterToTable(table, combinePredicates(pushDown));
            stmt.setFrom(newTable);
            stmt.setWhere(combinePredicates(remaining));
        }
        
        return stmt;
    }
    
    // 子查询扁平化
    public Expression flattenSubqueries(Expression expr) {
        // 简化嵌套子查询
        return expr;
    }
}

## 二、语义检查

```java
public class SemanticChecker {
    private SymbolTable symbolTable;
    private List<String> errors = new ArrayList<>();
    
    public void check(Statement stmt) {
        if (stmt instanceof SelectStatement) {
            checkSelect((SelectStatement) stmt);
        } else if (stmt instanceof InsertStatement) {
            checkInsert((InsertStatement) stmt);
        } else if (stmt instanceof UpdateStatement) {
            checkUpdate((UpdateStatement) stmt);
        } else if (stmt instanceof CreateTableStatement) {
            checkCreateTable((CreateTableStatement) stmt);
        }
    }
    
    private void checkSelect(SelectStatement stmt) {
        // 检查表是否存在
        if (stmt.getFrom() != null) {
            TableReference table = stmt.getFrom();
            if (!symbolTable.existsTable(table.getName())) {
                errors.add("Table '" + table.getName() + "' does not exist");
            }
        }
        
        // 检查列是否有效
        for (Expression col : stmt.getColumns()) {
            if (col instanceof ColumnExpression) {
                ColumnExpression column = (ColumnExpression) col;
                if (!symbolTable.existsColumn(column.getTable(), column.getColumn())) {
                    errors.add("Column '" + column.getColumn() + "' does not exist");
                }
            }
        }
    }
    
    public boolean hasErrors() {
        return !errors.isEmpty();
    }
    
    public List<String> getErrors() {
        return errors;
    }
}
```
