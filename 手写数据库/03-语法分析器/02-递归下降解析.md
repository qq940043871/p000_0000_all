## 一、解析器框架

```java
public class SqlParser {
    private List<Token> tokens;
    private int pos = 0;
    
    public SqlParser(List<Token> tokens) {
        this.tokens = tokens;
    }
    
    public Statement parse() {
        Token first = peek();
        
        switch (first.type) {
            case SELECT: return parseSelect();
            case INSERT: return parseInsert();
            case UPDATE: return parseUpdate();
            case DELETE: return parseDelete();
            case CREATE: return parseCreate();
            case DROP: return parseDrop();
            default: throw new SqlException("Unexpected token: " + first);
        }
    }
    
    private Token peek() {
        return pos < tokens.size() ? tokens.get(pos) : tokens.get(tokens.size() - 1);
    }
    
    private Token advance() {
        return pos < tokens.size() ? tokens.get(pos++) : tokens.get(tokens.size() - 1);
    }
    
    private boolean match(Token.Type... types) {
        for (Token.Type type : types) {
            if (peek().type == type) {
                advance();
                return true;
            }
        }
        return false;
    }
}
```

## 二、SELECT解析

```java
public class SqlParser {
    
    public SelectStatement parseSelect() {
        advance(); // 跳过 SELECT
        
        // 解析列列表
        List<Expression> columns = parseExpressionList();
        if (columns.isEmpty()) {
            columns.add(new StarExpression());
        }
        
        // 解析 FROM
        TableReference table = null;
        if (match(Token.Type.FROM)) {
            table = parseTableReference();
        }
        
        // 解析 WHERE
        Expression where = null;
        if (match(Token.Type.WHERE)) {
            where = parseCondition();
        }
        
        // 解析 GROUP BY
        List<Expression> groupBy = null;
        if (match(Token.Type.GROUP)) {
            expect(Token.Type.BY);
            groupBy = parseExpressionList();
        }
        
        // 解析 HAVING
        Expression having = null;
        if (match(Token.Type.HAVING)) {
            having = parseCondition();
        }
        
        // 解析 ORDER BY
        List<OrderByItem> orderBy = null;
        if (match(Token.Type.ORDER)) {
            expect(Token.Type.BY);
            orderBy = parseOrderBy();
        }
        
        // 解析 LIMIT
        Integer limit = null;
        Integer offset = null;
        if (match(Token.Type.LIMIT)) {
            limit = Integer.parseInt(expect(Token.Type.INTEGER).value);
            if (match(Token.Type.COMMA)) {
                offset = limit;
                limit = Integer.parseInt(expect(Token.Type.INTEGER).value);
            } else if (match(Token.Type.OFFSET)) {
                offset = Integer.parseInt(expect(Token.Type.INTEGER).value);
            }
        }
        
        return new SelectStatement(columns, table, where, groupBy, having, orderBy, limit, offset);
    }
    
    private List<Expression> parseExpressionList() {
        List<Expression> exprs = new ArrayList<>();
        
        do {
            exprs.add(parseExpression());
        } while (match(Token.Type.COMMA));
        
        return exprs;
    }
    
    private Expression parseExpression() {
        return parseOr();
    }
    
    private Expression parseOr() {
        Expression left = parseAnd();
        
        while (match(Token.Type.OR)) {
            Expression right = parseAnd();
            left = new BinaryExpression(BinaryExpression.Type.OR, left, right);
        }
        
        return left;
    }
    
    private Expression parseAnd() {
        Expression left = parseComparison();
        
        while (match(Token.Type.AND)) {
            Expression right = parseComparison();
            left = new BinaryExpression(BinaryExpression.Type.AND, left, right);
        }
        
        return left;
    }
    
    private Expression parseComparison() {
        Expression left = parsePrimary();
        
        Token.Type op = peek().type;
        if (op == Token.Type.EQ || op == Token.Type.NE || 
            op == Token.Type.LT || op == Token.Type.GT ||
            op == Token.Type.LE || op == Token.Type.GE) {
            advance();
            Expression right = parsePrimary();
            return new BinaryExpression(convertOp(op), left, right);
        }
        
        return left;
    }
    
    private Expression parsePrimary() {
        Token token = peek();
        
        if (token.type == Token.Type.IDENTIFIER) {
            advance();
            if (match(Token.Type.DOT)) {
                String table = token.value;
                String column = expect(Token.Type.IDENTIFIER).value;
                return new ColumnExpression(table, column);
            }
            return new ColumnExpression(null, token.value);
        }
        
        if (token.type == Token.Type.INTEGER) {
            advance();
            return new LiteralExpression(Integer.parseInt(token.value));
        }
        
        if (token.type == Token.Type.STRING) {
            advance();
            return new LiteralExpression(token.value);
        }
        
        if (match(Token.Type.LPAREN)) {
            Expression expr = parseExpression();
            expect(Token.Type.RPAREN);
            return expr;
        }
        
        throw new SqlException("Unexpected token: " + token);
    }
}
```
