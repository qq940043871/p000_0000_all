## 一、完整词法分析器

```java
public class SqlLexer {
    private final String input;
    private int pos = 0;
    private int line = 1;
    private int column = 1;
    
    private static final Map<String, Token.Type> KEYWORDS = new HashMap<>();
    
    static {
        KEYWORDS.put("SELECT", Token.Type.SELECT);
        KEYWORDS.put("FROM", Token.Type.FROM);
        KEYWORDS.put("WHERE", Token.Type.WHERE);
        KEYWORDS.put("AND", Token.Type.AND);
        KEYWORDS.put("OR", Token.Type.OR);
        KEYWORDS.put("NOT", Token.Type.NOT);
        KEYWORDS.put("IN", Token.Type.IN);
        KEYWORDS.put("LIKE", Token.Type.LIKE);
        KEYWORDS.put("BETWEEN", Token.Type.BETWEEN);
        KEYWORDS.put("INSERT", Token.Type.INSERT);
        KEYWORDS.put("INTO", Token.Type.INTO);
        KEYWORDS.put("VALUES", Token.Type.VALUES);
        KEYWORDS.put("UPDATE", Token.Type.UPDATE);
        KEYWORDS.put("SET", Token.Type.SET);
        KEYWORDS.put("DELETE", Token.Type.DELETE);
        KEYWORDS.put("CREATE", Token.Type.CREATE);
        KEYWORDS.put("TABLE", Token.Type.TABLE);
        KEYWORDS.put("DROP", Token.Type.DROP);
        KEYWORDS.put("INDEX", Token.Type.INDEX);
        KEYWORDS.put("ON", Token.Type.ON);
        KEYWORDS.put("AS", Token.Type.AS);
        KEYWORDS.put("ASC", Token.Type.ASC);
        KEYWORDS.put("DESC", Token.Type.DESC);
        KEYWORDS.put("LIMIT", Token.Type.LIMIT);
        KEYWORDS.put("OFFSET", Token.Type.OFFSET);
        KEYWORDS.put("NULL", Token.Type.NULL);
        KEYWORDS.put("TRUE", Token.Type.BOOLEAN);
        KEYWORDS.put("FALSE", Token.Type.BOOLEAN);
        KEYWORDS.put("PRIMARY", Token.Type.PRIMARY);
        KEYWORDS.put("KEY", Token.Type.KEY);
        KEYWORDS.put("INT", Token.Type.INT);
        KEYWORDS.put("INTEGER", Token.Type.INT);
        KEYWORDS.put("VARCHAR", Token.Type.VARCHAR);
        KEYWORDS.put("TEXT", Token.Type.TEXT);
        KEYWORDS.put("DATE", Token.Type.DATE);
        KEYWORDS.put("DATETIME", Token.Type.DATETIME);
    }
    
    public SqlLexer(String input) {
        this.input = input;
    }
    
    public List<Token> tokenize() {
        List<Token> tokens = new ArrayList<>();
        
        while (pos < input.length()) {
            skipWhitespace();
            if (pos >= input.length()) break;
            
            char c = current();
            
            if (isIdentifierStart(c)) {
                tokens.add(readIdentifier());
            } else if (isDigit(c)) {
                tokens.add(readNumber());
            } else if (c == '\'' || c == '"') {
                tokens.add(readString());
            } else if (c == '-' && next() == '-') {
                skipLineComment();
            } else if (c == '/' && next() == '*') {
                skipBlockComment();
            } else {
                tokens.add(readOperator());
            }
        }
        
        tokens.add(new Token(Token.Type.EOF, "", pos, line, column));
        return tokens;
    }
    
    private Token readIdentifier() {
        int startPos = pos;
        int startLine = line;
        int startColumn = column;
        
        StringBuilder sb = new StringBuilder();
        while (pos < input.length() && isIdentifierPart(current())) {
            sb.append(current());
            advance();
        }
        
        String text = sb.toString();
        Token.Type type = KEYWORDS.getOrDefault(text.toUpperCase(), Token.Type.IDENTIFIER);
        
        return new Token(type, text, startPos, startLine, startColumn);
    }
    
    private Token readNumber() {
        int startPos = pos;
        StringBuilder sb = new StringBuilder();
        
        while (pos < input.length() && (isDigit(current()) || current() == '.')) {
            sb.append(current());
            advance();
        }
        
        String num = sb.toString();
        boolean isFloat = num.contains(".");
        
        return new Token(isFloat ? Token.Type.FLOAT : Token.Type.INTEGER, num, startPos, line, column);
    }
    
    private Token readString() {
        int startPos = pos;
        char quote = current();
        advance();
        
        StringBuilder sb = new StringBuilder();
        while (pos < input.length() && current() != quote) {
            sb.append(current());
            advance();
        }
        
        if (pos < input.length()) advance(); // 跳过结束引号
        
        return new Token(Token.Type.STRING, sb.toString(), startPos, line, column);
    }
    
    private Token readOperator() {
        int startPos = pos;
        char c = current();
        advance();
        
        switch (c) {
            case '+': return new Token(Token.Type.PLUS, "+", startPos, line, column);
            case '-': return new Token(Token.Type.MINUS, "-", startPos, line, column);
            case '*': return new Token(Token.Type.STAR, "*", startPos, line, column);
            case '/': return new Token(Token.Type.SLASH, "/", startPos, line, column);
            case '%': return new Token(Token.Type.PERCENT, "%", startPos, line, column);
            case '(': return new Token(Token.Type.LPAREN, "(", startPos, line, column);
            case ')': return new Token(Token.Type.RPAREN, ")", startPos, line, column);
            case ',': return new Token(Token.Type.COMMA, ",", startPos, line, column);
            case '.': return new Token(Token.Type.DOT, ".", startPos, line, column);
            case ';': return new Token(Token.Type.SEMICOLON, ";", startPos, line, column);
            case '=':
                return new Token(Token.Type.EQ, "=", startPos, line, column);
            case '<':
                if (current() == '=') { advance(); return new Token(Token.Type.LE, "<=", startPos, line, column); }
                if (current() == '>') { advance(); return new Token(Token.Type.NE, "<>", startPos, line, column); }
                return new Token(Token.Type.LT, "<", startPos, line, column);
            case '>':
                if (current() == '=') { advance(); return new Token(Token.Type.GE, ">=", startPos, line, column); }
                return new Token(Token.Type.GT, ">", startPos, line, column);
            case '!':
                if (current() == '=') { advance(); return new Token(Token.Type.NE, "!=", startPos, line, column); }
                return new Token(Token.Type.ERROR, "!", startPos, line, column);
            default:
                return new Token(Token.Type.ERROR, String.valueOf(c), startPos, line, column);
        }
    }
    
    private void skipWhitespace() {
        while (pos < input.length() && Character.isWhitespace(current())) {
            advance();
        }
    }
    
    private void skipLineComment() {
        while (pos < input.length() && current() != '\n') advance();
    }
    
    private void skipBlockComment() {
        advance(); advance();
        while (pos < input.length() - 1 && !(current() == '*' && next() == '/')) advance();
        if (pos < input.length()) advance();
        if (pos < input.length()) advance();
    }
    
    private char current() { return pos < input.length() ? input.charAt(pos) : '\0'; }
    private char next() { return pos + 1 < input.length() ? input.charAt(pos + 1) : '\0'; }
    
    private void advance() {
        if (pos < input.length()) {
            if (input.charAt(pos) == '\n') { line++; column = 1; }
            else { column++; }
            pos++;
        }
    }
    
    private boolean isIdentifierStart(char c) { return Character.isLetter(c) || c == '_'; }
    private boolean isIdentifierPart(char c) { return Character.isLetterOrDigit(c) || c == '_'; }
    private boolean isDigit(char c) { return c >= '0' && c <= '9'; }
}
```
