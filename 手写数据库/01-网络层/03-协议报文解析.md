## 一、报文类型

```
MySQL协议报文：
  1. 握手包（服务端→客户端）
     - 协议版本
     - 服务器版本
     - 连接ID
     - 认证插件
  
  2. 请求包（客户端→服务端）
     - 命令类型
     - 参数
  
  3. 响应包（服务端→客户端）
     - OK包/Error包
     - 结果集包
```

## 二、握手认证

```java
public class HandshakeHandler {
    
    public byte[] createHandshakePacket() {
        ByteArrayOutputStream out = new ByteArrayOutputStream();
        
        // 协议版本
        out.write(0x0A);
        
        // 服务器版本
        out.write("8.0.0".getBytes());
        out.write(0x00);
        
        // 连接ID（4字节）
        out.write(intToBytes(12345));
        
        // 挑战码（8字节）
        byte[] scramble = generateScramble();
        out.write(scramble);
        
        // 填充1字节
        out.write(0x00);
        
        // 能力标志（4字节，简化版）
        int capabilities = 0
            | 0x0001  // CLIENT_LONG_PASSWORD
            | 0x0002  // CLIENT_FOUND_ROWS
            | 0x0008  // CLIENT_IGNORE_SPACE
            | 0x0200  // CLIENT_PROTOCOL_41
            | 0x8000; // CLIENT_SECURE_CONNECTION
        out.write(intToBytes(capabilities));
        
        // 字符集
        out.write(0x21);  // utf8_general_ci
        
        // 状态标志
        out.write(new byte[]{0x02, 0x00});
        
        // 认证插件名
        out.write("mysql_native_password".getBytes());
        out.write(0x00);
        
        return out.toByteArray();
    }
    
    public boolean authenticate(AuthPacket authPacket) {
        // 简化认证：用户名root，无密码
        return "root".equals(authPacket.getUsername());
    }
}
```

## 三、命令解析

```java
public class CommandParser {
    
    public static CommandType getCommandType(byte code) {
        switch (code) {
            case 0x01: return CommandType.COM_QUIT;
            case 0x02: return CommandType.COM_INIT_DB;
            case 0x03: return CommandType.COM_QUERY;
            case 0x04: return CommandType.COM_FIELD_LIST;
            case 0x05: return CommandType.COM_CREATE_DB;
            case 0x06: return CommandType.COM_DROP_DB;
            case 0x07: return CommandType.COM_REFRESH;
            case 0x08: return CommandType.COM_SHUTDOWN;
            case 0x0E: return CommandType.COM_PROCESS_INFO;
            case 0x10: return CommandType.COM_STMT_PREPARE;
            case 0x11: return CommandType.COM_STMT_EXECUTE;
            case 0x12: return CommandType.COM_STMT_CLOSE;
            default: return CommandType.COM_UNKNOWN;
        }
    }
}

public enum CommandType {
    COM_QUIT,
    COM_INIT_DB,
    COM_QUERY,
    COM_FIELD_LIST,
    COM_CREATE_DB,
    COM_DROP_DB,
    COM_REFRESH,
    COM_SHUTDOWN,
    COM_PROCESS_INFO,
    COM_STMT_PREPARE,
    COM_STMT_EXECUTE,
    COM_STMT_CLOSE,
    COM_UNKNOWN
}
```

## 四、结果打包

```java
public class ResultPacker {
    
    public byte[] packResultSet(ResultSet result) {
        ByteArrayOutputStream out = new ByteArrayOutputStream();
        
        // 列计数
        packInteger(out, result.getColumnCount());
        
        // 列定义
        for (ColumnDefinition col : result.getColumns()) {
            packColumnDefinition(out, col);
        }
        
        // EOF
        out.write(0xFE);
        out.write(0x00);
        out.write(0x00);
        out.write(0x00);
        
        // 行数据
        while (result.hasNext()) {
            Row row = result.next();
            packRow(out, row, result.getColumns());
        }
        
        // EOF
        out.write(0xFE);
        out.write(0x02);
        out.write(0x00);
        out.write(0x00);
        
        return out.toByteArray();
    }
}
```
