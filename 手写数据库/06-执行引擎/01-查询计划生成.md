## 一、查询计划结构

```
计划树结构：
┌─────────────────────────────────────────┐
│  Project (SELECT id, name)               │
│    └─ NestedLoopJoin                     │
│         ├─ Scan(users) ─→ Index Scan    │
│         └─ Scan(orders) ─→ Full Scan    │
│              └─ Filter (amount > 100)   │
└─────────────────────────────────────────┘
```

## 二、计划节点

```java
public interface PlanNode {
    List<PlanNode> getChildren();
    Schema getOutputSchema();
    double estimateCost();
}

// 扫描节点
public class ScanNode implements PlanNode {
    private String tableName;
    private Schema schema;
    private boolean useIndex;
    private IndexInfo index;
    
    @Override
    public Schema getOutputSchema() {
        return schema;
    }
    
    @Override
    public double estimateCost() {
        if (useIndex) {
            return estimateIndexCost();
        }
        return estimateFullScanCost();
    }
}

// 连接节点
public class JoinNode implements PlanNode {
    private PlanNode left;
    private PlanNode right;
    private Expression joinCondition;
    private JoinType joinType;  // INNER, LEFT, RIGHT, FULL
}

// 聚合节点
public class AggregateNode implements PlanNode {
    private PlanNode child;
    private List<Expression> groupBy;
    private List<AggregateFunction> aggregates;
}
```

## 三、计划生成

```java
public class Planner {
    
    public PlanNode createPlan(SelectStatement stmt) {
        // 1. From子句 → 扫描节点
        PlanNode from = createScanNode(stmt.getFrom());
        
        // 2. Where子句 → 过滤节点
        PlanNode filtered = createFilterNode(from, stmt.getWhere());
        
        // 3. Group By → 聚合节点
        PlanNode grouped = createAggregateNode(filtered, stmt.getGroupBy());
        
        // 4. Select列表 → 投影节点
        PlanNode projected = createProjectNode(grouped, stmt.getColumns());
        
        // 5. Order By → 排序节点
        PlanNode ordered = createSortNode(projected, stmt.getOrderBy());
        
        // 6. Limit → 限制节点
        PlanNode limited = createLimitNode(ordered, stmt.getLimit(), stmt.getOffset());
        
        return limited;
    }
    
    private PlanNode createScanNode(TableReference table) {
        // 检查是否有可用索引
        IndexInfo index = findUsableIndex(table.getName(), whereCondition);
        
        if (index != null) {
            return new IndexScanNode(table, index);
        }
        
        return new TableScanNode(table);
    }
    
    private PlanNode createFilterNode(PlanNode child, Expression condition) {
        if (condition == null) return child;
        return new FilterNode(child, condition);
    }
}
```
