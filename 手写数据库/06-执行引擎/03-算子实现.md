## 一、连接算子

```java
// 嵌套循环连接
public class NestedLoopJoinOperator implements Operator {
    private Operator left;
    private Operator right;
    private Expression condition;
    private Row leftRow;
    
    @Override
    public void open() {
        left.open();
        right.open();
        leftRow = left.next();
    }
    
    @Override
    public Row next() {
        while (true) {
            while (leftRow != null) {
                Row rightRow = right.next();
                
                if (rightRow == null) {
                    // 左表当前行已匹配完，换下一行
                    leftRow = left.next();
                    right.open();  // 重置右表
                    continue;
                }
                
                // 检查连接条件
                Row combined = combineRows(leftRow, rightRow);
                if (Evaluator.evaluate(condition, combined)) {
                    return combined;
                }
            }
            return null;
        }
    }
}

// 哈希连接
public class HashJoinOperator implements Operator {
    private Operator buildSide;
    private Operator probeSide;
    private Expression condition;
    private Map<Object, List<Row>> hashTable;
    
    @Override
    public void open() {
        // 构建哈希表
        hashTable = new HashMap<>();
        buildSide.open();
        
        Row row;
        while ((row = buildSide.next()) != null) {
            Object key = Evaluator.evaluate(getJoinKey(condition, true), row);
            hashTable.computeIfAbsent(key, k -> new ArrayList<>()).add(row);
        }
        buildSide.close();
        
        // 探查阶段
        probeSide.open();
    }
    
    @Override
    public Row next() {
        while ((row = probeSide.next()) != null) {
            Object key = Evaluator.evaluate(getJoinKey(condition, false), row);
            List<Row> matches = hashTable.get(key);
            
            if (matches != null) {
                for (Row buildRow : matches) {
                    return combineRows(buildRow, row);
                }
            }
        }
        return null;
    }
}
```

## 二、排序算子

```java
public class SortOperator implements Operator {
    private Operator child;
    private List<OrderByItem> orderBy;
    private List<Row> buffer;
    private Iterator<Row> iterator;
    
    @Override
    public void open() {
        // 先收集所有数据
        buffer = new ArrayList<>();
        child.open();
        
        Row row;
        while ((row = child.next()) != null) {
            buffer.add(row);
        }
        child.close();
        
        // 排序
        buffer.sort((a, b) -> compare(a, b));
        iterator = buffer.iterator();
    }
    
    @Override
    public Row next() {
        return iterator.hasNext() ? iterator.next() : null;
    }
    
    private int compare(Row a, Row b) {
        for (OrderByItem item : orderBy) {
            int cmp = Comparator.compare(
                a.getValue(item.getColumn()),
                b.getValue(item.getColumn())
            );
            if (item.isDescending()) cmp = -cmp;
            if (cmp != 0) return cmp;
        }
        return 0;
    }
}
```

## 三、执行器框架

```java
public class ExecutionEngine {
    
    public ResultSet execute(PlanNode plan) {
        Operator root = convertToOperator(plan);
        root.open();
        
        List<Row> results = new ArrayList<>();
        Row row;
        
        while ((row = root.next()) != null) {
            results.add(row);
        }
        
        root.close();
        return new ResultSet(results, plan.getOutputSchema());
    }
    
    private Operator convertToOperator(PlanNode node) {
        if (node instanceof ScanNode) {
            return new ScanOperator((ScanNode) node);
        } else if (node instanceof FilterNode) {
            FilterNode filter = (FilterNode) node;
            return new FilterOperator(convertToOperator(filter.getChild()), filter.getCondition());
        } else if (node instanceof ProjectNode) {
            ProjectNode project = (ProjectNode) node;
            return new ProjectOperator(convertToOperator(project.getChild()), project.getColumns());
        } else if (node instanceof JoinNode) {
            return new HashJoinOperator(
                convertToOperator(((JoinNode) node).getLeft()),
                convertToOperator(((JoinNode) node).getRight()),
                ((JoinNode) node).getJoinCondition()
            );
        } else if (node instanceof AggregateNode) {
            return new AggregateOperator(
                convertToOperator(((AggregateNode) node).getChild()),
                ((AggregateNode) node).getAggregates()
            );
        }
        
        throw new UnsupportedOperationException("Unsupported node type: " + node.getClass());
    }
}
```
