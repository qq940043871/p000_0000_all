## 一、火山模型原理

```
火山模型（Iterator Pattern）：
  - 每个算子实现 open() / next() / close()
  - 算子从下往上拉取数据
  - 数据流像"火山喷发"一样向上传递

特点：
  - 简单易实现
  - 延迟执行
  - 内存占用低
```

## 二、算子接口

```java
public interface Operator {
    void open() throws Exception;
    Row next() throws Exception;
    void close() throws Exception;
}

// 扫描算子
public class ScanOperator implements Operator {
    private String tableName;
    private Iterator<Row> iterator;
    
    @Override
    public void open() {
        iterator = tableScanner.scan();
    }
    
    @Override
    public Row next() {
        if (iterator.hasNext()) {
            return iterator.next();
        }
        return null;  // 表示结束
    }
    
    @Override
    public void close() {
        iterator = null;
    }
}

// 过滤算子
public class FilterOperator implements Operator {
    private Operator child;
    private Expression condition;
    
    @Override
    public void open() {
        child.open();
    }
    
    @Override
    public Row next() {
        while (true) {
            Row row = child.next();
            if (row == null) return null;
            
            if (Evaluator.evaluate(condition, row)) {
                return row;
            }
        }
    }
    
    @Override
    public void close() {
        child.close();
    }
}
```

## 三、投影算子

```java
public class ProjectOperator implements Operator {
    private Operator child;
    private List<Expression> expressions;
    
    @Override
    public Row next() {
        Row input = child.next();
        if (input == null) return null;
        
        // 根据表达式计算输出列
        Object[] values = new Object[expressions.size()];
        for (int i = 0; i < expressions.size(); i++) {
            values[i] = Evaluator.evaluate(expressions.get(i), input);
        }
        
        return new Row(values);
    }
}

// 聚合算子
public class AggregateOperator implements Operator {
    private Operator child;
    private List<Expression> groupBy;
    private List<AggregateFunction> aggregates;
    private Map<GroupKey, Aggregator> aggregators;
    
    @Override
    public Row next() {
        // 读取所有数据并聚合
        while (true) {
            Row row = child.next();
            if (row == null) break;
            
            GroupKey key = computeGroupKey(row);
            Aggregator agg = aggregators.computeIfAbsent(key, k -> new Aggregator(aggregates));
            agg.accumulate(row);
        }
        
        // 返回聚合结果
        if (!aggregators.hasNext()) return null;
        return aggregators.next().getResult();
    }
}
```
