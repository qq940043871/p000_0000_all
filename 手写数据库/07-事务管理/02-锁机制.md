## 一、锁类型

```
S锁（Shared Lock，共享锁）：
  - 用于读取操作
  - 多个事务可以同时持有
  - 与S锁兼容，与X锁互斥

X锁（Exclusive Lock，排他锁）：
  - 用于写入操作
  - 排他锁，其他事务不能同时持有
  - 与任何锁都互斥

IS锁（Intention Shared）：
  表示有意获取共享锁

IX锁（Intention Exclusive）：
  表示有意获取排他锁
```

## 二、锁管理器

```java
public class LockManager {
    private Map<ResourceId, LockEntry> lockTable;
    private Map<Long, Set<ResourceId>> transactionLocks;
    private LockWaitGraph waitGraph;
    
    public boolean acquireLock(long txId, ResourceId resource, LockMode mode) {
        synchronized (lockTable) {
            LockEntry entry = lockTable.get(resource);
            
            if (entry == null) {
                // 无锁，直接获取
                entry = new LockEntry(resource);
                lockTable.put(resource, entry);
            }
            
            if (entry.isCompatible(mode)) {
                entry.addLock(txId, mode);
                transactionLocks.computeIfAbsent(txId, k -> new HashSet<>()).add(resource);
                return true;
            }
            
            // 锁冲突，加入等待队列
            return false;
        }
    }
    
    public void releaseLock(long txId, ResourceId resource) {
        synchronized (lockTable) {
            LockEntry entry = lockTable.get(resource);
            if (entry != null) {
                entry.removeLock(txId);
                if (entry.isEmpty()) {
                    lockTable.remove(resource);
                }
                transactionLocks.get(txId).remove(resource);
            }
        }
    }
}

public class LockEntry {
    private ResourceId resource;
    private List<Lock> locks;
    
    public boolean isCompatible(LockMode newMode) {
        for (Lock lock : locks) {
            if (!lock.getMode().isCompatible(newMode)) {
                return false;
            }
        }
        return true;
    }
}
```

## 三、死锁处理

```java
public class DeadLockDetector {
    private LockWaitGraph waitGraph;
    
    public boolean detectDeadLock() {
        return waitGraph.hasCycle();
    }
    
    public Set<Long> findDeadLockTransactions() {
        List<Long> cycle = waitGraph.findCycle();
        return new HashSet<>(cycle);
    }
    
    public void resolveDeadLock(Set<Long> transactions) {
        // 选择最早开始的事务进行回滚
        long victim = transactions.stream()
            .min(Comparator.comparing(this::getTransactionStartTime))
            .orElseThrow();
        
        transactionManager.rollback(victim);
    }
}

public class LockWaitGraph {
    private Map<Long, Set<Long>> edges;  // txId → 等待的事务
    
    public void addEdge(long waiter, long holder) {
        edges.computeIfAbsent(waiter, k -> new HashSet<>()).add(holder);
    }
    
    public boolean hasCycle() {
        Set<Long> visited = new HashSet<>();
        Set<Long> recursion = new HashSet<>();
        
        for (Long txId : edges.keySet()) {
            if (hasCycleDFS(txId, visited, recursion)) {
                return true;
            }
        }
        return false;
    }
}
```
