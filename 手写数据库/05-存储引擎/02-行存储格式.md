## 一、字段编码

```java
public class FieldEncoder {
    
    public static byte[] encode(Object value, ColumnDefinition col) {
        if (value == null) {
            return new byte[0];
        }
        
        switch (col.getType()) {
            case INT:
                return encodeInt(((Number) value).intValue());
            case BIGINT:
                return encodeLong(((Number) value).longValue());
            case VARCHAR:
                return encodeVarchar((String) value);
            case CHAR:
                return encodeChar((String) value, col.getLength());
            case DATE:
                return encodeDate((Date) value);
            default:
                return encodeVarchar(value.toString());
        }
    }
    
    public static byte[] encodeInt(int value) {
        return ByteBuffer.allocate(4).order(ByteOrder.LITTLE_ENDIAN).putInt(value).array();
    }
    
    public static byte[] encodeLong(long value) {
        return ByteBuffer.allocate(8).order(ByteOrder.LITTLE_ENDIAN).putLong(value).array();
    }
    
    public static byte[] encodeVarchar(String value) {
        try {
            return value.getBytes("UTF-8");
        } catch (UnsupportedEncodingException e) {
            return value.getBytes();
        }
    }
    
    public static byte[] encodeChar(String value, int length) {
        byte[] bytes = encodeVarchar(value);
        return Arrays.copyOf(bytes, length);
    }
}
```

## 二、行解析

```java
public class RowParser {
    
    public static Object[] parse(byte[] data, TableSchema schema) {
        List<Object> values = new ArrayList<>();
        
        // 解析NULL位图
        int columnCount = schema.getColumnCount();
        int nullBitmapSize = (columnCount + 7) / 8;
        byte[] nullBitmap = Arrays.copyOfRange(data, 0, nullBitmapSize);
        
        int offset = nullBitmapSize;
        
        // 解析每个字段
        for (int i = 0; i < columnCount; i++) {
            ColumnDefinition col = schema.getColumn(i);
            
            if (isNull(nullBitmap, i)) {
                values.add(null);
            } else if (col.isVariableLength()) {
                // 读取变长字段
                int len = decodeLength(data, offset);
                offset += 2;
                byte[] fieldData = Arrays.copyOfRange(data, offset, offset + len);
                values.add(decode(fieldData, col));
                offset += len;
            } else {
                // 读取固定长度字段
                int len = col.getFixedLength();
                byte[] fieldData = Arrays.copyOfRange(data, offset, offset + len);
                values.add(decode(fieldData, col));
                offset += len;
            }
        }
        
        return values.toArray();
    }
    
    private static boolean isNull(byte[] nullBitmap, int columnIndex) {
        int byteIndex = columnIndex / 8;
        int bitIndex = columnIndex % 8;
        return (nullBitmap[byteIndex] & (1 << bitIndex)) != 0;
    }
}
```
