## 一、B+树结构

```
B+树特点：
  - 所有数据都在叶子节点
  - 叶子节点之间用双向链表连接
  - 内部节点只存储键
  - 查询复杂度：O(log n)

结构示意：
        [50 | 100 | 150]
       /     |      \
   [1-49]  [50-99]  [100-149] [150-200]
    ↓        ↓         ↓         ↓
  数据     数据       数据      数据
```

## 二、节点结构

```java
public class BPlusNode {
    private boolean isLeaf;
    private List<byte[]> keys;       // 索引键
    private List<Long> childPointers; // 子页指针（内部节点）
    private List<RowPointer> rowPointers; // 行指针（叶子节点）
    private long pageId;
    
    public BPlusNode(boolean isLeaf) {
        this.isLeaf = isLeaf;
        this.keys = new ArrayList<>();
        if (isLeaf) {
            this.rowPointers = new ArrayList<>();
        } else {
            this.childPointers = new ArrayList<>();
        }
    }
    
    // 查找键的位置
    public int search(byte[] key, Comparator<byte[]> cmp) {
        int lo = 0, hi = keys.size() - 1;
        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            int cmpResult = cmp.compare(keys.get(mid), key);
            if (cmpResult == 0) return mid;
            if (cmpResult < 0) lo = mid + 1;
            else hi = mid - 1;
        }
        return lo;
    }
}
```

## 三、B+树操作

```java
public class BPlusTree {
    private long rootPageId;
    private BPlusTreeManager manager;
    
    // 插入
    public void insert(byte[] key, RowPointer pointer) {
        BPlusNode root = loadNode(rootPageId);
        
        if (root.isFull()) {
            // 分裂根节点
            BPlusNode newRoot = splitRoot(root);
            rootPageId = newRoot.getPageId();
        }
        
        insertNonFull(root, key, pointer);
    }
    
    private void insertNonFull(BPlusNode node, byte[] key, RowPointer pointer) {
        if (node.isLeaf()) {
            int pos = node.search(key, comparator);
            node.getKeys().add(pos, key);
            node.getRowPointers().add(pos, pointer);
        } else {
            int childPos = node.search(key, comparator);
            BPlusNode child = loadNode(node.getChildPointer(childPos));
            
            if (child.isFull()) {
                splitChild(node, childPos);
                // 重新确定插入位置
                if (comparator.compare(key, node.getKey(childPos)) > 0) {
                    childPos++;
                }
                child = loadNode(node.getChildPointer(childPos));
            }
            
            insertNonFull(child, key, pointer);
        }
    }
    
    // 查找
    public List<RowPointer> search(byte[] key) {
        BPlusNode node = loadNode(rootPageId);
        
        while (!node.isLeaf()) {
            int pos = node.search(key, comparator);
            node = loadNode(node.getChildPointer(pos));
        }
        
        int pos = node.search(key, comparator);
        if (pos < node.getKeys().size() && comparator.compare(node.getKey(pos), key) == 0) {
            return node.getRowPointers().subList(pos, pos + 1);
        }
        return Collections.emptyList();
    }
    
    // 范围查找
    public List<RowPointer> rangeSearch(byte[] startKey, byte[] endKey) {
        List<RowPointer> result = new ArrayList<>();
        BPlusNode leaf = findLeaf(startKey);
        
        while (leaf != null) {
            for (int i = 0; i < leaf.getKeyCount(); i++) {
                byte[] key = leaf.getKey(i);
                if (comparator.compare(key, endKey) > 0) {
                    return result;
                }
                if (comparator.compare(key, startKey) >= 0) {
                    result.add(leaf.getRowPointer(i));
                }
            }
            leaf = leaf.getNextLeaf();
        }
        return result;
    }
}
```
