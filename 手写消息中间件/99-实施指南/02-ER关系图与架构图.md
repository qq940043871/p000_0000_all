## 1. 核心实体关系总览

```plantuml
@startuml ER_OVERVIEW

entity "vhost\n虚拟主机" as VHOST {
  * vhost_id : VARCHAR(64)
  --
  vhost_name : VARCHAR(128)
  description : VARCHAR(512)
  status : ENUM
  created_at : DATETIME
}

entity "exchange_info\n交换机" as EXCHANGE {
  * exchange_id : VARCHAR(64)
  --
  exchange_name : VARCHAR(255)
  exchange_type : ENUM(direct/fanout/topic/headers)
  vhost_id : FK → VHOST
  durable : BOOLEAN
  status : ENUM
}

entity "binding\n绑定关系" as BINDING {
  * binding_id : VARCHAR(64)
  --
  source_exchange_id : FK → EXCHANGE
  destination_type : ENUM
  destination_name : VARCHAR(255)
  binding_key : VARCHAR(1024)
  binding_headers : JSON
  active : BOOLEAN
}

entity "queue_info\n队列" as QUEUE {
  * queue_id : VARCHAR(64)
  --
  queue_name : VARCHAR(255)
  vhost_id : FK → VHOST
  queue_type : ENUM(classic/quorum/stream)
  durable : BOOLEAN
  max_length : BIGINT
  overflow_policy : ENUM
  dlx_exchange_id : FK → EXCHANGE
  status : ENUM
}

entity "topic_info\n主题" as TOPIC {
  * topic_id : VARCHAR(64)
  --
  topic_name : VARCHAR(255)
  vhost_id : FK → VHOST
  topic_type : ENUM
  retention_days : INT
  status : ENUM
}

entity "message\n消息主表" as MSG {
  * message_id : VARCHAR(64)
  --
  exchange_id : FK → EXCHANGE
  routing_key : VARCHAR(1024)
  destination_type : ENUM
  destination_id : FK → QUEUE/TOPIC
  delivery_mode : TINYINT
  priority : TINYINT(0-9)
  message_status : ENUM
  redelivered : BOOLEAN
  redelivery_count : INT
  producer_id : FK → PRODUCER
  transaction_id : FK → TRANSACTION
  created_at : DATETIME
}

entity "consumer_info\n消费者" as CONSUMER {
  * consumer_id : VARCHAR(64)
  --
  consumer_tag : VARCHAR(255)
  vhost_id : FK → VHOST
  connection_id : FK → CONNECTION
  consume_mode : ENUM
  prefetch_count : INT
  auto_ack : BOOLEAN
  status : ENUM
}

entity "consumer_offset\n消费偏移量" as OFFSET {
  * offset_id : VARCHAR(64)
  --
  consumer_id : FK → CONSUMER
  destination_type : ENUM
  destination_id : FK → QUEUE/TOPIC
  offset : BIGINT
  lag : BIGINT
}

entity "producer_info\n生产者" as PRODUCER {
  * producer_id : VARCHAR(64)
  --
  producer_name : VARCHAR(255)
  vhost_id : FK → VHOST
  connection_id : FK → CONNECTION
  publisher_confirms : BOOLEAN
  publisher_returns : BOOLEAN
  status : ENUM
}

entity "transaction\n事务" as TRANSACTION {
  * transaction_id : VARCHAR(64)
  --
  transaction_type : ENUM
  coordinator_id : VARCHAR(64)
  status : ENUM
  started_at : DATETIME
  expired_at : DATETIME
}

entity "cluster_node\n集群节点" as NODE {
  * node_id : VARCHAR(64)
  --
  node_name : VARCHAR(128)
  host : VARCHAR(255)
  cluster_id : FK → CLUSTER
  node_type : ENUM
  role : ENUM
  status : ENUM
}

entity "cluster_config\n集群配置" as CLUSTER {
  * cluster_id : VARCHAR(64)
  --
  cluster_name : VARCHAR(128)
  consensus_algorithm : ENUM
  default_replicas : INT
  status : ENUM
}

entity "user\n用户" as USER {
  * user_id : VARCHAR(64)
  --
  username : VARCHAR(128)
  vhost_id : FK → VHOST
  auth_type : ENUM
  status : ENUM
}

entity "role\n角色" as ROLE {
  * role_id : VARCHAR(64)
  --
  role_name : VARCHAR(128)
  parent_role_id : FK → ROLE
}

entity "permission\n权限" as PERMISSION {
  * permission_id : VARCHAR(64)
  --
  role_id : FK → ROLE
  resource_type : ENUM
  resource_name : VARCHAR(255)
  actions : JSON
}

entity "alert_rule\n告警规则" as ALERT_RULE {
  * rule_id : VARCHAR(64)
  --
  rule_name : VARCHAR(255)
  metric_name : VARCHAR(128)
  condition_type : ENUM
  operator : ENUM
  threshold : DOUBLE
  severity : ENUM
  enabled : BOOLEAN
}

' Relationships
VHOST  ||--o{ EXCHANGE : hosts
VHOST  ||--o{ QUEUE : hosts
VHOST  ||--o{ TOPIC : hosts
VHOST  ||--o{ USER : belongs
VHOST  ||--o{ PRODUCER : belongs
VHOST  ||--o{ CONSUMER : belongs

EXCHANGE ||--o{ BINDING : owns
BINDING }o--|| QUEUE : binds_to
BINDING }o--|| TOPIC : binds_to

QUEUE ||--o{ MSG : receives
TOPIC ||--o{ MSG : receives

EXCHANGE ||--o{ MSG : receives
PRODUCER ||--o{ MSG : produces
CONSUMER ||--o{ MSG : consumes

MSG ||--o{ OFFSET : tracked_by
MSG ||--o{ TRANSACTION : belongs_to

CONNECTION ||--o{ CONSUMER : owns
CONNECTION ||--o{ PRODUCER : owns

CLUSTER ||--o{ NODE : contains

USER ||--o{ ROLE : assigned
ROLE ||--o{ PERMISSION : grants
ROLE ||--o{ ROLE : inherits

QUEUE ||--|| EXCHANGE : dlx_points_to (死信交换机)

@enduml
```

## 2. 消息完整生命周期状态机

```plantuml
@startuml LIFECYCLE

[*] --> Created

Created --> Confirmed : 写入存储成功
Confirmed --> Ready : 进入就绪队列

Ready --> InFlight : 投递到消费者
Ready --> Expired : TTL到期
Ready --> Backlog : 延迟消息未到时间

Backlog --> Ready : 到达可投递时间

InFlight --> Acknowledged : ACK成功
InFlight --> Redelivered : NACK/超时/Reject
InFlight --> Expired : 确认超时

Redelivered --> Ready : 重投入队(重投次数未满)
Redelivered --> DeadLetter : 超过最大重投次数

DeadLetter --> Reprocessed : 重新处理成功
DeadLetter --> Archived : 归档保存
DeadLetter --> Discarded : 丢弃

Expired --> Archived : 超过保留期

Acknowledged --> [*] : 删除
Archived --> [*]
Discarded --> [*]

@enduml
```

## 3. 消息路由架构图

```plantuml
@startuml ROUTING

skinparam backgroundColor #FEFEFE
skinparam componentStyle rectangle

package "Producer" {
  [业务系统A] as PA
  [业务系统B] as PB
}

package "Broker - 协议接入层" {
  [AMQP Gateway] as AMQP
  [MQTT Gateway] as MQTT
  [HTTP Gateway] as HTTP
}

package "Broker - 路由层" {
  [Exchange Router] as ROUTER
  database "exchange_info" as EX
}

package "Broker - 队列引擎" {
  [Queue Engine] as QE
  database "queue_info" as Q
  database "topic_info" as T
}

package "Broker - 存储层" {
  [Storage Engine] as SE
  database "Commit Log" as CL
  database "Message Index" as MI
}

package "Consumer" {
  [Consumer A] as CA
  [Consumer B] as CB
}

PA --> AMQP : 消息
PB --> MQTT : 消息

AMQP --> ROUTER
MQTT --> ROUTER
HTTP --> ROUTER

ROUTER --> EX : 查找交换机
EX --> BINDING : 查找绑定
BINDING --> QE : 路由目标

QE --> Q : 写入队列
QE --> T : 发布主题

Q --> SE : 持久化
T --> SE : 持久化

SE --> CL : 追加写
SE --> MI : 索引更新

Q --> CA : 投递消息
Q --> CB : 负载均衡

CA --> AMQP : ACK/NACK
CB --> MQTT : ACK/NACK

@enduml
```

## 4. 集群高可用架构图

```plantuml
@startuml HA_ARCH

skinparam componentStyle rectangle

[客户端] as C

package "负载均衡器" {
  [VIP / Keepalived] as LB
}

package "集群 (3节点)" {
  package "Node-1" {
    [Broker] as B1
    [Raft Leader] as L1
  }
  package "Node-2" {
    [Broker] as B2
    [Raft Follower] as F2
  }
  package "Node-3" {
    [Broker] as B3
    [Raft Follower] as F3
  }
}

database "存储 (共享)" {
  [Message Store] as MS
  [Metadata DB] as MD
}

C --> LB
LB --> B1
LB --> B2
LB --> B3

B1 --> L1 : Raft Consensus
B2 --> F2 : Raft Consensus
B3 --> F3 : Raft Consensus

L1 --> MS : 同步复制
F2 --> MS : 异步复制
F3 --> MS : 异步复制

L1 --> MD : 元数据写入
F2 --> MD : 元数据同步
F3 --> MD : 元数据同步

note right of L1
  Leader 处理所有写入
  同步复制到 Followers
end note

note left of B3
  Node-3 为 Arbiter 节点
  仅参与选举，不存储数据
end note

@enduml
```

## 5. 死信处理流程图

```plantuml
@startuml DLX_FLOW

|生产者|
start
:发送消息;

|交换机|
:路由到目标队列;

|队列|
:消息积压中;

note right
  重投次数 > max
  或 TTL 过期
end note

|死信交换机|
:接收死信消息;

|死信队列|
:存储死信;

fork
  :自动重试;
fork again
  :告警通知;
fork again
  :人工处理;
end fork

|处理完成|
stop

@enduml
```

## 6. 消费者分组 Rebalance 流程

```plantuml
@startuml REBALANCE

|#wheat|消费者组|

start

:消费者 C1, C2, C3 加入组;

|#lightblue|协调者|

:发起 Rebalance;

:计算分区分配方案;

partition "分配 Phase" {
  :撤销 C1 旧分区 [P0,P1];
  :撤销 C2 旧分区 [P2,P3];
  :撤销 C3 旧分区 [P4];
  
  :分配新分区;
  note right
    C1: [P0,P2,P4]
    C2: [P1]
    C3: [P3]
  end note
}

:发送 Assign 命令;

|#wheat|消费者|

while (处理新分区) is (每个分区)
  :停止消费旧分区;
  :保存 Offset;
  :开始消费新分区;
endwhile

:完成 Rebalance;

|#lightblue|协调者|

:Generation + 1;

stop

@enduml
```
