## 一、指令概述

JVM字节码指令由一个字节的操作码（opcode）和零或多个操作数（operand）组成。共256个操作码，目前已使用约200个。

```
指令格式：
┌────────┬────────────────┐
│ opcode │    operands    │
│  1字节  │   0-3字节     │
└────────┴────────────────┘
```

## 三、指令执行引擎

```java
public class Interpreter {
    
    private Frame currentFrame;
    
    public void interpret(Frame frame) {
        this.currentFrame = frame;
        
        while (frame.pc < frame.codeLength) {
            int opcode = frame.readUnsignedByte();
            execute(opcode);
        }
    }
    
    private void execute(int opcode) {
        switch (opcode) {
            case Opcode.ICONST_0:
                currentFrame.operandStack.pushInt(0);
                break;
            case Opcode.ICONST_1:
                currentFrame.operandStack.pushInt(1);
                break;
            case Opcode.IADD:
                int v2 = currentFrame.operandStack.popInt();
                int v1 = currentFrame.operandStack.popInt();
                currentFrame.operandStack.pushInt(v1 + v2);
                break;
            case Opcode.ISTORE_0:
                int val = currentFrame.operandStack.popInt();
                currentFrame.localVariables.setInt(0, val);
                break;
            case Opcode.ILOAD_0:
                int loadVal = currentFrame.localVariables.getInt(0);
                currentFrame.operandStack.pushInt(loadVal);
                break;
            case Opcode.IRETURN:
                // 返回结果
                break;
            case Opcode.GOTO:
                int offset = currentFrame.readSignedShort();
                currentFrame.pc += offset - 3; // 减去已读取的字节
                break;
            // ... 更多指令
            default:
                throw new RuntimeException("Unknown opcode: " + opcode);
        }
    }
}
```

## 四、方法调用机制

```java
// invokevirtual执行流程
public void invokevirtual(int index) {
    // 1. 解析常量池，获取方法引用
    MethodRef methodRef = constantPool.getMethodRef(index);
    
    // 2. 如果是接口方法，跳转到invokeinterface
    if (methodRef.isInterface()) {
        invokeinterface(methodRef);
        return;
    }
    
    // 3. 解析方法
    Method method = methodRef.resolveMethod();
    
    // 4. 如果是实例构造器、父类方法、私有方法
    if (method.isConstructor() || 
        method.isPrivate() || 
        method.isProtected()) {
        invokespecial(methodRef);
        return;
    }
    
    // 5. 否则进行虚分派（动态绑定）
    int argCount = method.getArgCount();
    Object ref = currentFrame.operandStack.getRef(argCount - 1);
    
    // 在对象的实际类中查找方法
    Method actualMethod = ref.getClass().lookupMethod(method.getName());
    
    // 调用方法
    invokeMethod(actualMethod);
}

// invokespecial执行流程
public void invokespecial(int index) {
    MethodRef methodRef = constantPool.getMethodRef(index);
    Method method = methodRef.resolveMethod();
    invokeMethod(method);
}
```

## 五、同步机制

```java
// 方法级同步
// 方法的access_flags中有ACC_SYNCHRONIZED标志
public void invokeMethod(Method method) {
    if (method.isSynchronized()) {
        Object lock = method.isStatic() 
            ? method.getClass()  // 静态方法：锁住类
            : currentFrame.localVariables.getRef(0); // 实例方法：锁住this
        
        synchronized (lock) {
            invokeNative(method);
        }
    } else {
        invokeNative(method);
    }
}

// 代码块同步
public void monitorenter() {
    Object ref = currentFrame.operandStack.popRef();
    ref.enterMonitor(); // 获取锁
}

public void monitorexit() {
    Object ref = currentFrame.operandStack.popRef();
    ref.exitMonitor(); // 释放锁
}
```

## 六、异常处理

```java
// 异常表结构
public class ExceptionTableEntry {
    public int startPc;
    public int endPc;
    public int handlerPc;
    public int catchType;  // 0表示finally
}

// athrow指令
public void athrow() {
    // 1. 弹出异常对象
    Object exception = currentFrame.operandStack.popRef();
    
    if (exception == null) {
        throw new NullPointerException();
    }
    
    // 2. 查找异常处理器
    ExceptionTableEntry handler = 
        findExceptionHandler(exception.getClass());
    
    if (handler != null) {
        // 3. 跳转到处理器
        currentFrame.operandStack.pushRef(exception);
        currentFrame.pc = handler.handlerPc;
    } else {
        // 4. 异常未处理，向上传播
        unwindAndPropagate(exception);
    }
}

private ExceptionTableEntry findExceptionHandler(Class<?> exceptionClass) {
    ExceptionTableEntry[] table = currentFrame.method.getExceptionTable();
    
    for (ExceptionTableEntry entry : table) {
        if (currentFrame.pc >= entry.startPc && 
            currentFrame.pc < entry.endPc) {
            
            if (entry.catchType == 0) {
                // catch-all (finally块)
                return entry;
            }
            
            // 检查类型匹配
            Class<?> catchClass = 
                currentFrame.constantPool.getClass(entry.catchType);
            
            if (catchClass.isAssignableFrom(exceptionClass)) {
                return entry;
            }
        }
    }
    return null;
}
```

## 七、总结

字节码指令集是JVM执行引擎的核心。理解这些指令对于调试、性能优化和理解Java语言特性都至关重要。
