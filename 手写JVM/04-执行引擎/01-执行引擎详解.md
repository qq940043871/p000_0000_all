## 二、解释器实现

```java
public class Interpreter {
    
    private JVMStack stack;
    private Heap heap;
    private MethodArea methodArea;
    
    public Object interpret(Method method, Object[] args) {
        
        // 1. 创建栈帧
        Frame frame = new Frame(method, args);
        
        // 2. 循环执行字节码
        while (frame.hasMoreInstructions()) {
            
            // 3. 获取指令
            int opcode = frame.readOpcode();
            
            // 4. 执行指令
            execute(frame, opcode);
        }
        
        // 5. 返回结果
        return frame.getReturnValue();
    }
    
    private void execute(Frame frame, int opcode) {
        switch (opcode) {
            case Opcode.ICONST_0:
                frame.pushInt(0);
                break;
                
            case Opcode.ICONST_1:
                frame.pushInt(1);
                break;
                
            case Opcode.IADD:
                int v2 = frame.popInt();
                int v1 = frame.popInt();
                frame.pushInt(v1 + v2);
                break;
                
            case Opcode.ISTORE_0:
                int value = frame.popInt();
                frame.setLocal(0, value);
                break;
                
            case Opcode.ILOAD_0:
                int local = frame.getLocalInt(0);
                frame.pushInt(local);
                break;
                
            case Opcode.IRETURN:
                frame.setReturnValue(frame.popInt());
                frame.setPC(Integer.MAX_VALUE); // 退出循环
                break;
                
            // ... 更多指令
        }
    }
}
```

## 三、栈帧结构

```java
public class Frame {
    
    private int maxLocals;
    private int maxStack;
    private Object[] localVariables;
    private Object[] operandStack;
    private int stackTop;
    
    private Method method;
    private int pc;
    private Frame nextFrame;  // 调用者的栈帧
    
    public Frame(Method method, Object[] args) {
        this.method = method;
        this.maxLocals = method.getMaxLocals();
        this.maxStack = method.getMaxStack();
        
        this.localVariables = new Object[maxLocals];
        this.operandStack = new Object[maxStack];
        this.stackTop = -1;
        this.pc = 0;
        
        // 初始化局部变量（this和参数）
        initLocalVariables(args);
    }
    
    private void initLocalVariables(Object[] args) {
        int index = 0;
        
        // this引用（实例方法）
        if (!method.isStatic()) {
            localVariables[index++] = args[0];
        }
        
        // 方法参数
        for (int i = 0; i < args.length; i++) {
            localVariables[index++] = args[i];
        }
    }
    
    public int readOpcode() {
        return method.getCode()[pc++];
    }
    
    public int readInt8() {
        return method.getCode()[pc++];
    }
    
    public int readInt16() {
        int b1 = method.getCode()[pc++] & 0xFF;
        int b2 = method.getCode()[pc++] & 0xFF;
        return (b1 << 8) | b2;
    }
    
    public int readInt32() {
        int b1 = method.getCode()[pc++] & 0xFF;
        int b2 = method.getCode()[pc++] & 0xFF;
        int b3 = method.getCode()[pc++] & 0xFF;
        int b4 = method.getCode()[pc++] & 0xFF;
        return (b1 << 24) | (b2 << 16) | (b3 << 8) | b4;
    }
    
    // 操作数栈操作
    public void pushInt(int value) {
        operandStack[++stackTop] = value;
    }
    
    public int popInt() {
        return (int) operandStack[stackTop--];
    }
    
    public void pushRef(Object ref) {
        operandStack[++stackTop] = ref;
    }
    
    public Object popRef() {
        return operandStack[stackTop--];
    }
    
    public boolean hasMoreInstructions() {
        return pc < method.getCodeLength();
    }
}
```

## 四、方法调用

```java
public class MethodInvoker {
    
    public Object invokeMethod(String className, String methodName, 
                              Object instance, Object[] args) {
        
        // 1. 查找方法
        Method method = MethodArea.findMethod(className, methodName);
        
        // 2. 如果是虚方法，需要动态分派
        if (method.isVirtual()) {
            method = resolveVirtualMethod(instance.getClass(), methodName);
        }
        
        // 3. 创建新栈帧
        Frame newFrame = new Frame(method, args);
        
        // 4. 保存当前帧
        Frame currentFrame = getCurrentFrame();
        newFrame.setNextFrame(currentFrame);
        
        // 5. 切换到新帧
        setCurrentFrame(newFrame);
        
        // 6. 执行方法
        Object result = interpreter.interpret(method, args);
        
        // 7. 恢复原帧
        setCurrentFrame(currentFrame);
        
        return result;
    }
    
    // 虚方法分派
    private Method resolveVirtualMethod(Class<?> actualClass, String methodName) {
        // 在实际类及其父类中查找方法
        Class<?> clazz = actualClass;
        while (clazz != null) {
            Method method = clazz.getDeclaredMethod(methodName);
            if (method != null) {
                return method;
            }
            clazz = clazz.getSuperclass();
        }
        throw new NoSuchMethodError(methodName);
    }
}
```
