## 一、核心概念

### 语义理解原理

```
文本 → 分词/编码 → Transformer编码 → 语义向量 → 相似度计算/匹配

核心任务：
1. 语义相似度：计算两个文本的语义距离
2. 文本匹配：判断两个文本是否匹配
3. 语义搜索：在大量文本中找到最相关的
4. 问答匹配：找到问题的最佳答案
```

## 二、常用类库

### 1. Sentence-Transformers

**特点**：语义向量最全、支持多语言、效果最好

```bash
pip install sentence-transformers
```

```python
from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

# 加载模型
model = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')

# 编码单个文本
embedding = model.encode("我喜欢深度学习")
print(f"向量维度: {embedding.shape}")  # (384,)

# 编码多个文本
sentences = [
    "人工智能将改变世界",
    "机器学习是AI的核心技术",
    "深度学习在图像识别上很强大",
    "今天天气真好"
]

embeddings = model.encode(sentences)
print(f"向量矩阵: {embeddings.shape}")  # (4, 384)

# 计算余弦相似度
similarity = cosine_similarity([embeddings[0]], [embeddings[1]])[0][0]
print(f"相似度: {similarity:.4f}")  # 0.7xx

# 语义搜索
query = "什么是机器学习？"
query_emb = model.encode(query)

# 计算与所有句子的相似度
similarities = cosine_similarity([query_emb], embeddings)[0]
top_idx = np.argsort(similarities)[::-1]

print(f"\n搜索: {query}")
for idx in top_idx[:3]:
    print(f"  {similarities[idx]:.4f} - {sentences[idx]}")
```

### 2. LangChain

**特点**：LLM应用开发框架、支持多种LLM

```bash
pip install langchain langchain-community
```

```python
from langchain.embeddings import HuggingFaceEmbeddings
from langchain.vectorstores import Chroma
from langchain.document_loaders import TextLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter

# 加载文档
loader = TextLoader("document.txt")
documents = loader.load()

# 分割文档
splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=50)
texts = splitter.split_documents(documents)

# 创建向量数据库
embeddings = HuggingFaceEmbeddings(model_name='paraphrase-multilingual-MiniLM-L12-v2')
db = Chroma.from_documents(texts, embeddings, persist_directory="db")

# 语义检索
query = "LangChain有哪些功能？"
docs = db.similarity_search(query, k=3)

print("检索结果:")
for i, doc in enumerate(docs):
    print(f"{i+1}. {doc.page_content[:100]}...")

# 带分数的检索
results = db.similarity_search_with_score(query, k=3)
for doc, score in results:
    print(f"分数: {score:.4f}, 内容: {doc.page_content[:50]}...")

# 保存数据库
db.persist()
```

### 3. FAISS (Facebook)

**特点**：高效向量检索、支持亿级向量

```bash
pip install faiss-cpu  # 或 faiss-gpu
```

```python
import numpy as np
import faiss

# 生成示例数据(10000个128维向量)
dimension = 128
num_vectors = 10000
np.random.seed(42)

vectors = np.random.random((num_vectors, dimension)).astype('float32')

# 构建索引
index = faiss.IndexFlatL2(dimension)  # L2距离
# 或使用内积索引
# index = faiss.IndexFlatIP(dimension)  # 余弦相似度

index.add(vectors)
print(f"索引中向量数: {index.ntotal}")

# 查询
query = np.random.random((5, dimension)).astype('float32')
distances, indices = index.search(query, k=5)

print(f"\n查询5个向量，最近邻:")
for i, (dists, idxs) in enumerate(zip(distances, indices)):
    print(f"  Query {i}: 最近邻索引={idxs[0]}, 距离={dists[0]:.4f}")

# IVF索引(加速)
nlist = 100  # 聚类中心数
quantizer = faiss.IndexFlatL2(dimension)
index = faiss.IndexIVFFlat(quantizer, dimension, nlist)
index.train(vectors)
index.add(vectors)

# 精确度 vs 速度
index.nprobe = 10  # 搜索的聚类数
distances, indices = index.search(query, k=5)
```

### 4. Milvus

**特点**：云原生向量数据库、支持分布式

```python
# 安装
# pip install pymilvus

from pymilvus import connections, Collection, CollectionSchema, FieldSchema, DataType, utility

# 连接
connections.connect(host='localhost', port='19530')

# 定义schema
fields = [
    FieldSchema(name="id", dtype=DataType.INT64, is_primary=True),
    FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=384),
    FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=65535)
]
schema = CollectionSchema(fields=fields, description="文档集合")

# 创建collection
collection = Collection(name="documents", schema=schema)

# 创建索引
index_params = {"index_type": "IVF_FLAT", "params": {"nlist": 128}, "metric_type": "L2"}
collection.create_index(field_name="embedding", index_params=index_params)

# 插入数据
from sentence_transformers import SentenceTransformer

model = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')
texts = ["文本1", "文本2", "文本3"]
embeddings = model.encode(texts).tolist()

data = [[1, 2, 3], embeddings, texts]
collection.insert(data)
collection.flush()

# 搜索
query_embedding = model.encode(["查询文本"]).tolist()
search_params = {"metric_type": "L2", "params": {"nprobe": 10}}
results = collection.search(query_embedding, "embedding", search_params, limit=5)

for result in results[0]:
    print(f"ID: {result.id}, 距离: {result.distance}, 文本: {result.entity.get('text')}")
```

## 三、语义匹配完整示例

```python
from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

class SemanticMatcher:
    def __init__(self, model_name='paraphrase-multilingual-MiniLM-L12-v2'):
        self.model = SentenceTransformer(model_name)
    
    def encode(self, texts):
        return self.model.encode(texts)
    
    def similarity(self, text1, text2):
        emb1, emb2 = self.encode([text1, text2])
        return cosine_similarity([emb1], [emb2])[0][0]
    
    def find_similar(self, query, candidates, top_k=5):
        query_emb = self.encode([query])
        cand_embs = self.encode(candidates)
        
        similarities = cosine_similarity(query_emb, cand_embs)[0]
        top_indices = np.argsort(similarities)[::-1][:top_k]
        
        return [(candidates[i], similarities[i]) for i in top_indices]

# 使用
matcher = SemanticMatcher()

# 测试
corpus = [
    "人工智能正在改变我们的生活方式",
    "机器学习是实现AI的重要方法",
    "深度学习在计算机视觉领域应用广泛",
    "自然语言处理技术日益成熟",
    "今天的天气非常舒适"
]

# 查询
query = "AI和机器学习有什么关系？"
results = matcher.find_similar(query, corpus, top_k=3)

print(f"查询: {query}\n")
for text, score in results:
    print(f"相似度: {score:.4f} - {text}")
```
