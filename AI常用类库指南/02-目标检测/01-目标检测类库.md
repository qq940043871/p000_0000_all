## 一、核心概念

### 目标检测原理

```
输入图像 → 特征提取(CNN) → 区域建议/RPN → 分类+回归 → 输出(类别, 位置)

检测任务：
1. 分类：判断目标属于哪个类别
2. 定位：输出目标的边界框(Bounding Box)
3. 置信度：预测的可信程度
```

## 二、常用类库

### 1. Ultralytics YOLO

**特点**：速度快、部署简单、效果好

```bash
pip install ultralytics
```

```python
from ultralytics import YOLO
import cv2

# 加载模型(yolov8n/yolov8s/yolov8m/yolov8l/yolov8x)
model = YOLO('yolov8n.pt')

# 推理
results = model('bus.jpg')
results = model('video.mp4', save=True)  # 视频

# 处理结果
for r in results:
    # 边界框
    boxes = r.boxes
    for box in boxes:
        x1, y1, x2, y2 = box.xyxy[0].tolist()
        confidence = box.conf[0].item()
        class_id = int(box.cls[0])
        class_name = model.names[class_id]
        
        print(f"{class_name}: {confidence:.2%} at [{x1:.0f},{y1:.0f},{x2:.0f},{y2:.0f}]")
    
    # 绘制结果
    annotated_frame = r.plot()
    cv2.imshow('YOLOv8', annotated_frame)
    cv2.waitKey(0)

# 实时检测
model = YOLO('yolov8n.pt')
camera = cv2.VideoCapture(0)

while camera.isOpened():
    ret, frame = camera.read()
    results = model(frame)
    annotated = results[0].plot()
    cv2.imshow('Camera', annotated)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

camera.release()
cv2.destroyAllWindows()
```

### 2. Detectron2 (Meta)

**特点**：功能强大，支持多种检测任务

```bash
pip install detectron2
# 需要从源码安装
python -m pip install 'git+https://github.com/facebookresearch/detectron2.git'
```

```python
import torch, cv2
from detectron2.config import get_cfg
from detectron2.engine import DefaultPredictor
from detectron2.model_zoo import model_zoo

# 配置
cfg = get_cfg()
cfg.merge_from_file(model_zoo.get_config_file("COCO-Detection/faster_rcnn_R_50_FPN_3x.yaml"))
cfg.MODEL.WEIGHTS = model_zoo.get_checkpoint_url("COCO-Detection/faster_rcnn_R_50_FPN_3x.yaml")
cfg.MODEL.DEVICE = 'cuda' if torch.cuda.is_available() else 'cpu'

predictor = DefaultPredictor(cfg)

# 读取图片
im = cv2.imread('input.jpg')
outputs = predictor(im)

# 获取结果
boxes = outputs["instances"].pred_boxes.tensor.cpu().numpy()
classes = outputs["instances"].pred_classes.cpu().numpy()
scores = outputs["instances"].scores.cpu().numpy()

# 绘制
for box, cls, score in zip(boxes, classes, scores):
    x1, y1, x2, y2 = box
    cv2.rectangle(im, (int(x1), int(y1)), (int(x2), int(y2)), (0, 255, 0), 2)
    cv2.putText(im, f"Class {cls}: {score:.2f}", (int(x1), int(y1)-5), 
                cv2.FONT_HERSHEY_SIMPLEX, 0.5, (0, 255, 0), 2)

cv2.imshow('Detection', im)
cv2.waitKey(0)
```

**特点**：OpenMMLab出品，模块化设计

```python
from mmdet.apis import init_detector, inference_detector
import mmcv

# 初始化模型
config_file = 'configs/yolo/yolov3_mobilenetv2_8x24_300e_coco.py'
checkpoint_file = 'checkpoints/yolov3_mobilenetv2_8x24_300e_coco_20210719_170329-6794b32c.pth'
model = init_detector(config_file, checkpoint_file, device='cuda:0')

# 推理
result = inference_detector(model, 'demo.jpg')

# 可视化
model.show_result('demo.jpg', result, out_file='result.jpg')

# 批量处理
for img in mmcv.scandir('images/'):
    result = inference_detector(model, img)
    model.show_result(img, result, out_file=f'outputs/{img}')
```

## 三、YOLO训练自定义数据

```python
from ultralytics import YOLO
import yaml

# 准备数据集(yolo格式)
# data/
#   images/
#     train/
#     val/
#   labels/
#     train/
#     val/

# 创建数据集配置
dataset_config = {
    'path': './data',
    'train': 'images/train',
    'val': 'images/val',
    'names': {
        0: 'person',
        1: 'car',
        2: 'dog',
        3: 'cat'
    }
}

with open('dataset.yaml', 'w') as f:
    yaml.dump(dataset_config, f)

# 训练
model = YOLO('yolov8n.yaml')
results = model.train(
    data='dataset.yaml',
    epochs=100,
    imgsz=640,
    batch=16,
    device=0,
    workers=8,
    patience=50,  # 早停
    save=True,
    plots=True
)

# 验证
metrics = model.val()
print(metrics)

# 导出
model.export(format='onnx')  # 导出为ONNX格式
```
