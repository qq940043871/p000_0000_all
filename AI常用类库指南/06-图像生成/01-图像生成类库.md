## 一、核心概念

### 图像生成原理

```
文本/图像 → 噪声 → UNet去噪 → 潜空间 → VAE解码 → 图像

扩散模型：逐步从噪声中恢复图像
GAN：生成器与判别器对抗训练
VAE：变分自编码器，编码到潜空间再解码
CLIP：文本编码器，建立文本-图像对应关系
```

## 二、常用类库

### 1. Stable Diffusion (Diffusers)

**特点**：开源最强、支持LoRA/ControlNet、显存要求低

```bash
pip install diffusers transformers accelerate safetensors
```

```python
from diffusers import StableDiffusionPipeline
import torch

# 加载模型
pipe = StableDiffusionPipeline.from_pretrained(
    "runwayml/stable-diffusion-v1-5",
    torch_dtype=torch.float16,
    safety_checker=None
)
pipe = pipe.to("cuda" if torch.cuda.is_available() else "cpu")

# 文生图
prompt = "a beautiful sunset over the ocean, highly detailed, 8k"
image = pipe(prompt, num_inference_steps=50, guidance_scale=7.5).images[0]

# 保存
image.save("output.png")

# 图生图
from diffusers import StableDiffusionImg2ImgPipeline

pipe_img2img = StableDiffusionImg2ImgPipeline.from_pretrained(
    "runwayml/stable-diffusion-v1-5"
)
pipe_img2img = pipe_img2img.to("cuda")

init_image = Image.open("input.jpg").convert("RGB")
init_image = init_image.resize((512, 512))

image = pipe_img2img(
    prompt="变成动漫风格",
    image=init_image,
    strength=0.75,  # 改变程度 0-1
    guidance_scale=7.5
).images[0]
image.save("output_img2img.png")
```

### 2. ControlNet

**特点**：精确控制生成、姿态/边缘/深度等多种控制

```bash
pip install controlnet-aux
```

```python
from diffusers import StableDiffusionControlNetPipeline, ControlNetModel
from controlnet_aux import OpenposeDetector
from PIL import Image
import torch

# 加载ControlNet
controlnet = ControlNetModel.from_pretrained(
    "lllyasviel/sd-controlnet-openpose",
    torch_dtype=torch.float16
)

# 加载主模型
pipe = StableDiffusionControlNetPipeline.from_pretrained(
    "runwayml/stable-diffusion-v1-5",
    controlnet=controlnet,
    torch_dtype=torch.float16
)
pipe = pipe.to("cuda")

# 检测姿态
detector = OpenposeDetector.from_pretrained("lllyasviel/control_v11p_sd15_openpose")
pose_image = detector(Image.open("person.jpg"))

# 生成
prompt = "a woman dancing in the park, highly detailed"
image = pipe(
    prompt,
    image=pose_image,
    num_inference_steps=20,
    controlnet_conditioning_scale=1.0
).images[0]
image.save("output_pose.png")

# 其他ControlNet类型
# canny: 边缘检测 -> lllyasviel/sd-controlnet-canny
# depth: 深度图 -> lllyasviel/sd-controlnet-depth
# hed: 软边缘 -> lllyasviel/sd-controlnet-hed
```

### 3. CLIP + 图像生成

```python
from transformers import CLIPProcessor, CLIPModel
from PIL import Image
import torch

# 加载CLIP
model = CLIPModel.from_pretrained("openai/clip-vit-base-patch32")
processor = CLIPProcessor.from_pretrained("openai/clip-vit-base-patch32")

# 计算图像-文本相似度
image = Image.open("dog.jpg")
texts = ["a dog", "a cat", "a car"]

inputs = processor(text=texts, images=image, return_tensors="pt", padding=True)
outputs = model(**inputs)

logits_per_image = outputs.logits_per_image
probs = logits_per_image.softmax(dim=1)

print("图像-文本匹配概率:")
for text, prob in zip(texts, probs[0]):
    print(f"  {text}: {prob.item():.2%}")

# 零样本图像分类
def zero_shot_classify(image, labels):
    """给定标签列表，对图像进行分类"""
    inputs = processor(text=labels, images=image, return_tensors="pt", padding=True)
    outputs = model(**inputs)
    probs = outputs.logits_per_image.softmax(dim=1)
    return {label: prob.item() for label, prob in zip(labels, probs[0])}

# 使用
labels = ["a photo of a cat", "a photo of a dog", "a photo of a bird"]
result = zero_shot_classify(Image.open("animal.jpg"), labels)
print(result)
```

## 三、LoRA微调

```python
from diffusers import StableDiffusionPipeline
from peft import LoraConfig, get_peft_model
import torch

# 加载基础模型
pipe = StableDiffusionPipeline.from_pretrained(
    "runwayml/stable-diffusion-v1-5",
    torch_dtype=torch.float16
)

# 配置LoRA
lora_config = LoraConfig(
    r=4,  # LoRA秩
    lora_alpha=4,
    target_modules=["to_k", "to_q", "to_v", "to_out.0"],
    lora_dropout=0.05,
    bias="none"
)

# 应用LoRA
pipe.unet = get_peft_model(pipe.unet, lora_config)
pipe.text_encoder = get_peft_model(pipe.text_encoder, lora_config)

# 训练配置
pipe.compile()
pipe.train()

# 开始训练...
# 训练完成后保存
pipe.save_pretrained("lora_model")
```
