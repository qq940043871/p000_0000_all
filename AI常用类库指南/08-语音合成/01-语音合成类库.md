## 一、核心概念

### 语音合成原理

```
文本 → 文本分析 → 韵律预测 → 声学模型 → 波形生成 → 音频

前端处理：分词、多音字、韵律标记
声学模型：预测声学特征(Mel频谱)
声码器：梅尔频谱 → 波形(Griffin-Lim / Neural Vocoder)
```

## 二、常用类库

### 1. Edge-TTS

**特点**：微软技术、免费、中文好、无需GPU

```bash
pip install edge-tts
```

```python
import asyncio
import edge_tts

async def text_to_speech():
    """文本转语音"""
    # 合成
    communicate = edge_tts.Communicate(
        text="你好，这是一段测试语音。",
        voice="zh-CN-XiaoxiaoNeural"  # 中文女声
    )
    await communicate.save("output.mp3")
    
    # 男声
    communicate = edge_tts.Communicate(
        text="你好，这是男声。",
        voice="zh-CN-YunxiNeural"
    )
    await communicate.save("male.mp3")
    
    # 不同风格
    # 新闻: zh-CN-XiaoxiaoNeural
    # 客服: zh-CN-XiaoyiNeural
    # 活泼: zh-CN-liaoning-XiaobaiNeural
    # 成熟: zh-CN-shaanxi-XiaosiNeural

async def long_text():
    """长文本合成(带进度)"""
    text = """
    这是一段较长的文本。语音合成技术正在改变人机交互的方式。
    通过深度学习模型，我们可以生成自然流畅的语音。
    这项技术广泛应用于智能助手、导航系统、有声读物等领域。
    """
    
    communicate = edge_tts.Communicate(text, voice="zh-CN-XiaoxiaoNeural")
    await communicate.save("long_text.mp3")

async def get_voices():
    """获取所有可用音色"""
    voices = await edge_tts.list_voices()
    for voice in voices:
        if voice["Locale"].startswith("zh"):
            print(f"{voice['Name']} - {voice['Gender']} - {voice['Locale']}")

# 运行
asyncio.run(text_to_speech())
```

### 3. pyttsx3 (离线)

**特点**：完全离线、不需要网络、系统自带

```bash
pip install pyttsx3
```

```python
import pyttsx3

# 初始化
engine = pyttsx3.init()

# 获取语音列表
voices = engine.getProperty('voices')
for voice in voices:
    print(f"ID: {voice.id}, Name: {voice.name}, Languages: {voice.languages}")

# 设置语音
engine.setProperty('voice', voices[0].id)  # 选择第一个语音

# 设置参数
engine.setProperty('rate', 150)    # 语速(默认200)
engine.setProperty('volume', 0.9)  # 音量(0-1)

# 合成
engine.say("你好，这是一段离线语音合成测试。")
engine.say("This is an offline TTS test.")
engine.runAndWait()

# 保存为文件
engine.save_to_file("output.wav", "hello")
engine.runAndWait()

# Windows使用SAPI5，Linux使用espeak
```

## 三、完整语音合成系统

```python
import edge_tts
import asyncio
import os
from pydub import AudioSegment

class VoiceSynthesizer:
    """语音合成系统"""
    
    def __init__(self):
        self.voices = {
            'xiaoxiao': 'zh-CN-XiaoxiaoNeural',  # 女声
            'xiaoyi': 'zh-CN-XiaoyiNeural',      # 女声
            'yunxi': 'zh-CN-YunxiNeural',        # 男声
            'en_female': 'en-US-JennyNeural',
            'en_male': 'en-US-GuyNeural'
        }
    
    async def synthesize(self, text, output_path, voice='xiaoxiao', 
                         pitch='+0%', rate='+0%'):
        """异步合成"""
        voice_id = self.voices.get(voice, self.voices['xiaoxiao'])
        
        communicate = edge_tts.Communicate(
            text=text,
            voice=voice_id,
            pitch=pitch,
            rate=rate
        )
        await communicate.save(output_path)
    
    def synthesize_sync(self, text, output_path, voice='xiaoxiao'):
        """同步合成"""
        asyncio.run(self.synthesize(text, output_path, voice))
    
    def merge_audio(self, audio_files, output_path, fade_duration=500):
        """合并多个音频"""
        combined = AudioSegment.empty()
        
        for i, file in enumerate(audio_files):
            audio = AudioSegment.from_mp3(file)
            
            # 添加淡入淡出
            if i > 0:
                audio = audio.fade_in(fade_duration)
            if i < len(audio_files) - 1:
                audio = audio.fade_out(fade_duration)
            
            combined += audio
        
        combined.export(output_path, format='mp3')
    
    def adjust_speed(self, audio_path, output_path, speed_factor=1.0):
        """调整语速"""
        audio = AudioSegment.from_mp3(audio_path)
        new_sample_rate = int(audio.frame_rate * speed_factor)
        
        faster = audio._spawn(audio.raw_data, overrides={
            'frame_rate': new_sample_rate
        })
        faster = faster.set_frame_rate(44100)
        faster.export(output_path, format='mp3')

# 使用示例
async def main():
    synthesizer = VoiceSynthesizer()
    
    # 合成多段文本
    files = []
    texts = [
        "欢迎收听今天的新闻。",
        "今天天气晴朗，适合外出活动。",
        "预计明天会有降雨，请注意带伞。"
    ]
    
    for i, text in enumerate(texts):
        path = f"/tmp/part_{i}.mp3"
        await synthesizer.synthesize(text, path)
        files.append(path)
    
    # 合并
    synthesizer.merge_audio(files, "news.mp3")
    
    # 调整语速
    synthesizer.adjust_speed("news.mp3", "news_fast.mp3", 1.2)

asyncio.run(main())
```
