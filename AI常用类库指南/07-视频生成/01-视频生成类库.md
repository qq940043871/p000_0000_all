## 一、核心概念

### 视频生成原理

```
文本/图像 → 时序建模 → 关键帧生成 → 插帧 → 视频

关键帧模型：首尾帧 + 中间帧预测
时序扩散：视频在时间维度上的扩散
光流预测：运动估计和补偿
```

```python
import cv2
import numpy as np
from PIL import Image

# ============ 视频读取 ============
cap = cv2.VideoCapture('input.mp4')

# 获取视频属性
fps = cap.get(cv2.CAP_PROP_FPS)
width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))

print(f"FPS: {fps}, 分辨率: {width}x{height}, 总帧数: {frame_count}")

# 逐帧读取
frames = []
while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break
    frames.append(frame)
cap.release()

print(f"读取了 {len(frames)} 帧")

# ============ 视频写入 ============
fourcc = cv2.VideoWriter_fourcc(*'mp4v')  # 或 'XVID', 'MJPG'
out = cv2.VideoWriter('output.mp4', fourcc, 30.0, (640, 480))

for i in range(100):
    frame = np.zeros((480, 640, 3), dtype=np.uint8)
    frame[:] = (0, 255, 0)  # 绿色背景
    cv2.putText(frame, f"Frame {i}", (50, 240), 
                 cv2.FONT_HERSHEY_SIMPLEX, 2, (255, 255, 255), 3)
    out.write(frame)

out.release()

# ============ 视频处理 ============
# 视频缩放
cap = cv2.VideoCapture('input.mp4')
out = cv2.VideoWriter('resized.mp4', fourcc, 30.0, (320, 240))

while True:
    ret, frame = cap.read()
    if not ret:
        break
    resized = cv2.resize(frame, (320, 240))
    out.write(resized)

cap.release()
out.release()

# ============ 抽帧 ============
cap = cv2.VideoCapture('input.mp4')
frame_idx = 0
saved_idx = 0

while True:
    ret, frame = cap.read()
    if not ret:
        break
    
    # 每30帧保存一张图
    if frame_idx % 30 == 0:
        cv2.imwrite(f'frame_{saved_idx:04d}.jpg', frame)
        saved_idx += 1
    
    frame_idx += 1

cap.release()
```

video = VideoFileClip("video.mp4")

### 3. Stable Video Diffusion (SVD)

**特点**：图片转视频、扩散模型驱动

```bash
pip install torch torchvision
pip install diffusers transformers
```

```python
from diffusers import StableVideoDiffusionPipeline
from diffusers.utils import load_image
import torch
from PIL import Image
import cv2

# 加载模型
pipe = StableVideoDiffusionPipeline.from_pretrained(
    "stabilityai/stable-video-diffusion-img2vid-xt",
    torch_dtype=torch.float16,
    variant="fp16"
)
pipe = pipe.to("cuda")

# 加载并预处理图像
image = load_image("input_image.jpg")
image = image.resize((1024, 576))  # 16:9比例

# 生成视频(25帧)
generator = torch.Generator(device="cuda").manual_seed(42)
frames = pipe(
    image,
    num_frames=25,
    decode_chunk_size=8,
    generator=generator,
    num_inference_steps=25,
    guidance_scale=3.0
).frames[0]

# 保存视频
output_path = "output_video.mp4"
height, width = frames[0].shape[:2]

fourcc = cv2.VideoWriter_fourcc(*'mp4v')
video = cv2.VideoWriter(output_path, fourcc, fps=8, frameSize=(width, height))

for frame in frames:
    video.write(cv2.cvtColor(np.array(frame), cv2.COLOR_RGB2BGR))

video.release()
```

## 三、视频处理Pipeline

```python
import cv2
import numpy as np
from PIL import Image

class VideoProcessor:
    def __init__(self, input_path):
        self.cap = cv2.VideoCapture(input_path)
        self.fps = self.cap.get(cv2.CAP_PROP_FPS)
        self.width = int(self.cap.get(cv2.CAP_PROP_FRAME_WIDTH))
        self.height = int(self.cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
    
    def read_frame(self):
        ret, frame = self.cap.read()
        return ret, frame
    
    def read_all_frames(self):
        frames = []
        while True:
            ret, frame = self.read_frame()
            if not ret:
                break
            frames.append(frame)
        return frames
    
    def apply_filter(self, frame, filter_type='grayscale'):
        if filter_type == 'grayscale':
            return cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        elif filter_type == 'blur':
            return cv2.GaussianBlur(frame, (15, 15), 0)
        elif filter_type == 'edge':
            gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
            return cv2.Canny(gray, 50, 150)
        return frame
    
    def process_video(self, output_path, filter_type='original'):
        fourcc = cv2.VideoWriter_fourcc(*'mp4v')
        out = cv2.VideoWriter(output_path, fourcc, self.fps, 
                              (self.width, self.height))
        
        while True:
            ret, frame = self.read_frame()
            if not ret:
                break
            
            processed = self.apply_filter(frame, filter_type)
            if len(processed.shape) == 2:
                processed = cv2.cvtColor(processed, cv2.COLOR_GRAY2BGR)
            
            out.write(processed)
        
        self.cap.release()
        out.release()
    
    def __del__(self):
        if hasattr(self, 'cap'):
            self.cap.release()

# 使用
processor = VideoProcessor('input.mp4')
processor.process_video('output_blur.mp4', filter_type='blur')
```

## 四、格式转换

```python
import subprocess

def convert_video(input_path, output_path, codec='libx264', fps=None):
    """使用ffmpeg转换视频格式"""
    cmd = ['ffmpeg', '-i', input_path]
    
    if fps:
        cmd.extend(['-r', str(fps)])
    
    cmd.extend(['-c:v', codec, '-c:a', 'aac', output_path])
    
    subprocess.run(cmd, check=True)

# 转换
convert_video('input.avi', 'output.mp4')
convert_video('input.mp4', 'output.gif', fps=10)
```
