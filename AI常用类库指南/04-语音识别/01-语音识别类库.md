## 一、核心概念

### 语音识别原理

```
音频 → 预处理(采样/归一化) → 特征提取(MFCC/FBank) → 声学模型 → 解码器 → 文本

MFCC: Mel频率倒谱系数，模拟人耳听觉特性
声学模型: CTC/Attention将声学特征转为音素
语言模型: 结合语言知识提高准确率
```

## 二、常用类库

### 1. Whisper (OpenAI)

**特点**：多语言、支持中文、效果好、开源

```bash
pip install openai-whisper
```

```python
import whisper

# 加载模型(base/small/medium/large)
model = whisper.load_model("base")

# 音频文件识别
result = model.transcribe("audio.mp3")
print(result["text"])

# 识别并翻译
result = model.transcribe("chinese_audio.mp3", language="zh", task="translate")
print(result["text"])  # 英文输出

# 实时录音识别
import pyaudio
import numpy as np

model = whisper.load_model("small")
p = pyaudio.PyAudio()
stream = p.open(format=pyaudio.paInt16, channels=1, rate=16000, input=True)

print("开始录音，按Ctrl+C停止...")
frames = []

while True:
    try:
        data = stream.read(16000, exception_on_overflow=False)
        frames.append(np.frombuffer(data, np.int16))
    except KeyboardInterrupt:
        break

audio = np.concatenate(frames)
result = model.transcribe(audio, language="zh")
print(f"识别结果: {result['text']}")

stream.stop_stream()
stream.close()
p.terminate()
```

### 2. SpeechRecognition

**特点**：支持多种引擎（Google/百度/讯飞）

```bash
pip install SpeechRecognition
pip install pyaudio  # 录音
```

```python
import speech_recognition as sr

# 创建识别器
r = sr.Recognizer()

# 使用麦克风录音
with sr.Microphone() as source:
    print("请说话...")
    audio = r.listen(source)

# Google在线识别
try:
    text = r.recognize_google(audio, language='zh-CN')
    print(f"Google识别: {text}")
except sr.UnknownValueError:
    print("无法识别音频")
except sr.RequestError:
    print("服务不可用")

# 识别音频文件
with sr.AudioFile('audio.wav') as source:
    audio = r.record(source)
    text = r.recognize_google(audio, language='zh-CN')
    print(f"识别结果: {text}")
```

### 3. Vosk

**特点**：离线识别、模型小巧、实时性好

```bash
pip install vosk
# 下载模型: https://alphacephei.com/vosk/models
```

```python
from vosk import Model, KaldiRecognizer
import pyaudio
import json

# 加载模型
model = Model('vosk-model-small-cn-0.3')
rec = KaldiRecognizer(model, 16000)

# 录音识别
p = pyaudio.PyAudio()
stream = p.open(format=pyaudio.paInt16, channels=1, rate=16000, input=True)

print("开始识别...")
full_text = []

while True:
    data = stream.read(16000, exception_on_overflow=False)
    if len(data) == 0:
        break
    
    if rec.AcceptWaveform(data):
        result = json.loads(rec.Result())
        if result['text']:
            full_text.append(result['text'])
            print(f"部分识别: {result['text']}")

# 最终结果
final = json.loads(rec.FinalResult())
full_text.append(final['text'])
print(f"\n最终结果: {''.join(full_text)}")

stream.stop_stream()
stream.close()
p.terminate()
```

```python
for audio in reader.read("stream://microphone"):
    result = model.generate(audio)
    print(result)
```

## 三、完整录音+识别示例

```python
import whisper
import soundfile as sf
import sounddevice as sd
import numpy as np

# 录音参数
SAMPLE_RATE = 16000
DURATION = 5  # 秒

print(f"准备录音 {DURATION} 秒...")

# 录音
audio = sd.rec(int(DURATION * SAMPLE_RATE), samplerate=SAMPLE_RATE, channels=1)
sd.wait()

# 保存为临时文件
audio_path = "/tmp/recording.wav"
sf.write(audio_path, audio, SAMPLE_RATE)

# 加载Whisper模型并识别
model = whisper.load_model("base")
result = model.transcribe(audio_path, language="zh")

print(f"\n识别结果: {result['text']}")

# 显示分段识别
print("\n详细结果:")
for segment in result['segments']:
    print(f"[{segment['start']:.1f}s - {segment['end']:.1f}s] {segment['text']}")
```
