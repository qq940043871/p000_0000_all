# 第二章：x86汇编语言

## 2.1 汇编语言基础

### 什么是汇编语言

汇编语言是最接近机器语言的高级语言，每条汇编指令对应一条机器指令。

**优点：**
- 执行效率高
- 可以直接操作硬件
- 可以访问所有CPU功能

**缺点：**
- 代码复杂，难以维护
- 与处理器架构紧密相关
- 开发效率低

### 汇编语言的语法

#### 指令格式

```asm
标签: 指令 操作数1, 操作数2  ; 注释
```

**例子：**
```asm
start:
    mov eax, 0x1000      ; 将0x1000移动到EAX
    add eax, ebx         ; EAX加上EBX
    jmp start            ; 跳转到start标签
```

#### 操作数类型

1. **立即数**：直接的数值
   ```asm
   mov eax, 0x1000      ; 0x1000是立即数
   ```

2. **寄存器**：CPU寄存器
   ```asm
   mov eax, ebx         ; ebx是寄存器
   ```

3. **内存**：内存地址
   ```asm
   mov eax, [0x1000]    ; 从地址0x1000读取数据
   mov [eax], ebx       ; 将ebx写入eax指向的地址
   ```

## 2.2 常用指令

### 数据传输指令

```asm
mov dest, src           ; 传输数据
movzx dest, src         ; 零扩展传输
movsx dest, src         ; 符号扩展传输
xchg dest, src          ; 交换数据
lea dest, [src]         ; 加载有效地址
```

**例子：**
```asm
mov eax, 0x1000         ; EAX = 0x1000
mov ebx, eax            ; EBX = EAX
movzx eax, al           ; EAX = AL（零扩展）
lea eax, [ebx + 4]      ; EAX = EBX + 4的地址
```

### 算术指令

```asm
add dest, src           ; 加法
sub dest, src           ; 减法
imul dest, src          ; 有符号乘法
idiv src                ; 有符号除法
inc dest                ; 加1
dec dest                ; 减1
neg dest                ; 取反
```

**例子：**
```asm
add eax, ebx            ; EAX = EAX + EBX
sub eax, 1              ; EAX = EAX - 1
imul eax, 2             ; EAX = EAX * 2
idiv ecx                ; EDX:EAX = EDX:EAX / ECX
```

### 逻辑指令

```asm
and dest, src           ; 按位与
or dest, src            ; 按位或
xor dest, src           ; 按位异或
not dest                ; 按位非
shl dest, count         ; 左移
shr dest, count         ; 右移
sar dest, count         ; 算术右移
```

**例子：**
```asm
and eax, 0xFF           ; EAX = EAX & 0xFF
or eax, 0x80            ; EAX = EAX | 0x80
xor eax, eax            ; EAX = 0（清零）
shl eax, 2              ; EAX = EAX << 2
```

### 比较与跳转指令

```asm
cmp dest, src           ; 比较（执行减法但不保存结果）
test dest, src          ; 测试（执行与操作但不保存结果）
jmp label               ; 无条件跳转
je label                ; 相等时跳转（ZF=1）
jne label               ; 不相等时跳转（ZF=0）
jl label                ; 小于时跳转（SF!=OF）
jle label               ; 小于等于时跳转
jg label                ; 大于时跳转（SF==OF && ZF=0）
jge label               ; 大于等于时跳转
```

**例子：**
```asm
cmp eax, ebx
je equal_label          ; 如果EAX==EBX，跳转到equal_label
jne not_equal_label     ; 如果EAX!=EBX，跳转到not_equal_label
```

### 栈操作指令

```asm
push src                ; 压栈
pop dest                ; 出栈
call label              ; 调用函数（压入返回地址）
ret                     ; 返回（弹出返回地址）
```

**例子：**
```asm
push eax                ; 将EAX压入栈
push ebx                ; 将EBX压入栈
call function           ; 调用函数
pop ebx                 ; 从栈弹出到EBX
pop eax                 ; 从栈弹出到EAX
```

## 2.3 寻址方式

### 直接寻址

```asm
mov eax, 0x1000         ; 立即数寻址
mov eax, [0x1000]       ; 直接寻址
```

### 寄存器寻址

```asm
mov eax, ebx            ; 寄存器寻址
mov eax, [ebx]          ; 寄存器间接寻址
```

### 变址寻址

```asm
mov eax, [ebx + 4]      ; 基址 + 偏移
mov eax, [ebx + ecx]    ; 基址 + 变址
mov eax, [ebx + ecx*2]  ; 基址 + 变址*比例
mov eax, [ebx + ecx*4 + 8]  ; 基址 + 变址*比例 + 偏移
```

### 相对寻址

```asm
jmp $+5                 ; 相对当前位置跳转
call $+10               ; 相对当前位置调用
```

## 2.4 函数调用约定

### cdecl调用约定（C语言默认）

**参数传递：**
- 参数从右到左压入栈
- 返回值在EAX中

**栈帧结构：**
```
ESP+8   参数2
ESP+4   参数1
ESP+0   返回地址
```

**例子：**
```asm
; 调用函数
push 2              ; 参数2
push 1              ; 参数1
call my_function    ; 调用函数
add esp, 8          ; 清理栈

; 函数定义
my_function:
    push ebp
    mov ebp, esp
    
    mov eax, [ebp+8]    ; 参数1
    mov ebx, [ebp+12]   ; 参数2
    add eax, ebx        ; 计算
    
    pop ebp
    ret
```

### stdcall调用约定

**参数传递：**
- 参数从右到左压入栈
- 返回值在EAX中
- 被调用函数负责清理栈

**例子：**
```asm
my_function:
    push ebp
    mov ebp, esp
    
    mov eax, [ebp+8]    ; 参数1
    mov ebx, [ebp+12]   ; 参数2
    add eax, ebx
    
    pop ebp
    ret 8               ; 返回并清理8字节栈
```

## 2.5 宏与伪指令

### 宏定义

```asm
%macro print_char 1
    mov al, %1
    mov ah, 0x0E
    int 0x10
%endmacro

; 使用宏
print_char 'A'
print_char 'B'
```

### 常用伪指令

```asm
db      ; 定义字节
dw      ; 定义字（2字节）
dd      ; 定义双字（4字节）
dq      ; 定义四字（8字节）
times   ; 重复定义
equ     ; 定义常量
```

**例子：**
```asm
msg db "Hello, World!", 0
buffer times 256 db 0
BUFFER_SIZE equ 256
```

## 2.6 汇编程序示例

### 例1：简单的循环

```asm
[bits 16]
org 0x7C00

start:
    mov cx, 10          ; 循环10次
    xor ax, ax          ; AX = 0

loop_start:
    add ax, 1           ; AX加1
    loop loop_start     ; 循环（CX减1，如果CX!=0则跳转）
    
    ; 此时AX = 10
    jmp $               ; 无限循环
```

### 例2：字符串复制

```asm
[bits 16]
org 0x7C00

start:
    mov si, src_str     ; 源字符串地址
    mov di, dst_str     ; 目的字符串地址
    
copy_loop:
    lodsb               ; 从[SI]加载到AL，SI加1
    stosb               ; 将AL存储到[DI]，DI加1
    cmp al, 0           ; 检查是否为字符串结束符
    jne copy_loop       ; 如果不是，继续循环
    
    jmp $

src_str db "Hello", 0
dst_str times 10 db 0
```

### 例3：计算阶乘

```asm
[bits 16]
org 0x7C00

start:
    mov eax, 5          ; 计算5的阶乘
    call factorial
    
    ; 结果在EAX中
    jmp $

factorial:
    cmp eax, 1
    je fact_end
    
    push eax
    dec eax
    call factorial
    pop ebx
    imul eax, ebx
    
fact_end:
    ret
```
