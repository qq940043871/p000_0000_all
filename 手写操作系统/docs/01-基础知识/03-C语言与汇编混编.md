# 第三章：C语言与汇编混编

## 3.1 混编基础

### 为什么需要混编

在操作系统开发中，某些操作必须用汇编语言实现：

1. **硬件操作**：直接操作寄存器、端口
2. **特权指令**：只能在内核模式执行
3. **性能关键代码**：需要最高效率
4. **上下文切换**：进程切换需要保存/恢复寄存器

### 混编的方式

#### 方式1：内联汇编（Inline Assembly）

在C代码中直接嵌入汇编代码：

```c
#include <stdio.h>

int main() {
    int a = 5, b = 3;
    int result;
    
    // GCC内联汇编
    asm("addl %1, %0" : "=r" (result) : "r" (a), "0" (b));
    
    printf("Result: %d\n", result);
    return 0;
}
```

#### 方式2：分离编译

汇编代码和C代码分别编译，然后链接：

```
C代码 → 编译 → 目标文件
汇编代码 → 汇编 → 目标文件
         ↓
      链接 → 可执行文件
```

## 3.2 GCC内联汇编

### 基本语法

```c
asm("汇编指令" : 输出操作数 : 输入操作数 : 破坏的寄存器);
```

### 操作数约束

| 约束 | 说明 |
|------|------|
| **r** | 任意通用寄存器 |
| **a** | EAX寄存器 |
| **b** | EBX寄存器 |
| **c** | ECX寄存器 |
| **d** | EDX寄存器 |
| **S** | ESI寄存器 |
| **D** | EDI寄存器 |
| **m** | 内存 |
| **i** | 立即数 |
| **=** | 输出操作数 |
| **+** | 输入输出操作数 |
| **&** | 早期破坏 |

### 例子

#### 例1：简单的加法

```c
int add(int a, int b) {
    int result;
    asm("addl %1, %0" : "=r" (result) : "r" (a), "0" (b));
    return result;
}
```

**说明：**
- `"addl %1, %0"`：汇编指令
- `"=r" (result)`：输出操作数，存储在寄存器中
- `"r" (a)`：输入操作数，存储在寄存器中
- `"0" (b)`：输入操作数，与第0个输出操作数使用同一个寄存器

#### 例2：读取寄存器

```c
unsigned int read_cr0() {
    unsigned int cr0;
    asm("movl %%cr0, %0" : "=r" (cr0));
    return cr0;
}
```

**说明：**
- `%%cr0`：两个%表示寄存器名称
- `"=r" (cr0)`：输出到寄存器

#### 例3：写入寄存器

```c
void write_cr0(unsigned int cr0) {
    asm("movl %0, %%cr0" : : "r" (cr0));
}
```

#### 例4：I/O端口操作

```c
// 读端口
unsigned char inb(unsigned short port) {
    unsigned char value;
    asm("inb %1, %0" : "=a" (value) : "d" (port));
    return value;
}

// 写端口
void outb(unsigned short port, unsigned char value) {
    asm("outb %0, %1" : : "a" (value), "d" (port));
}
```

## 3.3 分离编译

### 汇编文件（.asm）

```asm
; kernel.asm
[bits 32]

global kernel_main
global read_eax

kernel_main:
    push ebp
    mov ebp, esp
    
    ; 调用C函数
    call main
    
    pop ebp
    ret

read_eax:
    ; 返回EAX的值
    ret
```

### C文件（.c）

```c
// kernel.c
#include <stdio.h>

extern void kernel_main();
extern unsigned int read_eax();

void main() {
    printf("Hello from C!\n");
    unsigned int eax = read_eax();
    printf("EAX = 0x%x\n", eax);
}
```

### Makefile

```makefile
CC = gcc
AS = nasm
LD = ld

CFLAGS = -m32 -fno-builtin -nostdlib
ASFLAGS = -f elf32

OBJS = kernel.o kernel_main.o

kernel.elf: $(OBJS)
	$(LD) -m elf_i386 -o $@ $^

kernel.o: kernel.c
	$(CC) $(CFLAGS) -c $< -o $@

kernel_main.o: kernel.asm
	$(AS) $(ASFLAGS) $< -o $@

clean:
	rm -f *.o *.elf
```

## 3.4 常用的汇编函数

### 1. 读写控制寄存器

```c
// 读CR0
unsigned int read_cr0() {
    unsigned int cr0;
    asm("movl %%cr0, %0" : "=r" (cr0));
    return cr0;
}

// 写CR0
void write_cr0(unsigned int cr0) {
    asm("movl %0, %%cr0" : : "r" (cr0));
}

// 读CR3（页目录基址）
unsigned int read_cr3() {
    unsigned int cr3;
    asm("movl %%cr3, %0" : "=r" (cr3));
    return cr3;
}

// 写CR3
void write_cr3(unsigned int cr3) {
    asm("movl %0, %%cr3" : : "r" (cr3));
}
```

### 2. 中断操作

```c
// 禁用中断
void disable_interrupts() {
    asm("cli");
}

// 启用中断
void enable_interrupts() {
    asm("sti");
}

// 加载中断描述符表
void load_idt(unsigned int idt_ptr) {
    asm("lidt (%0)" : : "r" (idt_ptr));
}

// 加载全局描述符表
void load_gdt(unsigned int gdt_ptr) {
    asm("lgdt (%0)" : : "r" (gdt_ptr));
}
```

### 3. I/O端口操作

```c
// 读字节
unsigned char inb(unsigned short port) {
    unsigned char value;
    asm("inb %1, %0" : "=a" (value) : "d" (port));
    return value;
}

// 写字节
void outb(unsigned short port, unsigned char value) {
    asm("outb %0, %1" : : "a" (value), "d" (port));
}

// 读字
unsigned short inw(unsigned short port) {
    unsigned short value;
    asm("inw %1, %0" : "=a" (value) : "d" (port));
    return value;
}

// 写字
void outw(unsigned short port, unsigned short value) {
    asm("outw %0, %1" : : "a" (value), "d" (port));
}

// 读双字
unsigned int inl(unsigned short port) {
    unsigned int value;
    asm("inl %1, %0" : "=a" (value) : "d" (port));
    return value;
}

// 写双字
void outl(unsigned short port, unsigned int value) {
    asm("outl %0, %1" : : "a" (value), "d" (port));
}
```

### 4. 内存操作

```c
// 内存复制
void memcpy(void *dest, void *src, unsigned int size) {
    asm("cld; rep movsb" : : "D" (dest), "S" (src), "c" (size));
}

// 内存填充
void memset(void *dest, unsigned char value, unsigned int size) {
    asm("cld; rep stosb" : : "D" (dest), "a" (value), "c" (size));
}
```

### 5. 栈操作

```c
// 获取栈指针
unsigned int get_esp() {
    unsigned int esp;
    asm("movl %%esp, %0" : "=r" (esp));
    return esp;
}

// 获取基指针
unsigned int get_ebp() {
    unsigned int ebp;
    asm("movl %%ebp, %0" : "=r" (ebp));
    return ebp;
}
```

## 3.5 完整示例

### 例：内核初始化

**kernel.asm：**
```asm
[bits 32]

global kernel_main
global load_gdt
global load_idt

kernel_main:
    ; 设置栈
    mov esp, 0x90000
    
    ; 调用C函数
    call main
    
    ; 无限循环
    jmp $

load_gdt:
    mov eax, [esp+4]
    lgdt [eax]
    ret

load_idt:
    mov eax, [esp+4]
    lidt [eax]
    ret
```

**kernel.c：**
```c
#include "kernel.h"

extern void load_gdt(unsigned int gdt_ptr);
extern void load_idt(unsigned int idt_ptr);

// GDT定义
struct gdt_entry {
    unsigned short limit_low;
    unsigned short base_low;
    unsigned char base_mid;
    unsigned char access;
    unsigned char granularity;
    unsigned char base_high;
} __attribute__((packed));

struct gdt_ptr {
    unsigned short limit;
    unsigned int base;
} __attribute__((packed));

struct gdt_entry gdt[3];
struct gdt_ptr gdt_ptr;

void init_gdt() {
    // 空描述符
    gdt[0].limit_low = 0;
    gdt[0].base_low = 0;
    gdt[0].base_mid = 0;
    gdt[0].access = 0;
    gdt[0].granularity = 0;
    gdt[0].base_high = 0;
    
    // 代码段
    gdt[1].limit_low = 0xFFFF;
    gdt[1].base_low = 0;
    gdt[1].base_mid = 0;
    gdt[1].access = 0x9A;
    gdt[1].granularity = 0xCF;
    gdt[1].base_high = 0;
    
    // 数据段
    gdt[2].limit_low = 0xFFFF;
    gdt[2].base_low = 0;
    gdt[2].base_mid = 0;
    gdt[2].access = 0x92;
    gdt[2].granularity = 0xCF;
    gdt[2].base_high = 0;
    
    gdt_ptr.limit = sizeof(gdt) - 1;
    gdt_ptr.base = (unsigned int)&gdt;
    
    load_gdt((unsigned int)&gdt_ptr);
}

void main() {
    init_gdt();
    
    // 继续初始化...
}
```
