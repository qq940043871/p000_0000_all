## 4.4 Makefile编写

### 基础Makefile

```makefile
# 编译器和工具
CC = gcc
AS = nasm
LD = ld
OBJCOPY = objcopy

# 编译选项
CFLAGS = -m32 -fno-builtin -nostdlib -Wall -Wextra
ASFLAGS = -f elf32
LDFLAGS = -m elf_i386

# 源文件和目标文件
BOOT_SRC = src/boot/boot.asm
BOOT_OBJ = build/boot.o

KERNEL_SRC = src/kernel/kernel.c src/kernel/kernel.asm
KERNEL_OBJ = build/kernel.o build/kernel_asm.o

LIB_SRC = src/lib/string.c src/lib/stdio.c
LIB_OBJ = build/string.o build/stdio.o

# 最终输出
OS_IMG = build/os.img
OS_ELF = build/os.elf

# 默认目标
all: $(OS_IMG)

# 创建build目录
build:
	mkdir -p build

# 编译引导程序
$(BOOT_OBJ): $(BOOT_SRC) | build
	$(AS) $(ASFLAGS) $< -o $@

# 编译内核C代码
build/kernel.o: src/kernel/kernel.c | build
	$(CC) $(CFLAGS) -c $< -o $@

# 编译内核汇编代码
build/kernel_asm.o: src/kernel/kernel.asm | build
	$(AS) $(ASFLAGS) $< -o $@

# 编译库函数
build/%.o: src/lib/%.c | build
	$(CC) $(CFLAGS) -c $< -o $@

# 链接
$(OS_ELF): $(BOOT_OBJ) $(KERNEL_OBJ) $(LIB_OBJ)
	$(LD) $(LDFLAGS) -o $@ $^

# 生成镜像
$(OS_IMG): $(OS_ELF)
	$(OBJCOPY) -O binary $< $@

# 运行
run: $(OS_IMG)
	qemu-system-i386 -fda $<

# 调试
debug: $(OS_IMG)
	qemu-system-i386 -fda $< -s -S &
	gdb -ex "target remote localhost:1234" -ex "file $(OS_ELF)"

# 清理
clean:
	rm -rf build/

# 伪目标
.PHONY: all run debug clean
```

## 4.5 编译和运行

### 编译

```bash
# 编译整个项目
make

# 只编译特定部分
make build/boot.o
make build/kernel.o

# 清理编译结果
make clean
```

### 运行

```bash
# 运行操作系统
make run

# 调试操作系统
make debug
```

### 调试命令

```bash
# 在GDB中
(gdb) target remote localhost:1234    # 连接到QEMU
(gdb) break main                       # 设置断点
(gdb) continue                         # 继续执行
(gdb) step                             # 单步执行
(gdb) next                             # 执行下一行
(gdb) print eax                        # 打印寄存器
(gdb) x/10x 0x7c00                     # 查看内存
(gdb) disassemble main                 # 反汇编
```

## 4.7 常见问题

### 问题1：找不到32位库

**错误信息：**
```
/usr/bin/ld: cannot find -lc
```

**解决方案：**
```bash
# Ubuntu/Debian
sudo apt-get install gcc-multilib g++-multilib

# 或者使用-nostdlib选项避免链接C库
gcc -m32 -nostdlib ...
```

### 问题2：QEMU找不到

**错误信息：**
```
qemu-system-i386: command not found
```

**解决方案：**
```bash
# 重新安装QEMU
sudo apt-get install qemu-system-x86

# 或者使用完整路径
/usr/bin/qemu-system-i386
```

### 问题3：GDB连接失败

**错误信息：**
```
Remote debugging using localhost:1234
Couldn't establish connection to remote target
```

**解决方案：**
```bash
# 确保QEMU以调试模式启动
qemu-system-i386 -s -S ...

# 在GDB中重新连接
(gdb) target remote localhost:1234
```
