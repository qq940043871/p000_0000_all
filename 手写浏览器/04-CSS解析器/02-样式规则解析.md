## 一、CSS规则结构

```cpp
struct CSSRule {
    std::vector<std::string> selectors;
    std::map<std::string, std::string> declarations;
    int specificity;
};

class CSSParser {
public:
    std::vector<CSSRule> parse(const std::string& css) {
        CSSTokenizer tokenizer;
        std::vector<CSSToken> tokens = tokenizer.tokenize(css);
        
        std::vector<CSSRule> rules;
        size_t pos = 0;
        
        while (pos < tokens.size()) {
            if (tokens[pos].type == CSSTokenType::AT_RULE) {
                pos = parseAtRule(tokens, pos, rules);
            } else if (tokens[pos].type == CSSTokenType::SELECTOR) {
                CSSRule rule = parseRule(tokens, pos, pos);
                if (!rule.selectors.empty()) {
                    rules.push_back(rule);
                }
            } else {
                pos++;
            }
        }
        
        return rules;
    }
    
private:
    CSSRule parseRule(std::vector<CSSToken>& tokens, 
                     size_t start, size_t& end) {
        CSSRule rule;
        
        // 解析选择器列表
        while (tokens[end].type == CSSTokenType::SELECTOR ||
               tokens[end].type == CSSTokenType::COMMA) {
            if (tokens[end].type == CSSTokenType::SELECTOR) {
                rule.selectors.push_back(tokens[end].value);
            }
            end++;
        }
        
        // 跳过左花括号
        if (tokens[end].type == CSSTokenType::LBRACE) end++;
        
        // 解析声明块
        while (tokens[end].type != CSSTokenType::RBRACE &&
               tokens[end].type != CSSTokenType::EOF) {
            if (tokens[end].type == CSSTokenType::SELECTOR) {
                std::string property = tokens[end++].value;
                if (tokens[end].type == CSSTokenType::COLON) end++;
                std::string value = parseValue(tokens, end);
                rule.declarations[property] = value;
            }
            end++;
        }
        
        // 计算特异性
        rule.specificity = calculateSpecificity(rule.selectors);
        
        return rule;
    }
    
    int calculateSpecificity(const std::vector<std::string>& selectors) {
        int specificity = 0;
        
        for (const std::string& selector : selectors) {
            // ID选择器
            specificity += countChar(selector, '#') * 100;
            // 类、属性、伪类
            specificity += countChar(selector, '.') * 10;
            specificity += countMatches(selector, "\\[") * 10;
            specificity += countMatches(selector, ":") * 10;
            // 标签、伪元素
            specificity += countMatches(selector, "^[^.\\[#:]") * 1;
        }
        
        return specificity;
    }
};
```
