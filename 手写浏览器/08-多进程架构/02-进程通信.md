## 二、C++实现

```cpp
class IPCChannel {
public:
    IPCChannel(int peerPid) {
        // 创建管道
        int fds[2];
        pipe(fds);
        
        readFd = fds[0];
        writeFd = fds[1];
        peerPid_ = peerPid;
    }
    
    void send(const IPCMessage& message) {
        // 序列化消息
        std::vector<uint8_t> data = serialize(message);
        
        // 写入管道
        write(readFd, data.data(), data.size());
    }
    
    void receive(IPCMessage& message) {
        uint8_t buffer[4096];
        ssize_t size = read(writeFd, buffer, sizeof(buffer));
        
        if (size > 0) {
            message = deserialize(buffer, size);
        }
    }
    
private:
    int readFd;
    int writeFd;
    int peerPid_;
};

class IPCRouter {
private:
    std::map<int, IPCChannel*> channels;
    
public:
    void send(int routingId, const IPCMessage& message) {
        int pid = getRendererPID(routingId);
        IPCChannel* channel = getChannel(pid);
        channel->send(message);
    }
    
    void handleMessage(const IPCMessage& message) {
        auto handler = handlers_[message.type()];
        if (handler) {
            handler(message);
        }
    }
};
```
