## 二、扩展API

```cpp
class ExtensionAPI {
public:
    void registerExtension(const std::string& id, 
                          const ExtensionManifest& manifest) {
        // 加载扩展资源
        loadResources(id, manifest);
        
        // 注册消息处理
        registerMessageHandlers(id);
        
        // 注入content script
        for (const std::string& url : manifest.content_scripts) {
            injectContentScript(id, url);
        }
    }
    
    void handleMessage(const ExtensionMessage& message) {
        if (message.target == "background") {
            // background脚本处理
        } else if (message.target == "content") {
            // content脚本处理
        }
    }
};
```
