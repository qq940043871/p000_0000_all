  ┌────────────┐    ┌────────────┐    ┌────────────┐

## 二、C++绘制实现

```cpp
class RenderObject {
public:
    virtual void paint(GraphicsContext* gc) = 0;
    
protected:
    Rect bounds;
    CSSComputedStyle style;
};

class BoxRenderObject : public RenderObject {
public:
    void paint(GraphicsContext* gc) override {
        // 1. 绘制背景
        paintBackground(gc);
        
        // 2. 绘制边框
        paintBorder(gc);
        
        // 3. 绘制内容
        paintContent(gc);
        
        // 4. 绘制子元素
        for (auto& child : children) {
            child->paint(gc);
        }
    }
    
private:
    void paintBackground(GraphicsContext* gc) {
        std::string bgColor = style.getProperty("background-color");
        if (!bgColor.empty()) {
            Color color = parseColor(bgColor);
            
            std::string bgImage = style.getProperty("background-image");
            if (!bgImage.empty() && bgImage != "none") {
                // 绘制背景图片
            } else {
                gc->fillRect(bounds, color);
            }
        }
    }
    
    void paintBorder(GraphicsContext* gc) {
        double top = bounds.y;
        double left = bounds.x;
        double width = bounds.width;
        double height = bounds.height;
        
        // 上边框
        if (style.getProperty("border-top-width") != "0") {
            gc->fillRect(left, top, width, 
                parseLength(style.getProperty("border-top-width"), 0),
                parseColor(style.getProperty("border-top-color")));
        }
        // 其他边框...
    }
};
```
