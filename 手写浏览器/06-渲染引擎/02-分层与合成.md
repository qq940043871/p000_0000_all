## 一、合成层

```
需要提升为合成层的条件：
  - position: fixed
  - position: sticky
  - will-change: transform/opacity
  - transform: translateZ(0)
  - 动画/过渡中的transform/opacity
  - video、canvas、iframe
  - CSS滤镜
```

## 二、CompositorLayer

```cpp
class CompositorLayer {
public:
    LayerId id;
    GraphicsLayer* graphicsLayer;
    Rect viewportRect;
    bool dirty;
    
    void update(const std::vector<RenderObject*>& changedObjects) {
        for (RenderObject* obj : changedObjects) {
            if (obj->needsOwnLayer()) {
                // 提升为独立层
                ensureLayer(obj);
            }
        }
        
        // 更新层内容
        for (auto& layer : layers) {
            layer->invalidate(obj->bounds);
        }
        
        dirty = true;
    }
};

class Compositor {
private:
    std::map<LayerId, CompositorLayer*> layers;
    CompositorLayer* rootLayer;
    
public:
    void composite() {
        // 1. 计算需要更新的层
        std::vector<CompositorLayer*> dirtyLayers;
        for (auto& pair : layers) {
            if (pair.second->dirty) {
                dirtyLayers.push_back(pair.second);
            }
        }
        
        // 2. 将内容绘制到纹理
        for (CompositorLayer* layer : dirtyLayers) {
            renderToTexture(layer);
        }
        
        // 3. 合成到屏幕
        for (CompositorLayer* layer : layers) {
            drawLayer(layer);
        }
    }
};
```
