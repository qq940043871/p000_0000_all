## 一、布局基础

```
布局类型：
  1. 块级布局（Block）
     - 垂直排列
     - width: 100%（默认）
  
  2. 行内布局（Inline）
     - 水平排列
     - width由内容决定
  
  3. Flexbox布局
     - 一维布局模型
     - 弹性盒子
  
  4. Grid布局
     - 二维布局模型
     - 网格系统
```

## 二、块级布局

```cpp
class BlockLayout {
public:
    void layout(LayoutBox* box, double containingWidth) {
        // 计算子元素高度
        double childY = 0;
        
        for (LayoutBox* child : box->children) {
            if (child->type == LayoutBox::BoxType::BLOCK) {
                layoutBlock(child, containingWidth);
                childY += child->model.totalHeight();
            } else if (child->type == LayoutBox::BoxType::INLINE) {
                layoutInline(child, containingWidth);
                childY += child->model.totalHeight();
            }
        }
        
        // 设置父元素高度
        if (box->model.height == LayoutAuto) {
            box->model.height = childY;
        }
    }
    
private:
    void layoutBlock(LayoutBox* box, double containingWidth) {
        // 计算宽度
        double width = computeWidth(box, containingWidth);
        box->model.width = width;
        
        // 计算位置
        double x = box->model.marginLeft + box->model.borderLeft + 
                   box->model.paddingLeft;
        double y = box->model.marginTop + box->model.borderTop + 
                   box->model.paddingTop;
        
        // 布局子元素
        for (LayoutBox* child : box->children) {
            layout(child, width);
        }
        
        // 计算高度
        computeHeight(box);
    }
    
    double computeWidth(LayoutBox* box, double containingWidth) {
        CSSComputedStyle& style = box->style;
        double width = 0;
        
        std::string widthValue = style.getProperty("width");
        if (widthValue == "auto") {
            // 计算可用宽度
            double margins = box->model.marginLeft + box->model.marginRight;
            double borders = box->model.borderLeft + box->model.borderRight;
            double paddings = box->model.paddingLeft + box->model.paddingRight;
            width = containingWidth - margins - borders - paddings;
        } else {
            width = parseLength(widthValue, containingWidth);
        }
        
        return width;
    }
};
```

## 三、Flexbox布局

```cpp
class FlexLayout {
public:
    void layout(LayoutBox* container) {
        // 确定主轴和交叉轴
        FlexDirection direction = parseFlexDirection(
            container->style.getProperty("flex-direction"));
        MainAxis mainAxis = (direction == FlexDirection::ROW || 
                            direction == FlexDirection::ROW_REVERSE) 
                            ? MainAxis::HORIZONTAL : MainAxis::VERTICAL;
        
        // 1. 收集Flex项目
        collectFlexItems(container);
        
        // 2. 确定主轴尺寸
        determineMainAxisSize(container);
        
        // 3. 确定交叉轴尺寸
        determineCrossAxisSize(container);
        
        // 4. 布局项目
        layoutFlexItems(container, mainAxis);
    }
    
private:
    void layoutFlexItems(LayoutBox* container, MainAxis mainAxis) {
        double mainPos = 0;
        std::string justifyContent = 
            container->style.getProperty("justify-content");
        
        for (LayoutBox* item : flexItems) {
            double flexGrow = parseNumber(item->style.getProperty("flex-grow"));
            double flexShrink = parseNumber(item->style.getProperty("flex-shrink"));
            double flexBasis = parseLength(item->style.getProperty("flex-basis"), 0);
            
            // 计算项目尺寸
            double itemMainSize = flexBasis;
            
            // 放置项目
            if (mainAxis == MainAxis::HORIZONTAL) {
                item->model.marginLeft = mainPos;
                item->model.width = itemMainSize;
                mainPos += itemMainSize + 
                           item->model.marginLeft + item->model.marginRight;
            }
        }
    }
};
```
