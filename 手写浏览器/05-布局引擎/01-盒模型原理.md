## 一、盒模型结构

```
┌─────────────────────────────────────────┐
│                  Margin                  │
│  ┌─────────────────────────────────────┐│
│  │               Border                 ││
│  │  ┌─────────────────────────────────┐││
│  │  │            Padding              │││
│  │  │  ┌─────────────────────────────┐│││
│  │  │  │                           ││││
│  │  │  │         Content           ││││
│  │  │  │                           ││││
│  │  │  └─────────────────────────────┘│││
│  │  └─────────────────────────────────┘││
│  └─────────────────────────────────────┘│
└─────────────────────────────────────────┘

盒子尺寸：
  Content Box：width/height
  Padding Box：padding
  Border Box：border
  Margin Box：margin
  
box-sizing：
  content-box（默认）：width = content
  border-box：width = content + padding + border
```

## 二、C++实现

```cpp
struct BoxModel {
    // 尺寸
    double width = 0;
    double height = 0;
    
    // Margin
    double marginTop = 0;
    double marginRight = 0;
    double marginBottom = 0;
    double marginLeft = 0;
    
    // Border
    double borderTop = 0;
    double borderRight = 0;
    double borderBottom = 0;
    double borderLeft = 0;
    
    // Padding
    double paddingTop = 0;
    double paddingRight = 0;
    double paddingBottom = 0;
    double paddingLeft = 0;
    
    // 计算属性
    double totalWidth() const {
        return marginLeft + borderLeft + paddingLeft + width + 
               paddingRight + borderRight + marginRight;
    }
    
    double totalHeight() const {
        return marginTop + borderTop + paddingTop + height + 
               paddingBottom + borderBottom + marginBottom;
    }
    
    double contentWidth() const { return width; }
    double contentHeight() const { return height; }
};

class LayoutBox {
public:
    enum class BoxType {
        BLOCK,
        INLINE,
        INLINE_BLOCK,
        ANONYMOUS
    };
    
    BoxType type;
    BoxModel model;
    CSSComputedStyle style;
    LayoutBox* parent = nullptr;
    std::vector<LayoutBox*> children;
    
    Rect getContentRect() const {
        return {
            model.marginLeft + model.borderLeft + model.paddingLeft,
            model.marginTop + model.borderTop + model.paddingTop,
            model.width,
            model.height
        };
    }
    
    Rect getBorderRect() const {
        return {
            model.marginLeft,
            model.marginTop,
            model.width + model.paddingLeft + model.paddingRight + 
                model.borderLeft + model.borderRight,
            model.height + model.paddingTop + model.paddingBottom + 
                model.borderTop + model.borderBottom
        };
    }
};
```
