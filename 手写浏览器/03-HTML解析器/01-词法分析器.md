## 一、HTML词法规则

```
标签类型：
  开始标签：<div>
  结束标签：</div>
  自闭合标签：<img />、<br />、<hr />
  DOCTYPE：<!DOCTYPE html>

属性：
  <div id="main" class="container" disabled>
  
文本内容：
  <p>这是文本内容</p>

注释：
  <!-- 注释内容 -->

CDATA：
  <![CDATA[ 不被解析的内容 ]]>
```

---

## 二、Token类型

```cpp
enum class HTMLTokenType {
    DOCTYPE,
    StartTag,
    EndTag,
    SelfClosingTag,
    Character,
    Comment,
    EOF
};

struct HTMLToken {
    HTMLTokenType type;
    std::string tagName;
    std::map<std::string, std::string> attributes;
    std::string data;
    bool selfClosing;
};
```

## 三、C++实现

```cpp
class HTMLTokenizer {
private:
    std::string input;
    size_t pos = 0;
    int line = 1;
    int column = 1;
    
public:
    std::vector<HTMLToken> tokenize() {
        std::vector<HTMLToken> tokens;
        
        while (pos < input.size()) {
            if (input[pos] == '<') {
                HTMLToken token = parseTag();
                tokens.push_back(token);
            } else {
                HTMLToken token = parseText();
                tokens.push_back(token);
            }
        }
        
        tokens.push_back({HTMLTokenType::EOF});
        return tokens;
    }
    
private:
    HTMLToken parseTag() {
        HTMLToken token;
        pos++; // skip '<'
        
        if (input[pos] == '!') {
            pos++;
            if (input.substr(pos, 2) == "--") {
                return parseComment();
            }
            return parseDoctype();
        }
        
        if (input[pos] == '/') {
            pos++;
            token.type = HTMLTokenType::EndTag;
        } else {
            token.type = HTMLTokenType::StartTag;
        }
        
        // 标签名
        token.tagName = parseTagName();
        
        // 属性
        while (pos < input.size() && input[pos] != '>' && 
               input[pos] != '/') {
            skipWhitespace();
            if (input[pos] == '>' || input[pos] == '/') break;
            
            std::string attrName = parseTagName();
            std::string attrValue = "";
            
            if (input[pos] == '=') {
                pos++;
                attrValue = parseAttributeValue();
            }
            
            token.attributes[attrName] = attrValue;
        }
        
        // 自闭合
        if (input[pos] == '/') {
            pos++;
            token.selfClosing = true;
        }
        
        if (input[pos] == '>') pos++;
        
        return token;
    }
    
    HTMLToken parseText() {
        HTMLToken token;
        token.type = HTMLTokenType::Character;
        
        std::ostringstream oss;
        while (pos < input.size() && input[pos] != '<') {
            oss << input[pos++];
        }
        token.data = oss.str();
        
        return token;
    }
    
    HTMLToken parseComment() {
        HTMLToken token;
        token.type = HTMLTokenType::Comment;
        pos += 2; // skip --
        
        size_t end = input.find("-->", pos);
        if (end != std::string::npos) {
            token.data = input.substr(pos, end - pos);
            pos = end + 3;
        }
        
        return token;
    }
};
```
