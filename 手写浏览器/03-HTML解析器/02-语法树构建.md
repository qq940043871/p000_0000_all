## 二、节点结构

```cpp
class HTMLNode {
public:
    enum class NodeType {
        DOCUMENT,
        ELEMENT,
        TEXT,
        COMMENT,
        DOCUMENT_TYPE
    };
    
    NodeType type;
    std::string tagName;
    std::string textContent;
    std::map<std::string, std::string> attributes;
    std::vector<HTMLNode*> children;
    HTMLNode* parent = nullptr;
    
    std::string getAttribute(const std::string& name) {
        auto it = attributes.find(name);
        return it != attributes.end() ? it->second : "";
    }
    
    bool hasAttribute(const std::string& name) {
        return attributes.find(name) != attributes.end();
    }
    
    HTMLNode* insertBefore(HTMLNode* newChild, HTMLNode* refChild) {
        // 插入节点
    }
    
    HTMLNode* appendChild(HTMLNode* child) {
        child->parent = this;
        children.push_back(child);
        return child;
    }
};
```

## 三、树构建器

```cpp
class HTMLTreeBuilder {
private:
    HTMLNode* document;
    HTMLNode* currentNode;
    std::vector<HTMLNode*> openElements;
    std::stack<std::string> formattingElements;
    
    // 作用域标签
    std::set<std::string> scopingTags = {
        "applet", "caption", "html", "marquee",
        "object", "table", "td", "th", "template"
    };
    
    // 自闭合标签
    std::set<std::string> voidTags = {
        "area", "base", "br", "col", "embed",
        "hr", "img", "input", "link", "meta",
        "param", "source", "track", "wbr"
    };
    
public:
    HTMLNode* build(std::vector<HTMLToken>& tokens) {
        document = new HTMLNode();
        document->type = HTMLNode::NodeType::DOCUMENT;
        currentNode = document;
        
        for (auto& token : tokens) {
            processToken(token);
        }
        
        return document;
    }
    
private:
    void processToken(HTMLToken& token) {
        switch (token.type) {
            case HTMLTokenType::StartTag:
                handleStartTag(token);
                break;
            case HTMLTokenType::EndTag:
                handleEndTag(token);
                break;
            case HTMLTokenType::Character:
                handleCharacter(token);
                break;
            case HTMLTokenType::Comment:
                handleComment(token);
                break;
            default:
                break;
        }
    }
    
    void handleStartTag(HTMLToken& token) {
        HTMLNode* node = new HTMLNode();
        node->type = HTMLNode::NodeType::ELEMENT;
        node->tagName = token.tagName;
        node->attributes = token.attributes;
        
        if (token.selfClosing || voidTags.count(token.tagName)) {
            currentNode->appendChild(node);
        } else {
            currentNode->appendChild(node);
            currentNode = node;
            openElements.push_back(node);
        }
    }
    
    void handleEndTag(HTMLToken& token) {
        if (token.tagName != currentNode->tagName) {
            // 处理标签不匹配
        }
        currentNode = currentNode->parent;
    }
    
    void handleCharacter(HTMLToken& token) {
        HTMLNode* text = new HTMLNode();
        text->type = HTMLNode::NodeType::TEXT;
        text->textContent = token.data;
        currentNode->appendChild(text);
    }
};
```
