## 一、断点调试

```cpp
class Debugger {
private:
    std::map<int, Breakpoint> breakpoints;
    bool paused = false;
    std::vector<StackFrame> callStack;
    
public:
    void setBreakpoint(const std::string& scriptId, int line) {
        Breakpoint bp;
        bp.scriptId = scriptId;
        bp.line = line;
        bp.condition = "";
        
        breakpoints[generateId()] = bp;
    }
    
    void removeBreakpoint(int id) {
        breakpoints.erase(id);
    }
    
    void onPause(const std::string& reason, 
                 const std::vector<StackFrame>& stack) {
        paused = true;
        callStack = stack;
        
        // 通知DevTools
        Json::Value params;
        params["callFrames"] = serializeStackFrames(stack);
        params["reason"] = reason;
        devToolsProtocol.sendEvent("Debugger.paused", params);
    }
    
    void stepOver() {
        // 执行到下一行
        // ...
    }
    
    void stepInto() {
        // 进入函数
        // ...
    }
    
    void stepOut() {
        // 执行到函数返回
        // ...
    }
    
    void resume() {
        paused = false;
        // 继续执行
    }
};
```
