## 一、性能指标

```
Core Web Vitals：
  - LCP（Largest Contentful Paint）：最大内容绘制 < 2.5s
  - FID（First Input Delay）：首次输入延迟 < 100ms
  - CLS（Cumulative Layout Shift）：累计布局偏移 < 0.1

其他指标：
  - TTFB（Time To First Byte）：首字节时间
  - FP（First Paint）：首次绘制
  - FCP（First Contentful Paint）：首次内容绘制
  - TTI（Time To Interactive）：可交互时间
```

## 二、性能追踪

```cpp
class PerformanceTracker {
public:
    void recordPaint(const std::string& name, double timestamp) {
        PerformanceEntry entry;
        entry.name = name;
        entry.entryType = "paint";
        entry.startTime = timestamp;
        entry.duration = 0;
        
        entries.push_back(entry);
    }
    
    void recordLayout(double startTime, double endTime) {
        PerformanceEntry entry;
        entry.name = "Layout";
        entry.entryType = "layout";
        entry.startTime = startTime;
        entry.duration = endTime - startTime;
        
        entries.push_back(entry);
    }
    
    void recordLongTask(double startTime, double endTime) {
        if (endTime - startTime > 50) { // > 50ms
            PerformanceEntry entry;
            entry.name = "longtask";
            entry.entryType = "longtask";
            entry.startTime = startTime;
            entry.duration = endTime - startTime;
            
            entries.push_back(entry);
        }
    }
    
    Json::Value generateTrace() {
        Json::Value trace;
        trace["traceEvents"] = serializeEntries();
        return trace;
    }
};
```
