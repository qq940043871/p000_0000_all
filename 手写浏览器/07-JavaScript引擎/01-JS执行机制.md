## 一、JS引擎架构

```
JS引擎组件：
  1. Parser（解析器）
     - 词法分析
     - 语法分析
     - 生成AST
  
  2. Compiler（编译器）
     - AST → Bytecode
     - 优化编译
  
  3. Interpreter（解释器）
     - 执行Bytecode
  
  4. Garbage Collector（垃圾回收）
     - 内存管理
```

## 二、简化JS引擎

```cpp
class JSEngine {
public:
    JSEngine() {
        globalContext = new JSContext();
        gc = new GarbageCollector();
    }
    
    JSValue* evaluate(const std::string& code) {
        // 1. 解析
        AST* ast = parser.parse(code);
        
        // 2. 编译
        Bytecode* bytecode = compiler.compile(ast);
        
        // 3. 执行
        JSValue* result = interpreter.execute(bytecode, globalContext);
        
        // 4. 垃圾回收
        if (gc->shouldCollect()) {
            gc->collect();
        }
        
        return result;
    }
    
private:
    JSParser* parser;
    JSCompiler* compiler;
    JSInterpreter* interpreter;
    GarbageCollector* gc;
    JSContext* globalContext;
};

class JSContext {
public:
    std::map<std::string, JSValue*> variables;
    JSContext* parent = nullptr;
    
    JSValue* get(const std::string& name) {
        auto it = variables.find(name);
        if (it != variables.end()) return it->second;
        if (parent) return parent->get(name);
        return nullptr;
    }
    
    void set(const std::string& name, JSValue* value) {
        variables[name] = value;
    }
};
```

## 三、内置对象

```cpp
class JSObject : public JSValue {
public:
    std::map<std::string, JSValue*> properties;
    JSFunction* prototype = nullptr;
    
    JSValue* get(const std::string& name) override {
        auto it = properties.find(name);
        if (it != properties.end()) return it->second;
        if (prototype) return prototype->get(name);
        return JSUndefined::instance();
    }
};

class JSString : public JSValue {
public:
    std::string value;
};

class JSNumber : public JSValue {
public:
    double value;
};

class JSArray : public JSObject {
public:
    std::vector<JSValue*> elements;
};
```
