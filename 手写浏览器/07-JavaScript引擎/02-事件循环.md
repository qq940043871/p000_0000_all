## 一、事件循环模型

```
事件循环：
  while (true) {
    // 1. 执行所有同步任务
    executeTaskQueue();
    
    // 2. 执行微任务
    executeMicrotaskQueue();
    
    // 3. 执行宏任务
    executeMacrotaskQueue();
    
    // 4. 渲染（如需要）
    if (needsRendering) {
        requestAnimationFrame();
        updateRendering();
    }
  }

微任务：
  - Promise.then/catch/finally
  - queueMicrotask()
  - MutationObserver

宏任务：
  - setTimeout
  - setInterval
  - I/O操作
  - UI渲染
  - requestAnimationFrame
```

## 二、C++实现

```cpp
class EventLoop {
private:
    std::queue<std::function<void()>> taskQueue;
    std::queue<std::function<void()>> microtaskQueue;
    std::queue<std::function<void()>> animationCallbacks;
    bool running = false;
    
public:
    void run() {
        running = true;
        
        while (running) {
            // 1. 处理任务队列
            processTasks();
            
            // 2. 处理微任务
            processMicrotasks();
            
            // 3. 处理动画回调
            processAnimationCallbacks();
            
            // 4. 等待下一个任务
            if (taskQueue.empty() && microtaskQueue.empty()) {
                waitForEvents();
            }
        }
    }
    
    void processTasks() {
        while (!taskQueue.empty()) {
            std::function<void()> task = taskQueue.front();
            taskQueue.pop();
            task();
        }
    }
    
    void processMicrotasks() {
        while (!microtaskQueue.empty()) {
            std::function<void()> task = microtaskQueue.front();
            microtaskQueue.pop();
            task();
        }
    }
    
    void enqueueTask(std::function<void()> task) {
        taskQueue.push(task);
    }
    
    void enqueueMicrotask(std::function<void()> task) {
        microtaskQueue.push(task);
    }
};

class Timer {
public:
    static void setTimeout(std::function<void()> callback, int delay) {
        uint32_t id = nextId++;
        timers[id] = {callback, delay, getCurrentTime() + delay};
        eventLoop.enqueueTask([id]() {
            if (timers[id].callback) {
                timers[id].callback();
            }
            timers.erase(id);
        });
    }
    
    static void setInterval(std::function<void()> callback, int interval) {
        uint32_t id = nextId++;
        timers[id] = {callback, interval, getCurrentTime() + interval, true};
    }
};
```
