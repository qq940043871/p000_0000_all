## 一、MIME类型

```
常见MIME类型：
  文本：
    text/html              .html
    text/css               .css
    text/plain             .txt
    text/javascript         .js
    application/json        .json
    application/xml         .xml
  
  图片：
    image/png              .png
    image/jpeg             .jpg/.jpeg
    image/gif              .gif
    image/webp             .webp
    image/svg+xml           .svg
  
  视频：
    video/mp4              .mp4
    video/webm             .webm
  
  音频：
    audio/mpeg             .mp3
    audio/ogg              .ogg
  
  字体：
    font/woff              .woff
    font/woff2             .woff2
  
  其他：
    application/octet-stream
```

## 二、资源类型检测

```cpp
class ResourceTypeDetector {
private:
    std::map<std::string, std::string> mimeTypes = {
        {".html", "text/html"},
        {".htm", "text/html"},
        {".css", "text/css"},
        {".js", "text/javascript"},
        {".json", "application/json"},
        {".xml", "application/xml"},
        {".png", "image/png"},
        {".jpg", "image/jpeg"},
        {".jpeg", "image/jpeg"},
        {".gif", "image/gif"},
        {".webp", "image/webp"},
        {".svg", "image/svg+xml"},
        {".ico", "image/x-icon"},
        {".woff", "font/woff"},
        {".woff2", "font/woff2"},
        {".ttf", "font/ttf"},
        {".mp4", "video/mp4"},
        {".webm", "video/webm"},
        {".mp3", "audio/mpeg"},
    };
    
public:
    std::string detectFromExtension(const std::string& path) {
        size_t dotPos = path.rfind('.');
        if (dotPos != std::string::npos) {
            std::string ext = path.substr(dotPos);
            auto it = mimeTypes.find(ext);
            if (it != mimeTypes.end()) {
                return it->second;
            }
        }
        return "application/octet-stream";
    }
    
    std::string detectFromContent(const std::string& content,
                                  const std::string& mimeHint) {
        // 检查Content-Type头
        if (!mimeHint.empty() && mimeHint != "application/octet-stream") {
            return mimeHint;
        }
        
        // 检查文件内容特征
        if (content.size() >= 2) {
            // PNG: 89 50 4E 47
            if (content[0] == '\x89' && content[1] == 'P') {
                return "image/png";
            }
            // JPEG: FF D8 FF
            if ((unsigned char)content[0] == 0xFF && 
                (unsigned char)content[1] == 0xD8) {
                return "image/jpeg";
            }
            // GIF: 47 49 46 38
            if (content[0] == 'G' && content[1] == 'I' && 
                content[2] == 'F' && content[3] == '8') {
                return "image/gif";
            }
            // PDF: 25 50 44 46
            if (content[0] == '%' && content[1] == 'P' && 
                content[2] == 'D' && content[3] == 'F') {
                return "application/pdf";
            }
        }
        
        return detectFromExtension("");
    }
};
```
