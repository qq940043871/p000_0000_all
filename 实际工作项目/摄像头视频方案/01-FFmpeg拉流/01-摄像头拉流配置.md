## 一、RTSP摄像头连接

### 常见摄像头RTSP地址

```bash
# 海康威视
rtsp://admin:password@192.168.1.64:554/Streaming/Channels/101
# 子码流: /Streaming/Channels/102

# 大华
rtsp://admin:password@192.168.1.64:554/cam/realmonitor?channel=1&subtype=0
# 子码流: subtype=1

# 宇视
rtsp://admin:password@192.168.1.64:554/unicast/c1/s0/live

# 华为
rtsp://admin:password@192.168.1.64:554/live/av0/channel/0

# Onvif发现设备
# 使用onvif_device_manager工具扫描
```

### FFmpeg拉流参数

```bash
# 基本拉流
ffmpeg -rtsp_transport tcp -i rtsp://camera/live -c copy output.mp4

# 参数说明
-rtsp_transport tcp      # TCP传输（稳定，推荐）
-rtsp_transport udp      # UDP传输（低延迟，可能丢包）
-rtsp_transport http     # HTTP隧道（穿透防火墙）
```

## 二、拉流录制脚本

### 基础录制

```bash
#!/bin/bash
# record_camera.sh

# 配置
CAMERA_IP="192.168.1.100"
CAMERA_USER="admin"
CAMERA_PASS="admin123"
STREAM_URL="rtsp://${CAMERA_USER}:${CAMERA_PASS}@${CAMERA_IP}:554/Streaming/Channels/101"
OUTPUT_DIR="/data/recordings"
DATE=$(date +%Y%m%d)

# 创建目录
mkdir -p ${OUTPUT_DIR}/${DATE}

# 开始录制（每小时一个文件）
ffmpeg -rtsp_transport tcp \
  -i "${STREAM_URL}" \
  -c:v copy -c:a aac \
  -f hls \
  -hls_time 3600 \
  -hls_list_size 0 \
  -hls_segment_filename "${OUTPUT_DIR}/${DATE}/%H_%M_%S.ts" \
  "${OUTPUT_DIR}/${DATE}/playlist.m3u8"
```

### 多摄像头录制

```bash
#!/bin/bash
# record_multi.sh

CAMERAS=(
  "rtsp://admin:admin@192.168.1.100:554/stream1"
  "rtsp://admin:admin@192.168.1.101:554/stream1"
  "rtsp://admin:admin@192.168.1.102:554/stream1"
)

OUTPUT_DIR="/data/recordings"
DATE=$(date +%Y%m%d)

mkdir -p ${OUTPUT_DIR}/${DATE}

for i in "${!CAMERAS[@]}"; do
  CAMERA="${CAMERAS[$i]}"
  mkdir -p "${OUTPUT_DIR}/${DATE}/camera_$i"
  
  ffmpeg -rtsp_transport tcp \
    -i "${CAMERA}" \
    -c:v copy -c:a aac \
    -f hls \
    -hls_time 3600 \
    -hls_list_size 0 \
    -strftime 1 \
    "${OUTPUT_DIR}/${DATE}/camera_$i/%H%M%S.m3u8" \
    >/dev/null 2>&1 &
    
  echo "摄像头 $i 录制中..."
done

echo "所有摄像头录制中，按Ctrl+C停止"
wait
```

## 三、Python拉流（OpenCV）

```python
import cv2
import subprocess
import time

class CameraRecorder:
    def __init__(self, rtsp_url, output_dir, segment_time=300):
        self.rtsp_url = rtsp_url
        self.output_dir = output_dir
        self.segment_time = segment_time
        self.recording = False
        
    def start_recording(self):
        """使用FFmpeg录制"""
        import os
        os.makedirs(self.output_dir, exist_ok=True)
        
        date_str = time.strftime("%Y%m%d")
        output_file = f"{self.output_dir}/{date_str}_%Y%m%d_%H%M%S.ts"
        
        cmd = [
            'ffmpeg',
            '-rtsp_transport', 'tcp',
            '-i', self.rtsp_url,
            '-c:v', 'copy',
            '-c:a', 'aac',
            '-f', 'segment',
            '-segment_time', str(self.segment_time),
            '-strftime', '1',
            output_file
        ]
        
        self.process = subprocess.Popen(cmd)
        self.recording = True
        print(f"开始录制: {self.rtsp_url}")
        
    def stop_recording(self):
        if self.recording:
            self.process.terminate()
            self.recording = False
            print("停止录制")
            
    def get_frame(self):
        """获取当前帧"""
        cap = cv2.VideoCapture(self.rtsp_url, cv2.CAP_FFMPEG)
        ret, frame = cap.read()
        cap.release()
        return ret, frame

# 使用
camera = CameraRecorder(
    rtsp_url="rtsp://admin:admin@192.168.1.100:554/stream1",
    output_dir="/data/recordings"
)

camera.start_recording()

# 录制1小时后停止
time.sleep(3600)
camera.stop_recording()
```

## 四、拉流质量优化

```bash
# 低延迟模式
ffmpeg -rtsp_transport tcp \
  -fflags nobuffer -flags low_delay \
  -i rtsp://camera/live \
  -analyzeduration 100000 \
  -probesize 1000000 \
  -fflags +genpts \
  -c:v copy \
  -f mpegts udp://127.0.0.1:8888

# 可靠性模式（网络不稳定时）
ffmpeg -rtsp_transport tcp \
  -i rtsp://camera/live \
  -reconnect 1 \
  -reconnect_streamed 1 \
  -reconnect_delay_max 5 \
  -c:v copy \
  -f mpegts rtmp://zlm/live/stream
```

## 五、常见问题

```bash
# 问题1: 认证失败
# 检查用户名密码是否正确
# 尝试URL编码密码中的特殊字符

# 问题2: 连接超时
# 增加超时时间
ffmpeg -timeout 10000000 -i rtsp://camera/live

# 问题3: 花屏/卡顿
# 可能是网络问题，尝试降低码率
# 或使用UDP但设置缓冲区
ffmpeg -rtsp_transport udp \
  -buffer_size 1024000 \
  -i rtsp://camera/live

# 问题4: 长时间运行内存泄漏
# 使用fork方式启动，自动恢复
ffmpeg -rtsp_transport tcp \
  -i rtsp://camera/live \
  -c:v copy \
  -f mpegts rtmp://zlm/live/stream \
  -watchvn 1 2>&1 | logger &
```
