## 一、录制方案

### 方案对比

| 方案 | 格式 | 优点 | 缺点 |

| FFmpeg录制 | MP4/TS | 灵活控制 | 需要管理文件 |

## 二、ZLM API录制

### 启用录制

```ini
# config.ini配置
[record]
# 开启录制
enable=1
# 录制文件保存目录
path=/data/record
# 应用名支持通配符
app=.*;
# 流名支持通配符
stream=.*;
```

### HTTP API控制录制

```bash
# ZLM默认API密钥
SECRET="035c73f7-186a-4a1e-8e25-8e3d3d5e5e5e"

# 开始录制
curl -X POST "http://localhost:9000/index/api/startRecord" \
  -d "secret=${SECRET}&vhost=__defaultVhost__&app=live&stream=stream&MP4=true"

# 停止录制
curl -X POST "http://localhost:9000/index/api/stopRecord" \
  -d "secret=${SECRET}&vhost=__defaultVhost__&app=live&stream=stream"

# 查询录制状态
curl "http://localhost:9000/index/api/getRecordStatus?secret=${SECRET}"
```

### Python录制控制

```python
import requests
import time

class ZLMRecorder:
    def __init__(self, api_url="http://localhost:9000", secret="035c73f7-186a-4a1e-8e25-8e3d3d5e5e5e"):
        self.api_url = api_url
        self.secret = secret
        
    def start_record(self, app="live", stream="stream"):
        """开始录制"""
        url = f"{self.api_url}/index/api/startRecord"
        params = {
            "secret": self.secret,
            "vhost": "__defaultVhost__",
            "app": app,
            "stream": stream,
            "MP4": True,
            "HLS": False,
            "MP4_max_second": 3600  # 每个文件最大时长
        }
        resp = requests.post(url, params=params)
        return resp.json()
    
    def stop_record(self, app="live", stream="stream"):
        """停止录制"""
        url = f"{self.api_url}/index/api/stopRecord"
        params = {
            "secret": self.secret,
            "vhost": "__defaultVhost__",
            "app": app,
            "stream": stream
        }
        resp = requests.post(url, params=params)
        return resp.json()
    
    def get_record_status(self):
        """获取录制状态"""
        url = f"{self.api_url}/index/api/getRecordStatus"
        params = {"secret": self.secret}
        resp = requests.get(url, params=params)
        return resp.json()
    
    def get_media_list(self):
        """获取在线流列表"""
        url = f"{self.api_url}/index/api/getMediaList"
        params = {"secret": self.secret}
        resp = requests.get(url, params=params)
        return resp.json()

# 使用
recorder = ZLMRecorder()
recorder.start_record(app="live", stream="camera1")

# 录制1小时
time.sleep(3600)

recorder.stop_record(app="live", stream="camera1")
```

## 三、MinIO存储集成

### 上传录制文件

```python
from minio import Minio
import os
import datetime

class MinioUploader:
    def __init__(self, endpoint="localhost:9000", access_key="minioadmin", 
                 secret_key="minioadmin123"):
        self.client = Minio(
            endpoint,
            access_key=access_key,
            secret_key=secret_key,
            secure=False
        )
        self.bucket = "recordings"
        
    def upload_file(self, file_path, object_name=None):
        """上传录制文件"""
        # 确保bucket存在
        if not self.client.bucket_exists(self.bucket):
            self.client.make_bucket(self.bucket)
        
        # 生成对象名
        if not object_name:
            date = datetime.datetime.now().strftime("%Y/%m/%d")
            filename = os.path.basename(file_path)
            object_name = f"{date}/{filename}"
        
        # 上传
        self.client.fput_object(
            self.bucket,
            object_name,
            file_path
        )
        
        print(f"上传成功: {object_name}")
        return object_name
    
    def get_playback_url(self, object_name, expiry=3600):
        """获取回放URL"""
        url = self.client.get_presigned_url(
            "GET",
            self.bucket,
            object_name,
            expires=datetime.timedelta(seconds=expiry)
        )
        return url

# 使用
uploader = MinioUploader()

# 上传录制文件
uploader.upload_file("/data/record/20260329_143000.mp4")

# 获取回放URL（有效期1小时）
playback_url = uploader.get_playback_url("2026/03/29/record_143000.mp4")
print(f"回放地址: {playback_url}")
```

### 定时同步脚本

```bash
#!/bin/bash
# sync_to_minio.sh

# 本地录制目录
RECORD_DIR="/data/record"
# MinIO上传目录
MINIO_DIR="/data/minio_upload"
# 上传到MinIO的bucket
BUCKET="recordings"

mkdir -p ${MINIO_DIR}

while true; do
    # 查找最近1小时的新文件
    for file in $(find ${RECORD_DIR} -name "*.mp4" -mmin -60 -type f); do
        filename=$(basename "$file")
        
        # 跳过已经在上传队列的
        if [ -f "${MINIO_DIR}/${filename}" ]; then
            continue
        fi
        
        # 移动到上传队列
        mv "$file" "${MINIO_DIR}/${filename}"
        echo "准备上传: ${filename}"
        
        # 后台上传
        mc cp "${MINIO_DIR}/${filename}" "myminio/${BUCKET}/" &
    done
    
    sleep 300  # 每5分钟检查一次
done
```

## 四、回放播放器

### 支持的格式

| 格式 | 协议 | 延迟 | 兼容性 |
|------|------|------|--------|
| MP4 | HTTP | 无 | 最好 |
| HLS | HTTP | 5-30s | 所有浏览器 |
| FLV | HTTP | 1-3s | 需flv.js |
| DASH | HTTP | 3-10s | 需dash.js |

### 完整回放页面

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="utf-8">
    <title>录像回放</title>
    <style>
        body { font-family: Arial; max-width: 1200px; margin: 0 auto; padding: 20px; }
        .video-container { display: flex; flex-wrap: wrap; gap: 20px; }
        .video-item { border: 1px solid #ddd; padding: 10px; border-radius: 8px; }
        video { max-width: 400px; }
        .time-list { margin-top: 10px; }
        .time-item { padding: 5px; cursor: pointer; }
        .time-item:hover { background: #f0f0f0; }
    </style>
</head>
<body>
    <h1>录像回放</h1>
    <div id="recordings"></div>
    
    <script src="https://cdn.jsdelivr.net/npm/hls.js@1.4.12/dist/hls.min.js"></script>
    
    <script>
        // 录像列表（实际从API获取）
        const recordings = [
            { time: '2026-03-29 10:00', url: 'http://localhost:8080/record/20260329_100000.mp4' },
            { time: '2026-03-29 11:00', url: 'http://localhost:8080/record/20260329_110000.mp4' },
        ];
        
        recordings.forEach((rec, index) => {
            const div = document.createElement('div');
            div.className = 'video-item';
            div.innerHTML = `
                <h3>${rec.time}</h3>
                <video id="video${index}" controls></video>
            `;
            document.getElementById('recordings').appendChild(div);
            
            // 使用HLS播放
            const video = document.getElementById(`video${index}`);
            if (rec.url.endsWith('.m3u8')) {
                if (Hls.isSupported()) {
                    const hls = new Hls();
                    hls.loadSource(rec.url);
                    hls.attachMedia(video);
                    hls.on(Hls.Events.MANIFEST_PARSED, () => video.play());
                } else if (video.canPlayType('application/vnd.apple.mpegurl')) {
                    video.src = rec.url;
                }
            } else {
                video.src = rec.url;
            }
        });
    </script>
</body>
</html>
```
