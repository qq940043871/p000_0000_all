## 一、摄像头基础操作

### 1.1 打开摄像头

```python
import cv2

cap = cv2.VideoCapture(0)

if not cap.isOpened():
    print("无法打开摄像头")
    exit()

ret, frame = cap.read()

if ret:
    cv2.imshow('Camera', frame)
    cv2.waitKey(0)

cap.release()
cv2.destroyAllWindows()
```

### 1.2 实时显示

```python
import cv2

cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    
    if not ret:
        print("无法获取画面")
        break
    
    cv2.imshow('Camera', frame)
    
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

### 1.3 设置摄像头参数

```python
cap = cv2.VideoCapture(0)

cap.set(cv2.CAP_PROP_FRAME_WIDTH, 1920)
cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 1080)
cap.set(cv2.CAP_PROP_FPS, 30)
cap.set(cv2.CAP_PROP_BRIGHTNESS, 0.5)
cap.set(cv2.CAP_PROP_CONTRAST, 0.5)

width = cap.get(cv2.CAP_PROP_FRAME_WIDTH)

fps = cap.get(cv2.CAP_PROP_FPS)

print(f"分辨率: {width}x{height}, FPS: {fps}")
```

## 二、多摄像头管理

### 2.1 检测可用摄像头

```python
def find_cameras(max_test=10):
    available = []
    for i in range(max_test):
        cap = cv2.VideoCapture(i)
        if cap.isOpened():
            available.append(i)
            cap.release()
    return available

cameras = find_cameras()
print(f"可用摄像头: {cameras}")
```

### 2.2 同时显示多摄像头

```python
import cv2

cap1 = cv2.VideoCapture(0)
cap2 = cv2.VideoCapture(1)

cap1.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
cap1.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)
cap2.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
cap2.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

while True:
    ret1, frame1 = cap1.read()
    ret2, frame2 = cap2.read()
    
    if not ret1 or not ret2:
        break
    
    cv2.imshow('Camera 1', frame1)
    cv2.imshow('Camera 2', frame2)
    
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap1.release()
cap2.release()
cv2.destroyAllWindows()
```

### 2.3 多画面拼接显示

```python
import cv2
import numpy as np

cap1 = cv2.VideoCapture(0)
cap2 = cv2.VideoCapture(1)

while True:
    ret1, frame1 = cap1.read()
    ret2, frame2 = cap2.read()
    
    if not ret1 or not ret2:
        break
    
    frame1 = cv2.resize(frame1, (640, 480))
    frame2 = cv2.resize(frame2, (640, 480))
    
    combined = np.hstack((frame1, frame2))
    
    cv2.imshow('Multi-Camera', combined)
    
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap1.release()
cap2.release()
cv2.destroyAllWindows()
```

### 3.2 断线重连机制

```python
import cv2
import time

def connect_rtsp(url, max_retries=5):
    for i in range(max_retries):
        cap = cv2.VideoCapture(url, cv2.CAP_FFMPEG)
        if cap.isOpened():
            return cap
        print(f"连接失败，重试 {i+1}/{max_retries}")
        time.sleep(2)
    return None

rtsp_url = "rtsp://admin:password@192.168.1.100:554/stream1"
cap = connect_rtsp(rtsp_url)

if cap is None:
    print("无法连接RTSP流")
    exit()

while True:
    ret, frame = cap.read()
    
    if not ret:
        print("连接断开，尝试重连...")
        cap.release()
        cap = connect_rtsp(rtsp_url)
        if cap is None:
            break
        continue
    
    cv2.imshow('RTSP', frame)
    
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

### 4.3 实时滤镜

```python
while True:

    gray_bgr = cv2.cvtColor(gray, cv2.COLOR_GRAY2BGR)

    cv2.imshow('Filters', combined)
    
    if cv2.waitKey(1) & 0xFF == ord('q'):
```

### 5.3 多线程处理

```python
import cv2
import threading
from queue import Queue

class CameraThread:
    def __init__(self, src=0):
        self.cap = cv2.VideoCapture(src)
        self.q = Queue()
        self.running = True
        
    def start(self):
        threading.Thread(target=self._reader, daemon=True).start()
        return self
    
    def _reader(self):
        while self.running:
            ret, frame = self.cap.read()
            if not ret:
                break
            if not self.q.empty():
                try:
                    self.q.get_nowait()
                except:
                    pass
            self.q.put(frame)
    
    def read(self):
        return self.q.get()
    
    def stop(self):
        self.running = False
        self.cap.release()

cam = CameraThread(0).start()

while True:
    frame = cam.read()
    cv2.imshow('Threaded', frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cam.stop()
cv2.destroyAllWindows()
```
