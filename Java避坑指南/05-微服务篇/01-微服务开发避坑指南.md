## 一、Feign超时配置

### ❌ 错误示例
```yaml
feign:
  client:
    config:
      default:
        connect-timeout: 5000
        read-timeout: 5000  # 默认5秒，可能不够
```

### ✅ 正确做法
```yaml
feign:
  client:
    config:
      default:
        connect-timeout: 3000      # 连接超时
        read-timeout: 10000        # 读超时
        logger-level: basic        # 开启日志
  circuitbreaker:
    enabled: true                  # 开启熔断器
```

## 二、TCC空回滚

### ❌ 错误示例
```java
@LocalTCC
public class OrderService {
    
    @Override
    public boolean tryDecreaseStock(Long productId, Integer count) {
        return stockService.tryDecrease(productId, count);
    }
    
    @Override
    public void confirmDecreaseStock(Long productId, Integer count) {
        // 如果Try未执行直接调用Confirm，会出现空回滚
        stockService.confirmDecrease(productId, count);
    }
}
```

### ✅ 正确做法
```java
@LocalTCC
public class OrderService {
    
    @Override
    public boolean tryDecreaseStock(Long productId, Integer count) {
        // 必须检测是否已经扣减过
        Stock stock = stockMapper.selectByProductId(productId);
        if (stock.getLockedCount() >= count) {
            return false;
        }
        stockService.lockStock(productId, count);
        return true;
    }
    
    @Override
    public void confirmDecreaseStock(Long productId, Integer count) {
        stockService.confirmDecrease(productId, count);
    }
}
```

## 三、接口幂等性

### ✅ 正确做法

#### 方法1：使用Token
```java
@PostMapping("/deduct")
public Result<Void> deduct(@RequestParam Long accountId,
                         @RequestParam BigDecimal amount,
                         @RequestHeader("Idempotent-Token") String token) {
    if (!idempotentService.tryLock(token)) {
        return Result.error("请求已处理，请勿重复提交");
    }
    try {
        accountService.deduct(accountId, amount);
        return Result.success();
    } finally {
        idempotentService.unlock(token);
    }
}
```

#### 方法2：数据库唯一索引
```java
@PostMapping("/create")
public Result<Long> createOrder(@RequestBody CreateOrderRequest request) {
    Order order = new Order();
    order.setIdempotentId(request.getIdempotentId());
    try {
        orderService.insert(order);
        return Result.success(order.getId());
    } catch (DuplicateKeyException e) {
        return Result.success(orderService.findByIdempotentId(request.getIdempotentId()).getId());
    }
}
```

#### 方法3：分布式锁
```java
@PostMapping("/deduct")
public Result<Void> deduct(@RequestParam Long accountId,
                         @RequestParam String idempotentId,
                         @RequestParam BigDecimal amount) {
    String lockKey = "deduct:" + accountId + ":" + idempotentId;
    return redisLock.execute(lockKey, () -> {
        accountService.deduct(accountId, amount);
        return Result.success();
    });
}
```

## 四、服务降级与熔断

### ✅ 正确做法
```java
// Sentinel注解方式
@SentinelResource(value = "getUser",
    blockHandler = "getUserBlockHandler",
    fallback = "getUserFallback")
public User getUser(Long id) {
    return userService.getById(id);
}

// 限流处理
public User getUserBlockHandler(Long id, BlockException e) {
    return User.builder().name("系统繁忙，请稍后再试").build();
}

// 降级处理
public User getUserFallback(Long id, Throwable e) {
    return User.builder().name("服务降级").build();
}
```

## 五、分布式Session

### ❌ 错误示例
```java
// 使用本地Session，在微服务架构下会失效
HttpSession session = request.getSession();
session.setAttribute("user", user);
```

### ✅ 正确做法
```java
// 方法1：JWT Token
@PostMapping("/login")
public String login(@RequestBody LoginRequest request) {
    User user = userService.login(request);
    return JWTUtil.generateToken(user.getId());
}

// 方法2：Redis Session
@Configuration
@EnableRedisHttpSession
public class RedisSessionConfig {
}

@PostMapping("/login")
public void login(HttpServletRequest request, @RequestBody LoginRequest request) {
    User user = userService.login(request);
    request.getSession().setAttribute("user", user);  // 自动存到Redis
}
```

## 六、Nacos配置热更新

### ❌ 错误示例
```java
@Value("${app.title}")
private String title;  // 只是普通注入，不会热更新
```

### ✅ 正确做法
```java
// 方法1：@RefreshScope
@RefreshScope
@RestController
public class ConfigController {
    @Value("${app.title}")
    private String title;  // 会自动刷新
}

// 方法2：@ConfigurationProperties
@Component
@ConfigurationProperties(prefix = "app")
@Data
public class AppProperties {
    private String title;
    private String description;
}
```
